029package clojure.asm;
030
031/**
032 * Information about the input stack map frame at the "current" instruction of a method. This is
033 * implemented as a Frame subclass for a "basic block" containing only one instruction.
034 *
035 * @author Eric Bruneton
036 */
037final class CurrentFrame extends Frame {
038
039  CurrentFrame(final Label owner) {
040    super(owner);
041  }
042
043  /**
044   * Sets this CurrentFrame to the input stack map frame of the next "current" instruction, i.e. the
045   * instruction just after the given one. It is assumed that the value of this object when this
046   * method is called is the stack map frame status just before the given instruction is executed.
047   */
048  @Override
049  void execute(
050      final int opcode, final int arg, final Symbol symbolArg, final SymbolTable symbolTable) {
051    super.execute(opcode, arg, symbolArg, symbolTable);
052    Frame successor = new Frame(null);
053    merge(symbolTable, successor, 0);
054    copyFrom(successor);
055  }
056}
