034/**
035 * A parser to make a {@link ClassVisitor} visit a ClassFile structure, as defined in the Java
036 * Virtual Machine Specification (JVMS). This class parses the ClassFile content and calls the
037 * appropriate visit methods of a given {@link ClassVisitor} for each field, method and bytecode
038 * instruction encountered.
039 *
040 * @see <a href="https://docs.oracle.com/javase/specs/jvms/se9/html/jvms-4.html">JVMS 4</a>
041 * @author Eric Bruneton
042 * @author Eugene Kuleshov
043 */
044public class ClassReader {
045
046  /**
047   * A flag to skip the Code attributes. If this flag is set the Code attributes are neither parsed
048   * nor visited.
049   */
050  public static final int SKIP_CODE = 1;
051
052  /**
053   * A flag to skip the SourceFile, SourceDebugExtension, LocalVariableTable, LocalVariableTypeTable
054   * and LineNumberTable attributes. If this flag is set these attributes are neither parsed nor
055   * visited (i.e. {@link ClassVisitor#visitSource}, {@link MethodVisitor#visitLocalVariable} and
056   * {@link MethodVisitor#visitLineNumber} are not called).
057   */
058  public static final int SKIP_DEBUG = 2;
059
060  /**
061   * A flag to skip the StackMap and StackMapTable attributes. If this flag is set these attributes
062   * are neither parsed nor visited (i.e. {@link MethodVisitor#visitFrame} is not called). This flag
063   * is useful when the {@link ClassWriter#COMPUTE_FRAMES} option is used: it avoids visiting frames
064   * that will be ignored and recomputed from scratch.
065   */
066  public static final int SKIP_FRAMES = 4;
067
068  /**
069   * A flag to expand the stack map frames. By default stack map frames are visited in their
070   * original format (i.e. "expanded" for classes whose version is less than V1_6, and "compressed"
071   * for the other classes). If this flag is set, stack map frames are always visited in expanded
072   * format (this option adds a decompression/compression step in ClassReader and ClassWriter which
073   * degrades performance quite a lot).
074   */
075  public static final int EXPAND_FRAMES = 8;
076
077  /**
078   * A flag to expand the ASM specific instructions into an equivalent sequence of standard bytecode
079   * instructions. When resolving a forward jump it may happen that the signed 2 bytes offset
080   * reserved for it is not sufficient to store the bytecode offset. In this case the jump
081   * instruction is replaced with a temporary ASM specific instruction using an unsigned 2 bytes
082   * offset (see {@link Label#resolve}). This internal flag is used to re-read classes containing
083   * such instructions, in order to replace them with standard instructions. In addition, when this
084   * flag is used, goto_w and jsr_w are <i>not</i> converted into goto and jsr, to make sure that
085   * infinite loops where a goto_w is replaced with a goto in ClassReader and converted back to a
086   * goto_w in ClassWriter cannot occur.
087   */
088  static final int EXPAND_ASM_INSNS = 256;
089
090  /** The size of the temporary byte array used to read class input streams chunk by chunk. */
091  private static final int INPUT_STREAM_DATA_CHUNK_SIZE = 4096;
092
093  /**
094   * A byte array containing the JVMS ClassFile structure to be parsed. <i>The content of this array
095   * must not be modified. This field is intended for {@link Attribute} sub classes, and is normally
096   * not needed by class visitors.</i>
097   *
098   * <p>NOTE: the ClassFile structure can start at any offset within this array, i.e. it does not
099   * necessarily start at offset 0. Use {@link #getItem} and {@link #header} to get correct
100   * ClassFile element offsets within this byte array.
101   */
102  public final byte[] b;
103
104  /**
105   * The offset in bytes, in {@link #b}, of each cp_info entry of the ClassFile's constant_pool
106   * array, <i>plus one</i>. In other words, the offset of constant pool entry i is given by
107   * cpInfoOffsets[i] - 1, i.e. its cp_info's tag field is given by b[cpInfoOffsets[i] - 1].
108   */
109  private final int[] cpInfoOffsets;
110
111  /**
112   * The value of each cp_info entry of the ClassFile's constant_pool array, <i>for Constant_Utf8
113   * and Constant_Dynamic constants only</i>. The value of constant pool entry i is given by
114   * cpInfoValues[i]. This cache avoids multiple parsing of those constant pool items.
115   */
116  private final Object[] cpInfoValues;
117
118  /**
119   * The start offsets in {@link #b} of each element of the bootstrap_methods array (in the
120   * BootstrapMethods attribute).
121   *
122   * @see <a href="https://docs.oracle.com/javase/specs/jvms/se9/html/jvms-4.html#jvms-4.7.23">JVMS
123   *     4.7.23</a>
124   */
125  private final int[] bootstrapMethodOffsets;
126
127  /**
128   * A conservative estimate of the maximum length of the strings contained in the constant pool of
129   * the class.
130   */
131  private final int maxStringLength;
132
133  /** The offset in bytes, in {@link #b}, of the ClassFile's access_flags field. */
134  public final int header;

161  /**
162   * Constructs a new {@link ClassReader} object. <i>This internal constructor must not be exposed
163   * as a public API</i>.
164   *
165   * @param classFileBuffer a byte array containing the JVMS ClassFile structure to be read.
166   * @param classFileOffset the offset in byteBuffer of the first byte of the ClassFile to be read.
167   * @param checkClassVersion whether to check the class version or not.
168   */
169  ClassReader(
170      final byte[] classFileBuffer, final int classFileOffset, final boolean checkClassVersion) {
171    this.b = classFileBuffer;
172    // Check the class' major_version. This field is after the magic and minor_version fields, which
173    // use 4 and 2 bytes respectively.
174    if (checkClassVersion && readShort(classFileOffset + 6) > Opcodes.V11) {
175      throw new IllegalArgumentException(
176          "Unsupported class file major version " + readShort(classFileOffset + 6));
177    }
178    // Create the constant pool arrays. The constant_pool_count field is after the magic,
179    // minor_version and major_version fields, which use 4, 2 and 2 bytes respectively.
180    int constantPoolCount = readUnsignedShort(classFileOffset + 8);
181    cpInfoOffsets = new int[constantPoolCount];
182    cpInfoValues = new Object[constantPoolCount];
183    // Compute the offset of each constant pool entry, as well as a conservative estimate of the
184    // maximum length of the constant pool strings. The first constant pool entry is after the
185    // magic, minor_version, major_version and constant_pool_count fields, which use 4, 2, 2 and 2
186    // bytes respectively.
187    int currentCpInfoIndex = 1;
188    int currentCpInfoOffset = classFileOffset + 10;
189    int currentMaxStringLength = 0;
190    // The offset of the other entries depend on the total size of all the previous entries.
191    while (currentCpInfoIndex < constantPoolCount) {
192      cpInfoOffsets[currentCpInfoIndex++] = currentCpInfoOffset + 1;
193      int cpInfoSize;
194      switch (classFileBuffer[currentCpInfoOffset]) {
195        case Symbol.CONSTANT_FIELDREF_TAG:
196        case Symbol.CONSTANT_METHODREF_TAG:
197        case Symbol.CONSTANT_INTERFACE_METHODREF_TAG:
198        case Symbol.CONSTANT_INTEGER_TAG:
199        case Symbol.CONSTANT_FLOAT_TAG:
200        case Symbol.CONSTANT_NAME_AND_TYPE_TAG:
201        case Symbol.CONSTANT_INVOKE_DYNAMIC_TAG:
202        case Symbol.CONSTANT_DYNAMIC_TAG:
203          cpInfoSize = 5;
204          break;
205        case Symbol.CONSTANT_LONG_TAG:
206        case Symbol.CONSTANT_DOUBLE_TAG:
207          cpInfoSize = 9;
208          currentCpInfoIndex++;
209          break;
210        case Symbol.CONSTANT_UTF8_TAG:
211          cpInfoSize = 3 + readUnsignedShort(currentCpInfoOffset + 1);
212          if (cpInfoSize > currentMaxStringLength) {
213            // The size in bytes of this CONSTANT_Utf8 structure provides a conservative estimate
214            // of the length in characters of the corresponding string, and is much cheaper to
215            // compute than this exact length.
216            currentMaxStringLength = cpInfoSize;
217          }
218          break;
219        case Symbol.CONSTANT_METHOD_HANDLE_TAG:
220          cpInfoSize = 4;
221          break;
222        case Symbol.CONSTANT_CLASS_TAG:
223        case Symbol.CONSTANT_STRING_TAG:
224        case Symbol.CONSTANT_METHOD_TYPE_TAG:
225        case Symbol.CONSTANT_PACKAGE_TAG:
226        case Symbol.CONSTANT_MODULE_TAG:
227          cpInfoSize = 3;
228          break;
229        default:
230          throw new IllegalArgumentException();
231      }
232      currentCpInfoOffset += cpInfoSize;
233    }
234    this.maxStringLength = currentMaxStringLength;
235    // The Classfile's access_flags field is just after the last constant pool entry.
236    this.header = currentCpInfoOffset;
237
238    // Read the BootstrapMethods attribute, if any (only get the offset of each method).
239    int currentAttributeOffset = getFirstAttributeOffset();
240    int[] currentBootstrapMethodOffsets = null;
241    for (int i = readUnsignedShort(currentAttributeOffset - 2); i > 0; --i) {
242      // Read the attribute_info's attribute_name and attribute_length fields.
243      String attributeName = readUTF8(currentAttributeOffset, new char[maxStringLength]);
244      int attributeLength = readInt(currentAttributeOffset + 2);
245      currentAttributeOffset += 6;
246      if (Constants.BOOTSTRAP_METHODS.equals(attributeName)) {
247        // Read the num_bootstrap_methods field and create an array of this size.
248        currentBootstrapMethodOffsets = new int[readUnsignedShort(currentAttributeOffset)];
249        // Compute and store the offset of each 'bootstrap_methods' array field entry.
250        int currentBootstrapMethodOffset = currentAttributeOffset + 2;
251        for (int j = 0; j < currentBootstrapMethodOffsets.length; ++j) {
252          currentBootstrapMethodOffsets[j] = currentBootstrapMethodOffset;
253          // Skip the bootstrap_method_ref and num_bootstrap_arguments fields (2 bytes each),
254          // as well as the bootstrap_arguments array field (of size num_bootstrap_arguments * 2).
255          currentBootstrapMethodOffset +=
256              4 + readUnsignedShort(currentBootstrapMethodOffset + 2) * 2;
257        }
258      }
259      currentAttributeOffset += attributeLength;
260    }
261    this.bootstrapMethodOffsets = currentBootstrapMethodOffsets;
262  }

264  /**
265   * Constructs a new {@link ClassReader} object.
266   *
267   * @param inputStream an input stream of the JVMS ClassFile structure to be read. This input
268   *     stream must contain nothing more than the ClassFile structure itself. It is read from its
269   *     current position to its end.
270   * @throws IOException if a problem occurs during reading.
271   */
272  public ClassReader(final InputStream inputStream) throws IOException {
273    this(readStream(inputStream, false));
274  }

276  /**
277   * Constructs a new {@link ClassReader} object.
278   *
279   * @param className the fully qualified name of the class to be read. The ClassFile structure is
280   *     retrieved with the current class loader's {@link ClassLoader#getSystemResourceAsStream}.
281   * @throws IOException if an exception occurs during reading.
282   */
283  public ClassReader(final String className) throws IOException {
284    this(
285        readStream(
286            ClassLoader.getSystemResourceAsStream(className.replace('.', '/') + ".class"), true));
287  }

289  /**
290   * Reads the given input stream and returns its content as a byte array.
291   *
292   * @param inputStream an input stream.
293   * @param close true to close the input stream after reading.
294   * @return the content of the given input stream.
295   * @throws IOException if a problem occurs during reading.
296   */
297  private static byte[] readStream(final InputStream inputStream, final boolean close)
298      throws IOException {
299    if (inputStream == null) {
300      throw new IOException("Class not found");
301    }
302    try {
303      ByteArrayOutputStream outputStream = new ByteArrayOutputStream();
304      byte[] data = new byte[INPUT_STREAM_DATA_CHUNK_SIZE];
305      int bytesRead;
306      while ((bytesRead = inputStream.read(data, 0, data.length)) != -1) {
307        outputStream.write(data, 0, bytesRead);
308      }
309      outputStream.flush();
310      return outputStream.toByteArray();
311    } finally {
312      if (close) {
313        inputStream.close();
314      }
315    }
316  }

322  /**
323   * Returns the class's access flags (see {@link Opcodes}). This value may not reflect Deprecated
324   * and Synthetic flags when bytecode is before 1.5 and those flags are represented by attributes.
325   *
326   * @return the class access flags.
327   * @see ClassVisitor#visit(int, int, String, String, String, String[])
328   */
329  public int getAccess() {
330    return readUnsignedShort(header);
331  }
332
333  /**
334   * Returns the internal name of the class (see {@link Type#getInternalName()}).
335   *
336   * @return the internal class name.
337   * @see ClassVisitor#visit(int, int, String, String, String, String[])
338   */
339  public String getClassName() {
340    // this_class is just after the access_flags field (using 2 bytes).
341    return readClass(header + 2, new char[maxStringLength]);
342  }
343
344  /**
345   * Returns the internal of name of the super class (see {@link Type#getInternalName()}). For
346   * interfaces, the super class is {@link Object}.
347   *
348   * @return the internal name of the super class, or <tt>null</tt> for {@link Object} class.
349   * @see ClassVisitor#visit(int, int, String, String, String, String[])
350   */
351  public String getSuperName() {
352    // super_class is after the access_flags and this_class fields (2 bytes each).
353    return readClass(header + 4, new char[maxStringLength]);
354  }
355
356  /**
357   * Returns the internal names of the implemented interfaces (see {@link Type#getInternalName()}).
358   *
359   * @return the internal names of the directly implemented interfaces. Inherited implemented
360   *     interfaces are not returned.
361   * @see ClassVisitor#visit(int, int, String, String, String, String[])
362   */
363  public String[] getInterfaces() {
364    // interfaces_count is after the access_flags, this_class and super_class fields (2 bytes each).
365    int currentOffset = header + 6;
366    int interfacesCount = readUnsignedShort(currentOffset);
367    String[] interfaces = new String[interfacesCount];
368    if (interfacesCount > 0) {
369      char[] charBuffer = new char[maxStringLength];
370      for (int i = 0; i < interfacesCount; ++i) {
371        currentOffset += 2;
372        interfaces[i] = readClass(currentOffset, charBuffer);
373      }
374    }
375    return interfaces;
376  }

382  /**
383   * Makes the given visitor visit the JVMS ClassFile structure passed to the constructor of this
384   * {@link ClassReader}.
385   *
386   * @param classVisitor the visitor that must visit this class.
387   * @param parsingOptions the options to use to parse this class. One or more of {@link
388   *     #SKIP_CODE}, {@link #SKIP_DEBUG}, {@link #SKIP_FRAMES} or {@link #EXPAND_FRAMES}.
389   */
390  public void accept(final ClassVisitor classVisitor, final int parsingOptions) {
391    accept(classVisitor, new Attribute[0], parsingOptions);
392  }
393
394  /**
395   * Makes the given visitor visit the JVMS ClassFile structure passed to the constructor of this
396   * {@link ClassReader}.
397   *
398   * @param classVisitor the visitor that must visit this class.
399   * @param attributePrototypes prototypes of the attributes that must be parsed during the visit of
400   *     the class. Any attribute whose type is not equal to the type of one the prototypes will not
401   *     be parsed: its byte array value will be passed unchanged to the ClassWriter. <i>This may
402   *     corrupt it if this value contains references to the constant pool, or has syntactic or
403   *     semantic links with a class element that has been transformed by a class adapter between
404   *     the reader and the writer</i>.
405   * @param parsingOptions the options to use to parse this class. One or more of {@link
406   *     #SKIP_CODE}, {@link #SKIP_DEBUG}, {@link #SKIP_FRAMES} or {@link #EXPAND_FRAMES}.
407   */
408  public void accept(
409      final ClassVisitor classVisitor,
410      final Attribute[] attributePrototypes,
411      final int parsingOptions) {
412    Context context = new Context();
413    context.attributePrototypes = attributePrototypes;
414    context.parsingOptions = parsingOptions;
415    context.charBuffer = new char[maxStringLength];
416
417    // Read the access_flags, this_class, super_class, interface_count and interfaces fields.
418    char[] charBuffer = context.charBuffer;
419    int currentOffset = header;
420    int accessFlags = readUnsignedShort(currentOffset);
421    String thisClass = readClass(currentOffset + 2, charBuffer);
422    String superClass = readClass(currentOffset + 4, charBuffer);
423    String[] interfaces = new String[readUnsignedShort(currentOffset + 6)];
424    currentOffset += 8;
425    for (int i = 0; i < interfaces.length; ++i) {
426      interfaces[i] = readClass(currentOffset, charBuffer);
427      currentOffset += 2;
428    }
429
430    // Read the class attributes (the variables are ordered as in Section 4.7 of the JVMS).
431    // Attribute offsets exclude the attribute_name_index and attribute_length fields.
432    // - The offset of the InnerClasses attribute, or 0.
433    int innerClassesOffset = 0;
434    // - The offset of the EnclosingMethod attribute, or 0.
435    int enclosingMethodOffset = 0;
436    // - The string corresponding to the Signature attribute, or null.
437    String signature = null;
438    // - The string corresponding to the SourceFile attribute, or null.
439    String sourceFile = null;
440    // - The string corresponding to the SourceDebugExtension attribute, or null.
441    String sourceDebugExtension = null;
442    // - The offset of the RuntimeVisibleAnnotations attribute, or 0.
443    int runtimeVisibleAnnotationsOffset = 0;
444    // - The offset of the RuntimeInvisibleAnnotations attribute, or 0.
445    int runtimeInvisibleAnnotationsOffset = 0;
446    // - The offset of the RuntimeVisibleTypeAnnotations attribute, or 0.
447    int runtimeVisibleTypeAnnotationsOffset = 0;
448    // - The offset of the RuntimeInvisibleTypeAnnotations attribute, or 0.
449    int runtimeInvisibleTypeAnnotationsOffset = 0;
450    // - The offset of the Module attribute, or 0.
451    int moduleOffset = 0;
452    // - The offset of the ModulePackages attribute, or 0.
453    int modulePackagesOffset = 0;
454    // - The string corresponding to the ModuleMainClass attribute, or null.
455    String moduleMainClass = null;
456    // - The string corresponding to the NestHost attribute, or null.
457    String nestHostClass = null;
458    // - The offset of the NestMembers attribute, or 0.
459    int nestMembersOffset = 0;
460    // - The non standard attributes (linked with their {@link Attribute#nextAttribute} field).
461    //   This list in the <i>reverse order</i> or their order in the ClassFile structure.
462    Attribute attributes = null;
463
464    int currentAttributeOffset = getFirstAttributeOffset();
465    for (int i = readUnsignedShort(currentAttributeOffset - 2); i > 0; --i) {
466      // Read the attribute_info's attribute_name and attribute_length fields.
467      String attributeName = readUTF8(currentAttributeOffset, charBuffer);
468      int attributeLength = readInt(currentAttributeOffset + 2);
469      currentAttributeOffset += 6;
470      // The tests are sorted in decreasing frequency order (based on frequencies observed on
471      // typical classes).
472      if (Constants.SOURCE_FILE.equals(attributeName)) {
473        sourceFile = readUTF8(currentAttributeOffset, charBuffer);
474      } else if (Constants.INNER_CLASSES.equals(attributeName)) {
475        innerClassesOffset = currentAttributeOffset;
476      } else if (Constants.ENCLOSING_METHOD.equals(attributeName)) {
477        enclosingMethodOffset = currentAttributeOffset;
478      } else if (Constants.NEST_HOST.equals(attributeName)) {
479        nestHostClass = readClass(currentAttributeOffset, charBuffer);
480      } else if (Constants.NEST_MEMBERS.equals(attributeName)) {
481        nestMembersOffset = currentAttributeOffset;
482      } else if (Constants.SIGNATURE.equals(attributeName)) {
483        signature = readUTF8(currentAttributeOffset, charBuffer);
484      } else if (Constants.RUNTIME_VISIBLE_ANNOTATIONS.equals(attributeName)) {
485        runtimeVisibleAnnotationsOffset = currentAttributeOffset;
486      } else if (Constants.RUNTIME_VISIBLE_TYPE_ANNOTATIONS.equals(attributeName)) {
487        runtimeVisibleTypeAnnotationsOffset = currentAttributeOffset;
488      } else if (Constants.DEPRECATED.equals(attributeName)) {
489        accessFlags |= Opcodes.ACC_DEPRECATED;
490      } else if (Constants.SYNTHETIC.equals(attributeName)) {
491        accessFlags |= Opcodes.ACC_SYNTHETIC;
492      } else if (Constants.SOURCE_DEBUG_EXTENSION.equals(attributeName)) {
493        sourceDebugExtension =
494            readUTF(currentAttributeOffset, attributeLength, new char[attributeLength]);
495      } else if (Constants.RUNTIME_INVISIBLE_ANNOTATIONS.equals(attributeName)) {
496        runtimeInvisibleAnnotationsOffset = currentAttributeOffset;
497      } else if (Constants.RUNTIME_INVISIBLE_TYPE_ANNOTATIONS.equals(attributeName)) {
498        runtimeInvisibleTypeAnnotationsOffset = currentAttributeOffset;
499      } else if (Constants.MODULE.equals(attributeName)) {
500        moduleOffset = currentAttributeOffset;
501      } else if (Constants.MODULE_MAIN_CLASS.equals(attributeName)) {
502        moduleMainClass = readClass(currentAttributeOffset, charBuffer);
503      } else if (Constants.MODULE_PACKAGES.equals(attributeName)) {
504        modulePackagesOffset = currentAttributeOffset;
505      } else if (Constants.BOOTSTRAP_METHODS.equals(attributeName)) {
506        // This attribute is read in the constructor.
507      } else {
508        Attribute attribute =
509            readAttribute(
510                attributePrototypes,
511                attributeName,
512                currentAttributeOffset,
513                attributeLength,
514                charBuffer,
515                -1,
516                null);
517        attribute.nextAttribute = attributes;
518        attributes = attribute;
519      }
520      currentAttributeOffset += attributeLength;
521    }
522
523    // Visit the class declaration. The minor_version and major_version fields start 6 bytes before
524    // the first constant pool entry, which itself starts at cpInfoOffsets[1] - 1 (by definition).
525    classVisitor.visit(
526        readInt(cpInfoOffsets[1] - 7), accessFlags, thisClass, signature, superClass, interfaces);
527
528    // Visit the SourceFile and SourceDebugExtenstion attributes.
529    if ((parsingOptions & SKIP_DEBUG) == 0
530        && (sourceFile != null || sourceDebugExtension != null)) {
531      classVisitor.visitSource(sourceFile, sourceDebugExtension);
532    }
533
534    // Visit the Module, ModulePackages and ModuleMainClass attributes.
535    if (moduleOffset != 0) {
536      readModule(classVisitor, context, moduleOffset, modulePackagesOffset, moduleMainClass);
537    }
538
539    // Visit the NestHost attribute.
540    if (nestHostClass != null) {
541      classVisitor.visitNestHostExperimental(nestHostClass);
542    }
543
544    // Visit the EnclosingMethod attribute.
545    if (enclosingMethodOffset != 0) {
546      String className = readClass(enclosingMethodOffset, charBuffer);
547      int methodIndex = readUnsignedShort(enclosingMethodOffset + 2);
548      String name = methodIndex == 0 ? null : readUTF8(cpInfoOffsets[methodIndex], charBuffer);
549      String type = methodIndex == 0 ? null : readUTF8(cpInfoOffsets[methodIndex] + 2, charBuffer);
550      classVisitor.visitOuterClass(className, name, type);
551    }
552
553    // Visit the RuntimeVisibleAnnotations attribute.
554    if (runtimeVisibleAnnotationsOffset != 0) {
555      int numAnnotations = readUnsignedShort(runtimeVisibleAnnotationsOffset);
556      int currentAnnotationOffset = runtimeVisibleAnnotationsOffset + 2;
557      while (numAnnotations-- > 0) {
558        // Parse the type_index field.
559        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
560        currentAnnotationOffset += 2;
561        // Parse num_element_value_pairs and element_value_pairs and visit these values.
562        currentAnnotationOffset =
563            readElementValues(
564                classVisitor.visitAnnotation(annotationDescriptor, /* visible = */ true),
565                currentAnnotationOffset,
566                /* named = */ true,
567                charBuffer);
568      }
569    }
570
571    // Visit the RuntimeInvisibleAnnotations attribute.
572    if (runtimeInvisibleAnnotationsOffset != 0) {
573      int numAnnotations = readUnsignedShort(runtimeInvisibleAnnotationsOffset);
574      int currentAnnotationOffset = runtimeInvisibleAnnotationsOffset + 2;
575      while (numAnnotations-- > 0) {
576        // Parse the type_index field.
577        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
578        currentAnnotationOffset += 2;
579        // Parse num_element_value_pairs and element_value_pairs and visit these values.
580        currentAnnotationOffset =
581            readElementValues(
582                classVisitor.visitAnnotation(annotationDescriptor, /* visible = */ false),
583                currentAnnotationOffset,
584                /* named = */ true,
585                charBuffer);
586      }
587    }
588
589    // Visit the RuntimeVisibleTypeAnnotations attribute.
590    if (runtimeVisibleTypeAnnotationsOffset != 0) {
591      int numAnnotations = readUnsignedShort(runtimeVisibleTypeAnnotationsOffset);
592      int currentAnnotationOffset = runtimeVisibleTypeAnnotationsOffset + 2;
593      while (numAnnotations-- > 0) {
594        // Parse the target_type, target_info and target_path fields.
595        currentAnnotationOffset = readTypeAnnotationTarget(context, currentAnnotationOffset);
596        // Parse the type_index field.
597        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
598        currentAnnotationOffset += 2;
599        // Parse num_element_value_pairs and element_value_pairs and visit these values.
600        currentAnnotationOffset =
601            readElementValues(
602                classVisitor.visitTypeAnnotation(
603                    context.currentTypeAnnotationTarget,
604                    context.currentTypeAnnotationTargetPath,
605                    annotationDescriptor,
606                    /* visible = */ true),
607                currentAnnotationOffset,
608                /* named = */ true,
609                charBuffer);
610      }
611    }
612
613    // Visit the RuntimeInvisibleTypeAnnotations attribute.
614    if (runtimeInvisibleTypeAnnotationsOffset != 0) {
615      int numAnnotations = readUnsignedShort(runtimeInvisibleTypeAnnotationsOffset);
616      int currentAnnotationOffset = runtimeInvisibleTypeAnnotationsOffset + 2;
617      while (numAnnotations-- > 0) {
618        // Parse the target_type, target_info and target_path fields.
619        currentAnnotationOffset = readTypeAnnotationTarget(context, currentAnnotationOffset);
620        // Parse the type_index field.
621        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
622        currentAnnotationOffset += 2;
623        // Parse num_element_value_pairs and element_value_pairs and visit these values.
624        currentAnnotationOffset =
625            readElementValues(
626                classVisitor.visitTypeAnnotation(
627                    context.currentTypeAnnotationTarget,
628                    context.currentTypeAnnotationTargetPath,
629                    annotationDescriptor,
630                    /* visible = */ false),
631                currentAnnotationOffset,
632                /* named = */ true,
633                charBuffer);
634      }
635    }
636
637    // Visit the non standard attributes.
638    while (attributes != null) {
639      // Copy and reset the nextAttribute field so that it can also be used in ClassWriter.
640      Attribute nextAttribute = attributes.nextAttribute;
641      attributes.nextAttribute = null;
642      classVisitor.visitAttribute(attributes);
643      attributes = nextAttribute;
644    }
645
646    // Visit the NestedMembers attribute.
647    if (nestMembersOffset != 0) {
648      int numberOfNestMembers = readUnsignedShort(nestMembersOffset);
649      int currentNestMemberOffset = nestMembersOffset + 2;
650      while (numberOfNestMembers-- > 0) {
651        classVisitor.visitNestMemberExperimental(readClass(currentNestMemberOffset, charBuffer));
652        currentNestMemberOffset += 2;
653      }
654    }
655
656    // Visit the InnerClasses attribute.
657    if (innerClassesOffset != 0) {
658      int numberOfClasses = readUnsignedShort(innerClassesOffset);
659      int currentClassesOffset = innerClassesOffset + 2;
660      while (numberOfClasses-- > 0) {
661        classVisitor.visitInnerClass(
662            readClass(currentClassesOffset, charBuffer),
663            readClass(currentClassesOffset + 2, charBuffer),
664            readUTF8(currentClassesOffset + 4, charBuffer),
665            readUnsignedShort(currentClassesOffset + 6));
666        currentClassesOffset += 8;
667      }
668    }
669
670    // Visit the fields and methods.
671    int fieldsCount = readUnsignedShort(currentOffset);
672    currentOffset += 2;
673    while (fieldsCount-- > 0) {
674      currentOffset = readField(classVisitor, context, currentOffset);
675    }
676    int methodsCount = readUnsignedShort(currentOffset);
677    currentOffset += 2;
678    while (methodsCount-- > 0) {
679      currentOffset = readMethod(classVisitor, context, currentOffset);
680    }
681
682    // Visit the end of the class.
683    classVisitor.visitEnd();
684  }

690  /**
691   * Reads the module attribute and visit it.
692   *
693   * @param classVisitor the current class visitor
694   * @param context information about the class being parsed.
695   * @param moduleOffset the offset of the Module attribute (excluding the attribute_info's
696   *     attribute_name_index and attribute_length fields).
697   * @param modulePackagesOffset the offset of the ModulePackages attribute (excluding the
698   *     attribute_info's attribute_name_index and attribute_length fields), or 0.
699   * @param moduleMainClass the string corresponding to the ModuleMainClass attribute, or null.
700   */
701  private void readModule(
702      final ClassVisitor classVisitor,
703      final Context context,
704      final int moduleOffset,
705      final int modulePackagesOffset,
706      final String moduleMainClass) {
707    char[] buffer = context.charBuffer;
708
709    // Read the module_name_index, module_flags and module_version_index fields and visit them.
710    int currentOffset = moduleOffset;
711    String moduleName = readModule(currentOffset, buffer);
712    int moduleFlags = readUnsignedShort(currentOffset + 2);
713    String moduleVersion = readUTF8(currentOffset + 4, buffer);
714    currentOffset += 6;
715    ModuleVisitor moduleVisitor = classVisitor.visitModule(moduleName, moduleFlags, moduleVersion);
716    if (moduleVisitor == null) {
717      return;
718    }
719
720    // Visit the ModuleMainClass attribute.
721    if (moduleMainClass != null) {
722      moduleVisitor.visitMainClass(moduleMainClass);
723    }
724
725    // Visit the ModulePackages attribute.
726    if (modulePackagesOffset != 0) {
727      int packageCount = readUnsignedShort(modulePackagesOffset);
728      int currentPackageOffset = modulePackagesOffset + 2;
729      while (packageCount-- > 0) {
730        moduleVisitor.visitPackage(readPackage(currentPackageOffset, buffer));
731        currentPackageOffset += 2;
732      }
733    }
734
735    // Read the 'requires_count' and 'requires' fields.
736    int requiresCount = readUnsignedShort(currentOffset);
737    currentOffset += 2;
738    while (requiresCount-- > 0) {
739      // Read the requires_index, requires_flags and requires_version fields and visit them.
740      String requires = readModule(currentOffset, buffer);
741      int requiresFlags = readUnsignedShort(currentOffset + 2);
742      String requiresVersion = readUTF8(currentOffset + 4, buffer);
743      currentOffset += 6;
744      moduleVisitor.visitRequire(requires, requiresFlags, requiresVersion);
745    }
746
747    // Read the 'exports_count' and 'exports' fields.
748    int exportsCount = readUnsignedShort(currentOffset);
749    currentOffset += 2;
750    while (exportsCount-- > 0) {
751      // Read the exports_index, exports_flags, exports_to_count and exports_to_index fields
752      // and visit them.
753      String exports = readPackage(currentOffset, buffer);
754      int exportsFlags = readUnsignedShort(currentOffset + 2);
755      int exportsToCount = readUnsignedShort(currentOffset + 4);
756      currentOffset += 6;
757      String[] exportsTo = null;
758      if (exportsToCount != 0) {
759        exportsTo = new String[exportsToCount];
760        for (int i = 0; i < exportsToCount; ++i) {
761          exportsTo[i] = readModule(currentOffset, buffer);
762          currentOffset += 2;
763        }
764      }
765      moduleVisitor.visitExport(exports, exportsFlags, exportsTo);
766    }
767
768    // Reads the 'opens_count' and 'opens' fields.
769    int opensCount = readUnsignedShort(currentOffset);
770    currentOffset += 2;
771    while (opensCount-- > 0) {
772      // Read the opens_index, opens_flags, opens_to_count and opens_to_index fields and visit them.
773      String opens = readPackage(currentOffset, buffer);
774      int opensFlags = readUnsignedShort(currentOffset + 2);
775      int opensToCount = readUnsignedShort(currentOffset + 4);
776      currentOffset += 6;
777      String[] opensTo = null;
778      if (opensToCount != 0) {
779        opensTo = new String[opensToCount];
780        for (int i = 0; i < opensToCount; ++i) {
781          opensTo[i] = readModule(currentOffset, buffer);
782          currentOffset += 2;
783        }
784      }
785      moduleVisitor.visitOpen(opens, opensFlags, opensTo);
786    }
787
788    // Read the 'uses_count' and 'uses' fields.
789    int usesCount = readUnsignedShort(currentOffset);
790    currentOffset += 2;
791    while (usesCount-- > 0) {
792      moduleVisitor.visitUse(readClass(currentOffset, buffer));
793      currentOffset += 2;
794    }
795
796    // Read the  'provides_count' and 'provides' fields.
797    int providesCount = readUnsignedShort(currentOffset);
798    currentOffset += 2;
799    while (providesCount-- > 0) {
800      // Read the provides_index, provides_with_count and provides_with_index fields and visit them.
801      String provides = readClass(currentOffset, buffer);
802      int providesWithCount = readUnsignedShort(currentOffset + 2);
803      currentOffset += 4;
804      String[] providesWith = new String[providesWithCount];
805      for (int i = 0; i < providesWithCount; ++i) {
806        providesWith[i] = readClass(currentOffset, buffer);
807        currentOffset += 2;
808      }
809      moduleVisitor.visitProvide(provides, providesWith);
810    }
811
812    // Visit the end of the module attributes.
813    moduleVisitor.visitEnd();
814  }

816  /**
817   * Reads a JVMS field_info structure and makes the given visitor visit it.
818   *
819   * @param classVisitor the visitor that must visit the field.
820   * @param context information about the class being parsed.
821   * @param fieldInfoOffset the start offset of the field_info structure.
822   * @return the offset of the first byte following the field_info structure.
823   */
824  private int readField(
825      final ClassVisitor classVisitor, final Context context, final int fieldInfoOffset) {
826    char[] charBuffer = context.charBuffer;
827
828    // Read the access_flags, name_index and descriptor_index fields.
829    int currentOffset = fieldInfoOffset;
830    int accessFlags = readUnsignedShort(currentOffset);
831    String name = readUTF8(currentOffset + 2, charBuffer);
832    String descriptor = readUTF8(currentOffset + 4, charBuffer);
833    currentOffset += 6;
834
835    // Read the field attributes (the variables are ordered as in Section 4.7 of the JVMS).
836    // Attribute offsets exclude the attribute_name_index and attribute_length fields.
837    // - The value corresponding to the ConstantValue attribute, or null.
838    Object constantValue = null;
839    // - The string corresponding to the Signature attribute, or null.
840    String signature = null;
841    // - The offset of the RuntimeVisibleAnnotations attribute, or 0.
842    int runtimeVisibleAnnotationsOffset = 0;
843    // - The offset of the RuntimeInvisibleAnnotations attribute, or 0.
844    int runtimeInvisibleAnnotationsOffset = 0;
845    // - The offset of the RuntimeVisibleTypeAnnotations attribute, or 0.
846    int runtimeVisibleTypeAnnotationsOffset = 0;
847    // - The offset of the RuntimeInvisibleTypeAnnotations attribute, or 0.
848    int runtimeInvisibleTypeAnnotationsOffset = 0;
849    // - The non standard attributes (linked with their {@link Attribute#nextAttribute} field).
850    //   This list in the <i>reverse order</i> or their order in the ClassFile structure.
851    Attribute attributes = null;
852
853    int attributesCount = readUnsignedShort(currentOffset);
854    currentOffset += 2;
855    while (attributesCount-- > 0) {
856      // Read the attribute_info's attribute_name and attribute_length fields.
857      String attributeName = readUTF8(currentOffset, charBuffer);
858      int attributeLength = readInt(currentOffset + 2);
859      currentOffset += 6;
860      // The tests are sorted in decreasing frequency order (based on frequencies observed on
861      // typical classes).
862      if (Constants.CONSTANT_VALUE.equals(attributeName)) {
863        int constantvalueIndex = readUnsignedShort(currentOffset);
864        constantValue = constantvalueIndex == 0 ? null : readConst(constantvalueIndex, charBuffer);
865      } else if (Constants.SIGNATURE.equals(attributeName)) {
866        signature = readUTF8(currentOffset, charBuffer);
867      } else if (Constants.DEPRECATED.equals(attributeName)) {
868        accessFlags |= Opcodes.ACC_DEPRECATED;
869      } else if (Constants.SYNTHETIC.equals(attributeName)) {
870        accessFlags |= Opcodes.ACC_SYNTHETIC;
871      } else if (Constants.RUNTIME_VISIBLE_ANNOTATIONS.equals(attributeName)) {
872        runtimeVisibleAnnotationsOffset = currentOffset;
873      } else if (Constants.RUNTIME_VISIBLE_TYPE_ANNOTATIONS.equals(attributeName)) {
874        runtimeVisibleTypeAnnotationsOffset = currentOffset;
875      } else if (Constants.RUNTIME_INVISIBLE_ANNOTATIONS.equals(attributeName)) {
876        runtimeInvisibleAnnotationsOffset = currentOffset;
877      } else if (Constants.RUNTIME_INVISIBLE_TYPE_ANNOTATIONS.equals(attributeName)) {
878        runtimeInvisibleTypeAnnotationsOffset = currentOffset;
879      } else {
880        Attribute attribute =
881            readAttribute(
882                context.attributePrototypes,
883                attributeName,
884                currentOffset,
885                attributeLength,
886                charBuffer,
887                -1,
888                null);
889        attribute.nextAttribute = attributes;
890        attributes = attribute;
891      }
892      currentOffset += attributeLength;
893    }
894
895    // Visit the field declaration.
896    FieldVisitor fieldVisitor =
897        classVisitor.visitField(accessFlags, name, descriptor, signature, constantValue);
898    if (fieldVisitor == null) {
899      return currentOffset;
900    }
901
902    // Visit the RuntimeVisibleAnnotations attribute.
903    if (runtimeVisibleAnnotationsOffset != 0) {
904      int numAnnotations = readUnsignedShort(runtimeVisibleAnnotationsOffset);
905      int currentAnnotationOffset = runtimeVisibleAnnotationsOffset + 2;
906      while (numAnnotations-- > 0) {
907        // Parse the type_index field.
908        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
909        currentAnnotationOffset += 2;
910        // Parse num_element_value_pairs and element_value_pairs and visit these values.
911        currentAnnotationOffset =
912            readElementValues(
913                fieldVisitor.visitAnnotation(annotationDescriptor, /* visible = */ true),
914                currentAnnotationOffset,
915                /* named = */ true,
916                charBuffer);
917      }
918    }
919
920    // Visit the RuntimeInvisibleAnnotations attribute.
921    if (runtimeInvisibleAnnotationsOffset != 0) {
922      int numAnnotations = readUnsignedShort(runtimeInvisibleAnnotationsOffset);
923      int currentAnnotationOffset = runtimeInvisibleAnnotationsOffset + 2;
924      while (numAnnotations-- > 0) {
925        // Parse the type_index field.
926        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
927        currentAnnotationOffset += 2;
928        // Parse num_element_value_pairs and element_value_pairs and visit these values.
929        currentAnnotationOffset =
930            readElementValues(
931                fieldVisitor.visitAnnotation(annotationDescriptor, /* visible = */ false),
932                currentAnnotationOffset,
933                /* named = */ true,
934                charBuffer);
935      }
936    }
937
938    // Visit the RuntimeVisibleTypeAnnotations attribute.
939    if (runtimeVisibleTypeAnnotationsOffset != 0) {
940      int numAnnotations = readUnsignedShort(runtimeVisibleTypeAnnotationsOffset);
941      int currentAnnotationOffset = runtimeVisibleTypeAnnotationsOffset + 2;
942      while (numAnnotations-- > 0) {
943        // Parse the target_type, target_info and target_path fields.
944        currentAnnotationOffset = readTypeAnnotationTarget(context, currentAnnotationOffset);
945        // Parse the type_index field.
946        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
947        currentAnnotationOffset += 2;
948        // Parse num_element_value_pairs and element_value_pairs and visit these values.
949        currentAnnotationOffset =
950            readElementValues(
951                fieldVisitor.visitTypeAnnotation(
952                    context.currentTypeAnnotationTarget,
953                    context.currentTypeAnnotationTargetPath,
954                    annotationDescriptor,
955                    /* visible = */ true),
956                currentAnnotationOffset,
957                /* named = */ true,
958                charBuffer);
959      }
960    }
961
962    // Visit the RuntimeInvisibleTypeAnnotations attribute.
963    if (runtimeInvisibleTypeAnnotationsOffset != 0) {
964      int numAnnotations = readUnsignedShort(runtimeInvisibleTypeAnnotationsOffset);
965      int currentAnnotationOffset = runtimeInvisibleTypeAnnotationsOffset + 2;
966      while (numAnnotations-- > 0) {
967        // Parse the target_type, target_info and target_path fields.
968        currentAnnotationOffset = readTypeAnnotationTarget(context, currentAnnotationOffset);
969        // Parse the type_index field.
970        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
971        currentAnnotationOffset += 2;
972        // Parse num_element_value_pairs and element_value_pairs and visit these values.
973        currentAnnotationOffset =
974            readElementValues(
975                fieldVisitor.visitTypeAnnotation(
976                    context.currentTypeAnnotationTarget,
977                    context.currentTypeAnnotationTargetPath,
978                    annotationDescriptor,
979                    /* visible = */ false),
980                currentAnnotationOffset,
981                /* named = */ true,
982                charBuffer);
983      }
984    }
985
986    // Visit the non standard attributes.
987    while (attributes != null) {
988      // Copy and reset the nextAttribute field so that it can also be used in FieldWriter.
989      Attribute nextAttribute = attributes.nextAttribute;
990      attributes.nextAttribute = null;
991      fieldVisitor.visitAttribute(attributes);
992      attributes = nextAttribute;
993    }
994
995    // Visit the end of the field.
996    fieldVisitor.visitEnd();
997    return currentOffset;
998  }

1000  /**
1001   * Reads a JVMS method_info structure and makes the given visitor visit it.
1002   *
1003   * @param classVisitor the visitor that must visit the method.
1004   * @param context information about the class being parsed.
1005   * @param methodInfoOffset the start offset of the method_info structure.
1006   * @return the offset of the first byte following the method_info structure.
1007   */
1008  private int readMethod(
1009      final ClassVisitor classVisitor, final Context context, final int methodInfoOffset) {
1010    char[] charBuffer = context.charBuffer;
1011
1012    // Read the access_flags, name_index and descriptor_index fields.
1013    int currentOffset = methodInfoOffset;
1014    context.currentMethodAccessFlags = readUnsignedShort(currentOffset);
1015    context.currentMethodName = readUTF8(currentOffset + 2, charBuffer);
1016    context.currentMethodDescriptor = readUTF8(currentOffset + 4, charBuffer);
1017    currentOffset += 6;
1018
1019    // Read the method attributes (the variables are ordered as in Section 4.7 of the JVMS).
1020    // Attribute offsets exclude the attribute_name_index and attribute_length fields.
1021    // - The offset of the Code attribute, or 0.
1022    int codeOffset = 0;
1023    // - The offset of the Exceptions attribute, or 0.
1024    int exceptionsOffset = 0;
1025    // - The strings corresponding to the Exceptions attribute, or null.
1026    String[] exceptions = null;
1027    // - Whether the method has a Synthetic attribute.
1028    boolean synthetic = false;
1029    // - The constant pool index contained in the Signature attribute, or 0.
1030    int signatureIndex = 0;
1031    // - The offset of the RuntimeVisibleAnnotations attribute, or 0.
1032    int runtimeVisibleAnnotationsOffset = 0;
1033    // - The offset of the RuntimeInvisibleAnnotations attribute, or 0.
1034    int runtimeInvisibleAnnotationsOffset = 0;
1035    // - The offset of the RuntimeVisibleParameterAnnotations attribute, or 0.
1036    int runtimeVisibleParameterAnnotationsOffset = 0;
1037    // - The offset of the RuntimeInvisibleParameterAnnotations attribute, or 0.
1038    int runtimeInvisibleParameterAnnotationsOffset = 0;
1039    // - The offset of the RuntimeVisibleTypeAnnotations attribute, or 0.
1040    int runtimeVisibleTypeAnnotationsOffset = 0;
1041    // - The offset of the RuntimeInvisibleTypeAnnotations attribute, or 0.
1042    int runtimeInvisibleTypeAnnotationsOffset = 0;
1043    // - The offset of the AnnotationDefault attribute, or 0.
1044    int annotationDefaultOffset = 0;
1045    // - The offset of the MethodParameters attribute, or 0.
1046    int methodParametersOffset = 0;
1047    // - The non standard attributes (linked with their {@link Attribute#nextAttribute} field).
1048    //   This list in the <i>reverse order</i> or their order in the ClassFile structure.
1049    Attribute attributes = null;
1050
1051    int attributesCount = readUnsignedShort(currentOffset);
1052    currentOffset += 2;
1053    while (attributesCount-- > 0) {
1054      // Read the attribute_info's attribute_name and attribute_length fields.
1055      String attributeName = readUTF8(currentOffset, charBuffer);
1056      int attributeLength = readInt(currentOffset + 2);
1057      currentOffset += 6;
1058      // The tests are sorted in decreasing frequency order (based on frequencies observed on
1059      // typical classes).
1060      if (Constants.CODE.equals(attributeName)) {
1061        if ((context.parsingOptions & SKIP_CODE) == 0) {
1062          codeOffset = currentOffset;
1063        }
1064      } else if (Constants.EXCEPTIONS.equals(attributeName)) {
1065        exceptionsOffset = currentOffset;
1066        exceptions = new String[readUnsignedShort(exceptionsOffset)];
1067        int currentExceptionOffset = exceptionsOffset + 2;
1068        for (int i = 0; i < exceptions.length; ++i) {
1069          exceptions[i] = readClass(currentExceptionOffset, charBuffer);
1070          currentExceptionOffset += 2;
1071        }
1072      } else if (Constants.SIGNATURE.equals(attributeName)) {
1073        signatureIndex = readUnsignedShort(currentOffset);
1074      } else if (Constants.DEPRECATED.equals(attributeName)) {
1075        context.currentMethodAccessFlags |= Opcodes.ACC_DEPRECATED;
1076      } else if (Constants.RUNTIME_VISIBLE_ANNOTATIONS.equals(attributeName)) {
1077        runtimeVisibleAnnotationsOffset = currentOffset;
1078      } else if (Constants.RUNTIME_VISIBLE_TYPE_ANNOTATIONS.equals(attributeName)) {
1079        runtimeVisibleTypeAnnotationsOffset = currentOffset;
1080      } else if (Constants.ANNOTATION_DEFAULT.equals(attributeName)) {
1081        annotationDefaultOffset = currentOffset;
1082      } else if (Constants.SYNTHETIC.equals(attributeName)) {
1083        synthetic = true;
1084        context.currentMethodAccessFlags |= Opcodes.ACC_SYNTHETIC;
1085      } else if (Constants.RUNTIME_INVISIBLE_ANNOTATIONS.equals(attributeName)) {
1086        runtimeInvisibleAnnotationsOffset = currentOffset;
1087      } else if (Constants.RUNTIME_INVISIBLE_TYPE_ANNOTATIONS.equals(attributeName)) {
1088        runtimeInvisibleTypeAnnotationsOffset = currentOffset;
1089      } else if (Constants.RUNTIME_VISIBLE_PARAMETER_ANNOTATIONS.equals(attributeName)) {
1090        runtimeVisibleParameterAnnotationsOffset = currentOffset;
1091      } else if (Constants.RUNTIME_INVISIBLE_PARAMETER_ANNOTATIONS.equals(attributeName)) {
1092        runtimeInvisibleParameterAnnotationsOffset = currentOffset;
1093      } else if (Constants.METHOD_PARAMETERS.equals(attributeName)) {
1094        methodParametersOffset = currentOffset;
1095      } else {
1096        Attribute attribute =
1097            readAttribute(
1098                context.attributePrototypes,
1099                attributeName,
1100                currentOffset,
1101                attributeLength,
1102                charBuffer,
1103                -1,
1104                null);
1105        attribute.nextAttribute = attributes;
1106        attributes = attribute;
1107      }
1108      currentOffset += attributeLength;
1109    }
1110
1111    // Visit the method declaration.
1112    MethodVisitor methodVisitor =
1113        classVisitor.visitMethod(
1114            context.currentMethodAccessFlags,
1115            context.currentMethodName,
1116            context.currentMethodDescriptor,
1117            signatureIndex == 0 ? null : readUTF(signatureIndex, charBuffer),
1118            exceptions);
1119    if (methodVisitor == null) {
1120      return currentOffset;
1121    }
1122
1123    // If the returned MethodVisitor is in fact a MethodWriter, it means there is no method
1124    // adapter between the reader and the writer. In this case, it might be possible to copy
1125    // the method attributes directly into the writer. If so, return early without visiting
1126    // the content of these attributes.
1127    if (methodVisitor instanceof MethodWriter) {
1128      MethodWriter methodWriter = (MethodWriter) methodVisitor;
1129      if (methodWriter.canCopyMethodAttributes(
1130          this,
1131          methodInfoOffset,
1132          currentOffset - methodInfoOffset,
1133          synthetic,
1134          (context.currentMethodAccessFlags & Opcodes.ACC_DEPRECATED) != 0,
1135          signatureIndex,
1136          exceptionsOffset)) {
1137        return currentOffset;
1138      }
1139    }
1140
1141    // Visit the MethodParameters attribute.
1142    if (methodParametersOffset != 0) {
1143      int parametersCount = readByte(methodParametersOffset);
1144      int currentParameterOffset = methodParametersOffset + 1;
1145      while (parametersCount-- > 0) {
1146        // Read the name_index and access_flags fields and visit them.
1147        methodVisitor.visitParameter(
1148            readUTF8(currentParameterOffset, charBuffer),
1149            readUnsignedShort(currentParameterOffset + 2));
1150        currentParameterOffset += 4;
1151      }
1152    }
1153
1154    // Visit the AnnotationDefault attribute.
1155    if (annotationDefaultOffset != 0) {
1156      AnnotationVisitor annotationVisitor = methodVisitor.visitAnnotationDefault();
1157      readElementValue(annotationVisitor, annotationDefaultOffset, null, charBuffer);
1158      if (annotationVisitor != null) {
1159        annotationVisitor.visitEnd();
1160      }
1161    }
1162
1163    // Visit the RuntimeVisibleAnnotations attribute.
1164    if (runtimeVisibleAnnotationsOffset != 0) {
1165      int numAnnotations = readUnsignedShort(runtimeVisibleAnnotationsOffset);
1166      int currentAnnotationOffset = runtimeVisibleAnnotationsOffset + 2;
1167      while (numAnnotations-- > 0) {
1168        // Parse the type_index field.
1169        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
1170        currentAnnotationOffset += 2;
1171        // Parse num_element_value_pairs and element_value_pairs and visit these values.
1172        currentAnnotationOffset =
1173            readElementValues(
1174                methodVisitor.visitAnnotation(annotationDescriptor, /* visible = */ true),
1175                currentAnnotationOffset,
1176                /* named = */ true,
1177                charBuffer);
1178      }
1179    }
1180
1181    // Visit the RuntimeInvisibleAnnotations attribute.
1182    if (runtimeInvisibleAnnotationsOffset != 0) {
1183      int numAnnotations = readUnsignedShort(runtimeInvisibleAnnotationsOffset);
1184      int currentAnnotationOffset = runtimeInvisibleAnnotationsOffset + 2;
1185      while (numAnnotations-- > 0) {
1186        // Parse the type_index field.
1187        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
1188        currentAnnotationOffset += 2;
1189        // Parse num_element_value_pairs and element_value_pairs and visit these values.
1190        currentAnnotationOffset =
1191            readElementValues(
1192                methodVisitor.visitAnnotation(annotationDescriptor, /* visible = */ false),

1194                /* named = */ true,

1196      }

1202      int currentAnnotationOffset = runtimeVisibleTypeAnnotationsOffset + 2;
1203      while (numAnnotations-- > 0) {
1204        // Parse the target_type, target_info and target_path fields.
1205        currentAnnotationOffset = readTypeAnnotationTarget(context, currentAnnotationOffset);
1206        // Parse the type_index field.
1207        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
1208        currentAnnotationOffset += 2;
1209        // Parse num_element_value_pairs and element_value_pairs and visit these values.
1210        currentAnnotationOffset =
1211            readElementValues(
1212                methodVisitor.visitTypeAnnotation(
1213                    context.currentTypeAnnotationTarget,
1214                    context.currentTypeAnnotationTargetPath,
1215                    annotationDescriptor,
1216                    /* visible = */ true),
1217                currentAnnotationOffset,
1218                /* named = */ true,
1219                charBuffer);
1220      }
1221    }
1222
1223    // Visit the RuntimeInvisibleTypeAnnotations attribute.
1224    if (runtimeInvisibleTypeAnnotationsOffset != 0) {
1225      int numAnnotations = readUnsignedShort(runtimeInvisibleTypeAnnotationsOffset);
1226      int currentAnnotationOffset = runtimeInvisibleTypeAnnotationsOffset + 2;
1227      while (numAnnotations-- > 0) {
1228        // Parse the target_type, target_info and target_path fields.
1229        currentAnnotationOffset = readTypeAnnotationTarget(context, currentAnnotationOffset);
1230        // Parse the type_index field.
1231        String annotationDescriptor = readUTF8(currentAnnotationOffset, charBuffer);
1232        currentAnnotationOffset += 2;
1233        // Parse num_element_value_pairs and element_value_pairs and visit these values.
1234        currentAnnotationOffset =
1235            readElementValues(
1236                methodVisitor.visitTypeAnnotation(
1237                    context.currentTypeAnnotationTarget,
1238                    context.currentTypeAnnotationTargetPath,
1239                    annotationDescriptor,
1240                    /* visible = */ false),
1241                currentAnnotationOffset,
1242                /* named = */ true,
1243                charBuffer);
1244      }
1245    }
1246
1247    // Visit the RuntimeVisibleParameterAnnotations attribute.
1248    if (runtimeVisibleParameterAnnotationsOffset != 0) {
1249      readParameterAnnotations(
1250          methodVisitor, context, runtimeVisibleParameterAnnotationsOffset, /* visible = */ true);
1251    }
1252
1253    // Visit the RuntimeInvisibleParameterAnnotations attribute.
1254    if (runtimeInvisibleParameterAnnotationsOffset != 0) {
1255      readParameterAnnotations(
1256          methodVisitor,
1257          context,
1258          runtimeInvisibleParameterAnnotationsOffset,
1259          /* visible = */ false);
1260    }
1261
1262    // Visit the non standard attributes.
1263    while (attributes != null) {
1264      // Copy and reset the nextAttribute field so that it can also be used in MethodWriter.
1265      Attribute nextAttribute = attributes.nextAttribute;
1266      attributes.nextAttribute = null;
1267      methodVisitor.visitAttribute(attributes);
1268      attributes = nextAttribute;
1269    }
1270
1271    // Visit the Code attribute.
1272    if (codeOffset != 0) {
1273      methodVisitor.visitCode();
1274      readCode(methodVisitor, context, codeOffset);
1275    }
1276
1277    // Visit the end of the method.
1278    methodVisitor.visitEnd();
1279    return currentOffset;
1280  }

1757      // stack map table, we look for 3 consecutive bytes that "look like" an UNINITIALIZED type
1758      // (tag ITEM_Uninitialized, offset within bytecode bounds, NEW instruction at this offset).
1759      // We may find false positives (i.e. not real UNINITIALIZED types), but this should be rare,
1760      // and the only consequence will be the creation of an unneeded label. This is better than
1761      // creating a label for each NEW instruction, and faster than fully decoding the whole stack
1762      // map table.

2333          typeTable[--typeTableIndex] = currentOffset + 6;
2334          typeTable[--typeTableIndex] = readUnsignedShort(currentOffset + 8);
2335          typeTable[--typeTableIndex] = readUnsignedShort(currentOffset);
2336          currentOffset += 10;
2337        }
2338      }
2339      int localVariableTableLength = readUnsignedShort(localVariableTableOffset);
2340      currentOffset = localVariableTableOffset + 2;
2341      while (localVariableTableLength-- > 0) {
2342        int startPc = readUnsignedShort(currentOffset);
2343        int length = readUnsignedShort(currentOffset + 2);
2344        String name = readUTF8(currentOffset + 4, charBuffer);
2345        String descriptor = readUTF8(currentOffset + 6, charBuffer);
2346        int index = readUnsignedShort(currentOffset + 8);
2347        currentOffset += 10;
2348        String signature = null;
2349        if (typeTable != null) {
2350          for (int i = 0; i < typeTable.length; i += 3) {
2351            if (typeTable[i] == startPc && typeTable[i + 1] == index) {
2352              signature = readUTF8(typeTable[i + 2], charBuffer);
2353              break;
2354            }
2355          }
2356        }
2357        methodVisitor.visitLocalVariable(
2358            name, descriptor, signature, labels[startPc], labels[startPc + length], index);
2359      }
2360    }
2361
2362    // Visit the local variable type annotations of the RuntimeVisibleTypeAnnotations attribute.
2363    if (visibleTypeAnnotationOffsets != null) {
2364      for (int i = 0; i < visibleTypeAnnotationOffsets.length; ++i) {
2365        int targetType = readByte(visibleTypeAnnotationOffsets[i]);
2366        if (targetType == TypeReference.LOCAL_VARIABLE
2367            || targetType == TypeReference.RESOURCE_VARIABLE) {
2368          // Parse the target_type, target_info and target_path fields.
2369          currentOffset = readTypeAnnotationTarget(context, visibleTypeAnnotationOffsets[i]);
2370          // Parse the type_index field.
2371          String annotationDescriptor = readUTF8(currentOffset, charBuffer);
2372          currentOffset += 2;
2373          // Parse num_element_value_pairs and element_value_pairs and visit these values.
2374          readElementValues(
2375              methodVisitor.visitLocalVariableAnnotation(
2376                  context.currentTypeAnnotationTarget,
2377                  context.currentTypeAnnotationTargetPath,
2378                  context.currentLocalVariableAnnotationRangeStarts,
2379                  context.currentLocalVariableAnnotationRangeEnds,
2380                  context.currentLocalVariableAnnotationRangeIndices,
2381                  annotationDescriptor,
2382                  /* visible = */ true),
2383              currentOffset,
2384              /* named = */ true,
2385              charBuffer);
2386        }
2387      }
2388    }
2389
2390    // Visit the local variable type annotations of the RuntimeInvisibleTypeAnnotations attribute.
2391    if (invisibleTypeAnnotationOffsets != null) {
2392      for (int i = 0; i < invisibleTypeAnnotationOffsets.length; ++i) {
2393        int targetType = readByte(invisibleTypeAnnotationOffsets[i]);
2394        if (targetType == TypeReference.LOCAL_VARIABLE
2395            || targetType == TypeReference.RESOURCE_VARIABLE) {
2396          // Parse the target_type, target_info and target_path fields.
2397          currentOffset = readTypeAnnotationTarget(context, invisibleTypeAnnotationOffsets[i]);
2398          // Parse the type_index field.
2399          String annotationDescriptor = readUTF8(currentOffset, charBuffer);
2400          currentOffset += 2;
2401          // Parse num_element_value_pairs and element_value_pairs and visit these values.
2402          readElementValues(
2403              methodVisitor.visitLocalVariableAnnotation(
2404                  context.currentTypeAnnotationTarget,
2405                  context.currentTypeAnnotationTargetPath,
2406                  context.currentLocalVariableAnnotationRangeStarts,
2407                  context.currentLocalVariableAnnotationRangeEnds,
2408                  context.currentLocalVariableAnnotationRangeIndices,
2409                  annotationDescriptor,
2410                  /* visible = */ false),
2411              currentOffset,
2412              /* named = */ true,
2413              charBuffer);
2414        }
2415      }
2416    }
2417
2418    // Visit the non standard attributes.
2419    while (attributes != null) {
2420      // Copy and reset the nextAttribute field so that it can also be used in MethodWriter.
2421      Attribute nextAttribute = attributes.nextAttribute;
2422      attributes.nextAttribute = null;
2423      methodVisitor.visitAttribute(attributes);
2424      attributes = nextAttribute;
2425    }
2426
2427    // Visit the max stack and max locals values.
2428    methodVisitor.visitMaxs(maxStack, maxLocals);
2429  }

2481  /**
2482   * Parses a Runtime[In]VisibleTypeAnnotations attribute to find the offset of each type_annotation
2483   * entry it contains, to find the corresponding labels, and to visit the try catch block
2484   * annotations.
2485   *
2486   * @param methodVisitor the method visitor to be used to visit the try catch block annotations.
2487   * @param context information about the class being parsed.
2488   * @param runtimeTypeAnnotationsOffset the start offset of a Runtime[In]VisibleTypeAnnotations
2489   *     attribute, excluding the attribute_info's attribute_name_index and attribute_length fields.
2490   * @param visible true if the attribute to parse is a RuntimeVisibleTypeAnnotations attribute,
2491   *     false it is a RuntimeInvisibleTypeAnnotations attribute.
2492   * @return the start offset of each entry of the Runtime[In]VisibleTypeAnnotations_attribute's
2493   *     'annotations' array field.
2494   */
2495  private int[] readTypeAnnotations(
2496      final MethodVisitor methodVisitor,
2497      final Context context,
2498      final int runtimeTypeAnnotationsOffset,
2499      final boolean visible) {
2500    char[] charBuffer = context.charBuffer;
2501    int currentOffset = runtimeTypeAnnotationsOffset;
2502    // Read the num_annotations field and create an array to store the type_annotation offsets.
2503    int[] typeAnnotationsOffsets = new int[readUnsignedShort(currentOffset)];
2504    currentOffset += 2;
2505    // Parse the 'annotations' array field.
2506    for (int i = 0; i < typeAnnotationsOffsets.length; ++i) {
2507      typeAnnotationsOffsets[i] = currentOffset;
2508      // Parse the type_annotation's target_type and the target_info fields. The size of the
2509      // target_info field depends on the value of target_type.
2510      int targetType = readInt(currentOffset);
2511      switch (targetType >>> 24) {
2512        case TypeReference.LOCAL_VARIABLE:
2513        case TypeReference.RESOURCE_VARIABLE:
2514          // A localvar_target has a variable size, which depends on the value of their table_length
2515          // field. It also references bytecode offsets, for which we need labels.
2516          int tableLength = readUnsignedShort(currentOffset + 1);
2517          currentOffset += 3;
2518          while (tableLength-- > 0) {
2519            int startPc = readUnsignedShort(currentOffset);
2520            int length = readUnsignedShort(currentOffset + 2);
2521            // Skip the index field (2 bytes).
2522            currentOffset += 6;
2523            createLabel(startPc, context.currentMethodLabels);
2524            createLabel(startPc + length, context.currentMethodLabels);
2525          }
2526          break;
2527        case TypeReference.CAST:
2528        case TypeReference.CONSTRUCTOR_INVOCATION_TYPE_ARGUMENT:
2529        case TypeReference.METHOD_INVOCATION_TYPE_ARGUMENT:
2530        case TypeReference.CONSTRUCTOR_REFERENCE_TYPE_ARGUMENT:
2531        case TypeReference.METHOD_REFERENCE_TYPE_ARGUMENT:
2532          currentOffset += 4;
2533          break;
2534        case TypeReference.CLASS_EXTENDS:
2535        case TypeReference.CLASS_TYPE_PARAMETER_BOUND:
2536        case TypeReference.METHOD_TYPE_PARAMETER_BOUND:
2537        case TypeReference.THROWS:
2538        case TypeReference.EXCEPTION_PARAMETER:
2539        case TypeReference.INSTANCEOF:
2540        case TypeReference.NEW:
2541        case TypeReference.CONSTRUCTOR_REFERENCE:
2542        case TypeReference.METHOD_REFERENCE:
2543          currentOffset += 3;
2544          break;
2545        case TypeReference.CLASS_TYPE_PARAMETER:
2546        case TypeReference.METHOD_TYPE_PARAMETER:
2547        case TypeReference.METHOD_FORMAL_PARAMETER:
2548        case TypeReference.FIELD:
2549        case TypeReference.METHOD_RETURN:
2550        case TypeReference.METHOD_RECEIVER:
2551        default:
2552          // TypeReference type which can't be used in Code attribute, or which is unknown.
2553          throw new IllegalArgumentException();
2554      }
2555      // Parse the rest of the type_annotation structure, starting with the target_path structure
2556      // (whose size depends on its path_length field).
2557      int pathLength = readByte(currentOffset);
2558      if ((targetType >>> 24) == TypeReference.EXCEPTION_PARAMETER) {
2559        // Parse the target_path structure and create a corresponding TypePath.
2560        TypePath path = pathLength == 0 ? null : new TypePath(b, currentOffset);
2561        currentOffset += 1 + 2 * pathLength;
2562        // Parse the type_index field.
2563        String annotationDescriptor = readUTF8(currentOffset, charBuffer);
2564        currentOffset += 2;
2565        // Parse num_element_value_pairs and element_value_pairs and visit these values.
2566        currentOffset =
2567            readElementValues(
2568                methodVisitor.visitTryCatchAnnotation(
2569                    targetType & 0xFFFFFF00, path, annotationDescriptor, visible),
2570                currentOffset,
2571                /* named = */ true,
2572                charBuffer);
2573      } else {
2574        // We don't want to visit the other target_type annotations, so we just skip them (which
2575        // requires some parsing because the element_value_pairs array has a variable size). First,
2576        // skip the target_path structure:
2577        currentOffset += 3 + 2 * pathLength;
2578        // Then skip the num_element_value_pairs and element_value_pairs fields (by reading them
2579        // with a null AnnotationVisitor).
2580        currentOffset =
2581            readElementValues(
2582                /* annotationVisitor = */ null, currentOffset, /* named = */ true, charBuffer);
2583      }
2584    }
2585    return typeAnnotationsOffsets;
2586  }

2589   * Returns the bytecode offset corresponding to the specified JVMS 'type_annotation' structure, or
2590   * -1 if there is no such type_annotation of if it does not have a bytecode offset.
2591   *
2592   * @param typeAnnotationOffsets the offset of each 'type_annotation' entry in a
2593   *     Runtime[In]VisibleTypeAnnotations attribute, or null.
2594   * @param typeAnnotationIndex the index a 'type_annotation' entry in typeAnnotationOffsets.
2595   * @return bytecode offset corresponding to the specified JVMS 'type_annotation' structure, or -1
2596   *     if there is no such type_annotation of if it does not have a bytecode offset.

2601        || typeAnnotationIndex >= typeAnnotationOffsets.length
2602        || readByte(typeAnnotationOffsets[typeAnnotationIndex]) < TypeReference.INSTANCEOF) {

2608  /**
2609   * Parses the header of a JVMS type_annotation structure to extract its target_type, target_info
2610   * and target_path (the result is stored in the given context), and returns the start offset of
2611   * the rest of the type_annotation structure.
2612   *
2613   * @param context information about the class being parsed. This is where the extracted
2614   *     target_type and target_path must be stored.
2615   * @param typeAnnotationOffset the start offset of a type_annotation structure.
2616   * @return the start offset of the rest of the type_annotation structure.
2617   */
2618  private int readTypeAnnotationTarget(final Context context, final int typeAnnotationOffset) {
2619    int currentOffset = typeAnnotationOffset;
2620    // Parse and store the target_type structure.
2621    int targetType = readInt(typeAnnotationOffset);
2622    switch (targetType >>> 24) {
2623      case TypeReference.CLASS_TYPE_PARAMETER:
2624      case TypeReference.METHOD_TYPE_PARAMETER:
2625      case TypeReference.METHOD_FORMAL_PARAMETER:
2626        targetType &= 0xFFFF0000;
2627        currentOffset += 2;
2628        break;
2629      case TypeReference.FIELD:
2630      case TypeReference.METHOD_RETURN:
2631      case TypeReference.METHOD_RECEIVER:
2632        targetType &= 0xFF000000;
2633        currentOffset += 1;
2634        break;
2635      case TypeReference.LOCAL_VARIABLE:
2636      case TypeReference.RESOURCE_VARIABLE:
2637        targetType &= 0xFF000000;
2638        int tableLength = readUnsignedShort(currentOffset + 1);
2639        currentOffset += 3;
2640        context.currentLocalVariableAnnotationRangeStarts = new Label[tableLength];
2641        context.currentLocalVariableAnnotationRangeEnds = new Label[tableLength];
2642        context.currentLocalVariableAnnotationRangeIndices = new int[tableLength];
2643        for (int i = 0; i < tableLength; ++i) {
2644          int startPc = readUnsignedShort(currentOffset);
2645          int length = readUnsignedShort(currentOffset + 2);
2646          int index = readUnsignedShort(currentOffset + 4);
2647          currentOffset += 6;
2648          context.currentLocalVariableAnnotationRangeStarts[i] =
2649              createLabel(startPc, context.currentMethodLabels);
2650          context.currentLocalVariableAnnotationRangeEnds[i] =
2651              createLabel(startPc + length, context.currentMethodLabels);
2652          context.currentLocalVariableAnnotationRangeIndices[i] = index;
2653        }
2654        break;
2655      case TypeReference.CAST:
2656      case TypeReference.CONSTRUCTOR_INVOCATION_TYPE_ARGUMENT:
2657      case TypeReference.METHOD_INVOCATION_TYPE_ARGUMENT:
2658      case TypeReference.CONSTRUCTOR_REFERENCE_TYPE_ARGUMENT:
2659      case TypeReference.METHOD_REFERENCE_TYPE_ARGUMENT:
2660        targetType &= 0xFF0000FF;
2661        currentOffset += 4;
2662        break;
2663      case TypeReference.CLASS_EXTENDS:
2664      case TypeReference.CLASS_TYPE_PARAMETER_BOUND:
2665      case TypeReference.METHOD_TYPE_PARAMETER_BOUND:
2666      case TypeReference.THROWS:
2667      case TypeReference.EXCEPTION_PARAMETER:
2668        targetType &= 0xFFFFFF00;
2669        currentOffset += 3;
2670        break;
2671      case TypeReference.INSTANCEOF:
2672      case TypeReference.NEW:
2673      case TypeReference.CONSTRUCTOR_REFERENCE:
2674      case TypeReference.METHOD_REFERENCE:
2675        targetType &= 0xFF000000;
2676        currentOffset += 3;
2677        break;
2678      default:
2679        throw new IllegalArgumentException();
2680    }
2681    context.currentTypeAnnotationTarget = targetType;
2682    // Parse and store the target_path structure.
2683    int pathLength = readByte(currentOffset);
2684    context.currentTypeAnnotationTargetPath =
2685        pathLength == 0 ? null : new TypePath(b, currentOffset);
2686    // Return the start offset of the rest of the type_annotation structure.
2687    return currentOffset + 1 + 2 * pathLength;
2688  }

2728  /**
2729   * Reads the element values of a JVMS 'annotation' structure and makes the given visitor visit
2730   * them. This method can also be used to read the values of the JVMS 'array_value' field of an
2731   * annotation's 'element_value'.
2732   *
2733   * @param annotationVisitor the visitor that must visit the values.
2734   * @param annotationOffset the start offset of an 'annotation' structure (excluding its type_index
2735   *     field) or of an 'array_value' structure.
2736   * @param named if the annotation values are named or not. This should be true to parse the values
2737   *     of a JVMS 'annotation' structure, and false to parse the JVMS 'array_value' of an
2738   *     annotation's element_value.
2739   * @param charBuffer the buffer used to read strings in the constant pool.
2740   * @return the end offset of the JVMS 'annotation' or 'array_value' structure.
2741   */
2742  private int readElementValues(
2743      final AnnotationVisitor annotationVisitor,
2744      final int annotationOffset,
2745      final boolean named,
2746      final char[] charBuffer) {
2747    int currentOffset = annotationOffset;
2748    // Read the num_element_value_pairs field (or num_values field for an array_value).
2749    int numElementValuePairs = readUnsignedShort(currentOffset);
2750    currentOffset += 2;
2751    if (named) {
2752      // Parse the element_value_pairs array.
2753      while (numElementValuePairs-- > 0) {
2754        String elementName = readUTF8(currentOffset, charBuffer);
2755        currentOffset =
2756            readElementValue(annotationVisitor, currentOffset + 2, elementName, charBuffer);
2757      }
2758    } else {
2759      // Parse the array_value array.
2760      while (numElementValuePairs-- > 0) {
2761        currentOffset =
2762            readElementValue(annotationVisitor, currentOffset, /* named = */ null, charBuffer);
2763      }
2764    }
2765    if (annotationVisitor != null) {
2766      annotationVisitor.visitEnd();
2767    }
2768    return currentOffset;
2769  }

2771  /**
2772   * Reads a JVMS 'element_value' structure and makes the given visitor visit it.
2773   *
2774   * @param annotationVisitor the visitor that must visit the element_value structure.
2775   * @param elementValueOffset the start offset in {@link #b} of the element_value structure to be
2776   *     read.
2777   * @param elementName the name of the element_value structure to be read, or <tt>null</tt>.
2778   * @param charBuffer the buffer used to read strings in the constant pool.
2779   * @return the end offset of the JVMS 'element_value' structure.
2780   */
2781  private int readElementValue(
2782      final AnnotationVisitor annotationVisitor,
2783      final int elementValueOffset,
2784      final String elementName,
2785      final char[] charBuffer) {
2786    int currentOffset = elementValueOffset;
2787    if (annotationVisitor == null) {
2788      switch (b[currentOffset] & 0xFF) {
2789        case 'e': // enum_const_value
2790          return currentOffset + 5;
2791        case '@': // annotation_value
2792          return readElementValues(null, currentOffset + 3, /* named = */ true, charBuffer);
2793        case '[': // array_value
2794          return readElementValues(null, currentOffset + 1, /* named = */ false, charBuffer);
2795        default:
2796          return currentOffset + 3;
2797      }
2798    }
2799    switch (b[currentOffset++] & 0xFF) {
2800      case 'B': // const_value_index, CONSTANT_Integer
2801        annotationVisitor.visit(
2802            elementName, (byte) readInt(cpInfoOffsets[readUnsignedShort(currentOffset)]));
2803        currentOffset += 2;
2804        break;
2805      case 'C': // const_value_index, CONSTANT_Integer
2806        annotationVisitor.visit(
2807            elementName, (char) readInt(cpInfoOffsets[readUnsignedShort(currentOffset)]));
2808        currentOffset += 2;
2809        break;
2810      case 'D': // const_value_index, CONSTANT_Double
2811      case 'F': // const_value_index, CONSTANT_Float
2812      case 'I': // const_value_index, CONSTANT_Integer
2813      case 'J': // const_value_index, CONSTANT_Long
2814        annotationVisitor.visit(
2815            elementName, readConst(readUnsignedShort(currentOffset), charBuffer));
2816        currentOffset += 2;
2817        break;
2818      case 'S': // const_value_index, CONSTANT_Integer
2819        annotationVisitor.visit(
2820            elementName, (short) readInt(cpInfoOffsets[readUnsignedShort(currentOffset)]));
2821        currentOffset += 2;
2822        break;
2823
2824      case 'Z': // const_value_index, CONSTANT_Integer
2825        annotationVisitor.visit(
2826            elementName,
2827            readInt(cpInfoOffsets[readUnsignedShort(currentOffset)]) == 0
2828                ? Boolean.FALSE
2829                : Boolean.TRUE);
2830        currentOffset += 2;
2831        break;
2832      case 's': // const_value_index, CONSTANT_Utf8
2833        annotationVisitor.visit(elementName, readUTF8(currentOffset, charBuffer));
2834        currentOffset += 2;
2835        break;
2836      case 'e': // enum_const_value
2837        annotationVisitor.visitEnum(
2838            elementName,
2839            readUTF8(currentOffset, charBuffer),
2840            readUTF8(currentOffset + 2, charBuffer));
2841        currentOffset += 4;
2842        break;
2843      case 'c': // class_info
2844        annotationVisitor.visit(elementName, Type.getType(readUTF8(currentOffset, charBuffer)));
2845        currentOffset += 2;
2846        break;
2847      case '@': // annotation_value
2848        currentOffset =
2849            readElementValues(
2850                annotationVisitor.visitAnnotation(elementName, readUTF8(currentOffset, charBuffer)),
2851                currentOffset + 2,
2852                true,
2853                charBuffer);
2854        break;
2855      case '[': // array_value
2856        int numValues = readUnsignedShort(currentOffset);
2857        currentOffset += 2;
2858        if (numValues == 0) {
2859          return readElementValues(
2860              annotationVisitor.visitArray(elementName),
2861              currentOffset - 2,
2862              /* named = */ false,
2863              charBuffer);
2864        }
2865        switch (b[currentOffset] & 0xFF) {
2866          case 'B':
2867            byte[] byteValues = new byte[numValues];
2868            for (int i = 0; i < numValues; i++) {
2869              byteValues[i] = (byte) readInt(cpInfoOffsets[readUnsignedShort(currentOffset + 1)]);
2870              currentOffset += 3;
2871            }
2872            annotationVisitor.visit(elementName, byteValues);
2873            break;
2874          case 'Z':
2875            boolean[] booleanValues = new boolean[numValues];
2876            for (int i = 0; i < numValues; i++) {
2877              booleanValues[i] = readInt(cpInfoOffsets[readUnsignedShort(currentOffset + 1)]) != 0;
2878              currentOffset += 3;
2879            }
2880            annotationVisitor.visit(elementName, booleanValues);
2881            break;
2882          case 'S':
2883            short[] shortValues = new short[numValues];
2884            for (int i = 0; i < numValues; i++) {
2885              shortValues[i] = (short) readInt(cpInfoOffsets[readUnsignedShort(currentOffset + 1)]);
2886              currentOffset += 3;
2887            }
2888            annotationVisitor.visit(elementName, shortValues);
2889            break;
2890          case 'C':
2891            char[] charValues = new char[numValues];
2892            for (int i = 0; i < numValues; i++) {

3025   * field (this is used to parse the legacy StackMap attributes).
3026   *

3029   *     frame_type field).

3033   * @param context where the parsed stack map frame must be stored.
3034   * @return the end offset of the JVMS 'stack_map_frame' or 'full_frame' structure.

3037      final int stackMapFrameOffset,
3038      final boolean compressed,
3039      final boolean expand,
3040      final Context context) {
3041    int currentOffset = stackMapFrameOffset;
3042    final char[] charBuffer = context.charBuffer;
3043    final Label[] labels = context.currentMethodLabels;
3044    int frameType;
3045    if (compressed) {
3046      // Read the frame_type field.
3047      frameType = b[currentOffset++] & 0xFF;
3048    } else {
3049      frameType = Frame.FULL_FRAME;
3050      context.currentFrameOffset = -1;
3051    }
3052    int offsetDelta;
3053    context.currentFrameLocalCountDelta = 0;
3054    if (frameType < Frame.SAME_LOCALS_1_STACK_ITEM_FRAME) {
3055      offsetDelta = frameType;
3056      context.currentFrameType = Opcodes.F_SAME;
3057      context.currentFrameStackCount = 0;
3058    } else if (frameType < Frame.RESERVED) {
3059      offsetDelta = frameType - Frame.SAME_LOCALS_1_STACK_ITEM_FRAME;
3060      currentOffset =
3061          readVerificationTypeInfo(
3062              currentOffset, context.currentFrameStackTypes, 0, charBuffer, labels);
3063      context.currentFrameType = Opcodes.F_SAME1;
3064      context.currentFrameStackCount = 1;
3065    } else if (frameType >= Frame.SAME_LOCALS_1_STACK_ITEM_FRAME_EXTENDED) {
3066      offsetDelta = readUnsignedShort(currentOffset);
3067      currentOffset += 2;
3068      if (frameType == Frame.SAME_LOCALS_1_STACK_ITEM_FRAME_EXTENDED) {
3069        currentOffset =
3070            readVerificationTypeInfo(
3071                currentOffset, context.currentFrameStackTypes, 0, charBuffer, labels);
3072        context.currentFrameType = Opcodes.F_SAME1;
3073        context.currentFrameStackCount = 1;
3074      } else if (frameType >= Frame.CHOP_FRAME && frameType < Frame.SAME_FRAME_EXTENDED) {
3075        context.currentFrameType = Opcodes.F_CHOP;
3076        context.currentFrameLocalCountDelta = Frame.SAME_FRAME_EXTENDED - frameType;
3077        context.currentFrameLocalCount -= context.currentFrameLocalCountDelta;
3078        context.currentFrameStackCount = 0;
3079      } else if (frameType == Frame.SAME_FRAME_EXTENDED) {
3080        context.currentFrameType = Opcodes.F_SAME;
3081        context.currentFrameStackCount = 0;
3082      } else if (frameType < Frame.FULL_FRAME) {
3083        int local = expand ? context.currentFrameLocalCount : 0;
3084        for (int k = frameType - Frame.SAME_FRAME_EXTENDED; k > 0; k--) {
3085          currentOffset =
3086              readVerificationTypeInfo(
3087                  currentOffset, context.currentFrameLocalTypes, local++, charBuffer, labels);
3088        }
3089        context.currentFrameType = Opcodes.F_APPEND;
3090        context.currentFrameLocalCountDelta = frameType - Frame.SAME_FRAME_EXTENDED;
3091        context.currentFrameLocalCount += context.currentFrameLocalCountDelta;
3092        context.currentFrameStackCount = 0;
3093      } else {
3094        final int numberOfLocals = readUnsignedShort(currentOffset);
3095        currentOffset += 2;
3096        context.currentFrameType = Opcodes.F_FULL;
3097        context.currentFrameLocalCountDelta = numberOfLocals;
3098        context.currentFrameLocalCount = numberOfLocals;
3099        for (int local = 0; local < numberOfLocals; ++local) {
3100          currentOffset =
3101              readVerificationTypeInfo(
3102                  currentOffset, context.currentFrameLocalTypes, local, charBuffer, labels);
3103        }
3104        final int numberOfStackItems = readUnsignedShort(currentOffset);
3105        currentOffset += 2;
3106        context.currentFrameStackCount = numberOfStackItems;
3107        for (int stack = 0; stack < numberOfStackItems; ++stack) {
3108          currentOffset =
3109              readVerificationTypeInfo(
3110                  currentOffset, context.currentFrameStackTypes, stack, charBuffer, labels);
3111        }
3112      }
3113    } else {
3114      throw new IllegalArgumentException();
3115    }
3116    context.currentFrameOffset += offsetDelta + 1;
3117    createLabel(context.currentFrameOffset, labels);
3118    return currentOffset;
3119  }

3183  /** @return the offset in {@link #b} of the first ClassFile's 'attributes' array field entry. */
3184  final int getFirstAttributeOffset() {
3185    // Skip the access_flags, this_class, super_class, and interfaces_count fields (using 2 bytes
3186    // each), as well as the interfaces array field (2 bytes per interface).
3187    int currentOffset = header + 8 + readUnsignedShort(header + 6) * 2;
3188
3189    // Read the fields_count field.
3190    int fieldsCount = readUnsignedShort(currentOffset);
3191    currentOffset += 2;
3192    // Skip the 'fields' array field.
3193    while (fieldsCount-- > 0) {
3194      // Invariant: currentOffset is the offset of a field_info structure.
3195      // Skip the access_flags, name_index and descriptor_index fields (2 bytes each), and read the
3196      // attributes_count field.
3197      int attributesCount = readUnsignedShort(currentOffset + 6);
3198      currentOffset += 8;
3199      // Skip the 'attributes' array field.
3200      while (attributesCount-- > 0) {
3201        // Invariant: currentOffset is the offset of an attribute_info structure.
3202        // Read the attribute_length field (2 bytes after the start of the attribute_info) and skip
3203        // this many bytes, plus 6 for the attribute_name_index and attribute_length fields
3204        // (yielding the total size of the attribute_info structure).
3205        currentOffset += 6 + readInt(currentOffset + 2);
3206      }
3207    }
3208
3209    // Skip the methods_count and 'methods' fields, using the same method as above.
3210    int methodsCount = readUnsignedShort(currentOffset);
3211    currentOffset += 2;
3212    while (methodsCount-- > 0) {
3213      int attributesCount = readUnsignedShort(currentOffset + 6);
3214      currentOffset += 8;
3215      while (attributesCount-- > 0) {
3216        currentOffset += 6 + readInt(currentOffset + 2);
3217      }
3218    }
3219
3220    // Skip the ClassFile's attributes_count field.
3221    return currentOffset + 2;

3501    if (constantDynamic != null) {
3502      return constantDynamic;

3504    int cpInfoOffset = cpInfoOffsets[constantPoolEntryIndex];
3505    int nameAndTypeCpInfoOffset = cpInfoOffsets[readUnsignedShort(cpInfoOffset + 2)];
3506    String name = readUTF8(nameAndTypeCpInfoOffset, charBuffer);

3522   * Reads a numeric or string constant pool entry in {@link #b}. <i>This method is intended for
3523   * {@link Attribute} sub classes, and is normally not needed by class generators or adapters.</i>
3524   *
3525   * @param constantPoolEntryIndex the index of a CONSTANT_Integer, CONSTANT_Float, CONSTANT_Long,
3526   *     CONSTANT_Double, CONSTANT_Class, CONSTANT_String, CONSTANT_MethodType,
3527   *     CONSTANT_MethodHandle or CONSTANT_Dynamic entry in the class's constant pool.
3528   * @param charBuffer the buffer to be used to read strings. This buffer must be sufficiently
3529   *     large. It is not automatically resized.
3530   * @return the {@link Integer}, {@link Float}, {@link Long}, {@link Double}, {@link String},
3531   *     {@link Type}, {@link Handle} or {@link ConstantDynamic} corresponding to the specified
3532   *     constant pool entry.
3533   */
3534  public Object readConst(final int constantPoolEntryIndex, final char[] charBuffer) {
3535    int cpInfoOffset = cpInfoOffsets[constantPoolEntryIndex];
3536    switch (b[cpInfoOffset - 1]) {
3537      case Symbol.CONSTANT_INTEGER_TAG:
3538        return readInt(cpInfoOffset);
3539      case Symbol.CONSTANT_FLOAT_TAG:
3540        return Float.intBitsToFloat(readInt(cpInfoOffset));
3541      case Symbol.CONSTANT_LONG_TAG:
3542        return readLong(cpInfoOffset);
3543      case Symbol.CONSTANT_DOUBLE_TAG:
3544        return Double.longBitsToDouble(readLong(cpInfoOffset));
3545      case Symbol.CONSTANT_CLASS_TAG:
3546        return Type.getObjectType(readUTF8(cpInfoOffset, charBuffer));
3547      case Symbol.CONSTANT_STRING_TAG:
3548        return readUTF8(cpInfoOffset, charBuffer);
3549      case Symbol.CONSTANT_METHOD_TYPE_TAG:
3550        return Type.getMethodType(readUTF8(cpInfoOffset, charBuffer));
3551      case Symbol.CONSTANT_METHOD_HANDLE_TAG:
3552        int referenceKind = readByte(cpInfoOffset);
3553        int referenceCpInfoOffset = cpInfoOffsets[readUnsignedShort(cpInfoOffset + 1)];
3554        int nameAndTypeCpInfoOffset = cpInfoOffsets[readUnsignedShort(referenceCpInfoOffset + 2)];
3555        String owner = readClass(referenceCpInfoOffset, charBuffer);
3556        String name = readUTF8(nameAndTypeCpInfoOffset, charBuffer);
3557        String descriptor = readUTF8(nameAndTypeCpInfoOffset + 2, charBuffer);
3558        boolean isInterface =
3559            b[referenceCpInfoOffset - 1] == Symbol.CONSTANT_INTERFACE_METHODREF_TAG;
3560        return new Handle(referenceKind, owner, name, descriptor, isInterface);
3561      case Symbol.CONSTANT_DYNAMIC_TAG:
3562        return readConstantDynamic(constantPoolEntryIndex, charBuffer);
3563      default:
3564        throw new IllegalArgumentException();
3565    }
