028package clojure.asm.commons;
029
030import clojure.asm.Label;
031
032/**
033 * A code generator for switch statements.
034 *
035 * @author Juozas Baliuka
036 * @author Chris Nokleberg
037 * @author Eric Bruneton
038 */
039public interface TableSwitchGenerator {
040
041  /**
042   * Generates the code for a switch case.
043   *
044   * @param key the switch case key.
045   * @param end a label that corresponds to the end of the switch statement.
046   */
047  void generateCase(int key, Label end);
048
049  /** Generates the code for the default switch case. */
050  void generateDefault();
051}
