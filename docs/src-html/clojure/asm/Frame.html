030/**
031 * The input and output stack map frames of a basic block.
032 *
033 * <p>Stack map frames are computed in two steps:
034 *
035 * <ul>
036 *   <li>During the visit of each instruction in MethodWriter, the state of the frame at the end of
037 *       the current basic block is updated by simulating the action of the instruction on the
038 *       previous state of this so called "output frame".
039 *   <li>After all instructions have been visited, a fix point algorithm is used in MethodWriter to
040 *       compute the "input frame" of each basic block (i.e. the stack map frame at the beginning of
041 *       the basic block). See {@link MethodWriter#computeAllFrames}.
042 * </ul>
043 *
044 * <p>Output stack map frames are computed relatively to the input frame of the basic block, which
045 * is not yet known when output frames are computed. It is therefore necessary to be able to
046 * represent abstract types such as "the type at position x in the input frame locals" or "the type
047 * at position x from the top of the input frame stack" or even "the type at position x in the input
048 * frame, with y more (or less) array dimensions". This explains the rather complicated type format
049 * used in this class, explained below.
050 *
051 * <p>The local variables and the operand stack of input and output frames contain values called
052 * "abstract types" hereafter. An abstract type is represented with 4 fields named DIM, KIND, FLAGS
053 * and VALUE, packed in a single int value for better performance and memory efficiency:
054 *
055 * <pre>
056 *   =====================================
057 *   |.DIM|KIND|FLAG|...............VALUE|
058 *   =====================================
059 * </pre>
060 *
061 * <ul>
062 *   <li>the DIM field, stored in the 4 most significant bits, is a signed number of array
063 *       dimensions (from -8 to 7, included). It can be retrieved with {@link #DIM_MASK} and a right
064 *       shift of {@link #DIM_SHIFT}.
065 *   <li>the KIND field, stored in 4 bits, indicates the kind of VALUE used. These 4 bits can be
066 *       retrieved with {@link #KIND_MASK} and, without any shift, must be equal to {@link
067 *       #CONSTANT_KIND}, {@link #REFERENCE_KIND}, {@link #UNINITIALIZED_KIND}, {@link #LOCAL_KIND}
068 *       or {@link #STACK_KIND}.
069 *   <li>the FLAGS field, stored in 4 bits, contains up to 4 boolean flags. Currently only one flag
070 *       is defined, namely {@link #TOP_IF_LONG_OR_DOUBLE_FLAG}.
071 *   <li>the VALUE field, stored in the remaining 20 bits, contains either
072 *       <ul>
073 *         <li>one of the constants {@link #ITEM_TOP}, {@link #ITEM_ASM_BOOLEAN}, {@link
074 *             #ITEM_ASM_BYTE}, {@link #ITEM_ASM_CHAR} or {@link #ITEM_ASM_SHORT}, {@link
075 *             #ITEM_INTEGER}, {@link #ITEM_FLOAT}, {@link #ITEM_LONG}, {@link #ITEM_DOUBLE}, {@link
076 *             #ITEM_NULL} or {@link #ITEM_UNINITIALIZED_THIS}, if KIND is equal to {@link
077 *             #CONSTANT_KIND}.
078 *         <li>the index of a {@link Symbol#TYPE_TAG} {@link Symbol} in the type table of a {@link
079 *             SymbolTable}, if KIND is equal to {@link #REFERENCE_KIND}.
080 *         <li>the index of an {@link Symbol#UNINITIALIZED_TYPE_TAG} {@link Symbol} in the type
081 *             table of a SymbolTable, if KIND is equal to {@link #UNINITIALIZED_KIND}.
082 *         <li>the index of a local variable in the input stack frame, if KIND is equal to {@link
083 *             #LOCAL_KIND}.
084 *         <li>a position relatively to the top of the stack of the input stack frame, if KIND is
085 *             equal to {@link #STACK_KIND},
086 *       </ul>
087 * </ul>
088 *
089 * <p>Output frames can contain abstract types of any kind and with a positive or negative array
090 * dimension (and even unassigned types, represented by 0 - which does not correspond to any valid
091 * abstract type value). Input frames can only contain CONSTANT_KIND, REFERENCE_KIND or
092 * UNINITIALIZED_KIND abstract types of positive or null array dimension. In all cases the type
093 * table contains only internal type names (array type descriptors are forbidden - array dimensions
094 * must be represented through the DIM field).
095 *
096 * <p>The LONG and DOUBLE types are always represented by using two slots (LONG + TOP or DOUBLE +
097 * TOP), for local variables as well as in the operand stack. This is necessary to be able to
098 * simulate DUPx_y instructions, whose effect would be dependent on the concrete types represented
099 * by the abstract types in the stack (which are not always known).
100 *
101 * @author Eric Bruneton
102 */
103class Frame {

200  /**
201   * The start of the output stack, relatively to the input stack. This offset is always negative or

209  private short outputStackTop;

212  private int initializationCount;

214  /**
215   * The abstract types that are initialized in the basic block. A constructor invocation on an
216   * UNINITIALIZED or UNINITIALIZED_THIS abstract type must replace <i>every occurrence</i> of this
217   * type in the local variables and in the operand stack. This cannot be done during the first step
218   * of the algorithm since, during this step, the local variables and the operand stack types are
219   * still abstract. It is therefore necessary to store the abstract types of the constructors which
220   * are invoked in the basic block, in order to do this replacement during the second step of the
221   * algorithm, where the frames are fully computed. Note that this array can contain abstract types
222   * that are relative to the input locals or to the input stack.
223   */
224  private int[] initializations;

227  // Static methods to get abstract types from other type formats

230  /**
231   * Returns the abstract type corresponding to the given public API frame element type.
232   *
233   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
234   * @param type a frame element type described using the same format as in {@link
235   *     MethodVisitor#visitFrame}, i.e. either {@link Opcodes#TOP}, {@link Opcodes#INTEGER}, {@link
236   *     Opcodes#FLOAT}, {@link Opcodes#LONG}, {@link Opcodes#DOUBLE}, {@link Opcodes#NULL}, or
237   *     {@link Opcodes#UNINITIALIZED_THIS}, or the internal name of a class, or a Label designating
238   *     a NEW instruction (for uninitialized types).
239   * @return the abstract type corresponding to the given frame element type.
240   */
241  static int getAbstractTypeFromApiFormat(final SymbolTable symbolTable, final Object type) {
242    if (type instanceof Integer) {
243      return CONSTANT_KIND | ((Integer) type).intValue();
244    } else if (type instanceof String) {
245      String descriptor = Type.getObjectType((String) type).getDescriptor();
246      return getAbstractTypeFromDescriptor(symbolTable, descriptor, 0);
247    } else {
248      return UNINITIALIZED_KIND
249          | symbolTable.addUninitializedType("", ((Label) type).bytecodeOffset);
250    }
251  }
252
253  /**
254   * Returns the abstract type corresponding to the internal name of a class.
255   *
256   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
257   * @param internalName the internal name of a class. This must <i>not</i> be an array type
258   *     descriptor.
259   * @return the abstract type value corresponding to the given internal name.
260   */
261  static int getAbstractTypeFromInternalName(
262      final SymbolTable symbolTable, final String internalName) {
263    return REFERENCE_KIND | symbolTable.addType(internalName);
264  }
265
266  /**
267   * Returns the abstract type corresponding to the given type descriptor.
268   *
269   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
270   * @param buffer a string ending with a type descriptor.
271   * @param offset the start offset of the type descriptor in buffer.
272   * @return the abstract type corresponding to the given type descriptor.
273   */
274  private static int getAbstractTypeFromDescriptor(
275      final SymbolTable symbolTable, final String buffer, final int offset) {
276    String internalName;
277    switch (buffer.charAt(offset)) {
278      case 'V':
279        return 0;
280      case 'Z':
281      case 'C':
282      case 'B':
283      case 'S':
284      case 'I':
285        return INTEGER;
286      case 'F':
287        return FLOAT;
288      case 'J':
289        return LONG;
290      case 'D':
291        return DOUBLE;
292      case 'L':
293        internalName = buffer.substring(offset + 1, buffer.length() - 1);
294        return REFERENCE_KIND | symbolTable.addType(internalName);
295      case '[':
296        int elementDescriptorOffset = offset + 1;
297        while (buffer.charAt(elementDescriptorOffset) == '[') {
298          ++elementDescriptorOffset;
299        }
300        int typeValue;
301        switch (buffer.charAt(elementDescriptorOffset)) {
302          case 'Z':
303            typeValue = BOOLEAN;
304            break;
305          case 'C':
306            typeValue = CHAR;
307            break;
308          case 'B':
309            typeValue = BYTE;
310            break;
311          case 'S':
312            typeValue = SHORT;
313            break;
314          case 'I':
315            typeValue = INTEGER;
316            break;
317          case 'F':
318            typeValue = FLOAT;
319            break;
320          case 'J':
321            typeValue = LONG;
322            break;
323          case 'D':
324            typeValue = DOUBLE;
325            break;
326          case 'L':
327            internalName = buffer.substring(elementDescriptorOffset + 1, buffer.length() - 1);
328            typeValue = REFERENCE_KIND | symbolTable.addType(internalName);
329            break;
330          default:
331            throw new IllegalArgumentException();
332        }
333        return ((elementDescriptorOffset - offset) << DIM_SHIFT) | typeValue;
334      default:
335        throw new IllegalArgumentException();
336    }
337  }

352  /**
353   * Sets this frame to the value of the given frame.
354   *
355   * <p>WARNING: after this method is called the two frames share the same data structures. It is
356   * recommended to discard the given frame to avoid unexpected side effects.
357   *
358   * @param frame The new frame value.
359   */
360  final void copyFrom(final Frame frame) {
361    inputLocals = frame.inputLocals;
362    inputStack = frame.inputStack;
363    outputStackStart = 0;
364    outputLocals = frame.outputLocals;
365    outputStack = frame.outputStack;
366    outputStackTop = frame.outputStackTop;
367    initializationCount = frame.initializationCount;
368    initializations = frame.initializations;
369  }

371  // -----------------------------------------------------------------------------------------------
372  // Methods related to the input frame
373  // -----------------------------------------------------------------------------------------------
374
375  /**
376   * Sets the input frame from the given method description. This method is used to initialize the
377   * first frame of a method, which is implicit (i.e. not stored explicitly in the StackMapTable
378   * attribute).
379   *
380   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
381   * @param access the method's access flags.
382   * @param descriptor the method descriptor.
383   * @param maxLocals the maximum number of local variables of the method.
384   */
385  final void setInputFrameFromDescriptor(
386      final SymbolTable symbolTable,
387      final int access,
388      final String descriptor,
389      final int maxLocals) {
390    inputLocals = new int[maxLocals];
391    inputStack = new int[0];
392    int inputLocalIndex = 0;
393    if ((access & Opcodes.ACC_STATIC) == 0) {
394      if ((access & Constants.ACC_CONSTRUCTOR) == 0) {
395        inputLocals[inputLocalIndex++] =
396            REFERENCE_KIND | symbolTable.addType(symbolTable.getClassName());
397      } else {
398        inputLocals[inputLocalIndex++] = UNINITIALIZED_THIS;
399      }
400    }
401    for (Type argumentType : Type.getArgumentTypes(descriptor)) {
402      int abstractType =
403          getAbstractTypeFromDescriptor(symbolTable, argumentType.getDescriptor(), 0);
404      inputLocals[inputLocalIndex++] = abstractType;
405      if (abstractType == LONG || abstractType == DOUBLE) {
406        inputLocals[inputLocalIndex++] = TOP;
407      }
408    }
409    while (inputLocalIndex < maxLocals) {
410      inputLocals[inputLocalIndex++] = TOP;
411    }
412  }
413
414  /**
415   * Sets the input frame from the given public API frame description.
416   *
417   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
418   * @param nLocal the number of local variables.
419   * @param local the local variable types, described using the same format as in {@link
420   *     MethodVisitor#visitFrame}.
421   * @param nStack the number of operand stack elements.
422   * @param stack the operand stack types, described using the same format as in {@link
423   *     MethodVisitor#visitFrame}.
424   */
425  final void setInputFrameFromApiFormat(
426      final SymbolTable symbolTable,
427      final int nLocal,
428      final Object[] local,
429      final int nStack,
430      final Object[] stack) {
431    int inputLocalIndex = 0;
432    for (int i = 0; i < nLocal; ++i) {
433      inputLocals[inputLocalIndex++] = getAbstractTypeFromApiFormat(symbolTable, local[i]);
434      if (local[i] == Opcodes.LONG || local[i] == Opcodes.DOUBLE) {
435        inputLocals[inputLocalIndex++] = TOP;
436      }
437    }
438    while (inputLocalIndex < inputLocals.length) {
439      inputLocals[inputLocalIndex++] = TOP;
440    }
441    int nStackTop = 0;
442    for (int i = 0; i < nStack; ++i) {
443      if (stack[i] == Opcodes.LONG || stack[i] == Opcodes.DOUBLE) {
444        ++nStackTop;
445      }
446    }
447    inputStack = new int[nStack + nStackTop];
448    int inputStackIndex = 0;
449    for (int i = 0; i < nStack; ++i) {
450      inputStack[inputStackIndex++] = getAbstractTypeFromApiFormat(symbolTable, stack[i]);
451      if (stack[i] == Opcodes.LONG || stack[i] == Opcodes.DOUBLE) {
452        inputStack[inputStackIndex++] = TOP;
453      }
454    }
455    outputStackTop = 0;
456    initializationCount = 0;
457  }
458
459  final int getInputStackSize() {
460    return inputStack.length;
461  }

467  /**
468   * Returns the abstract type stored at the given local variable index in the output frame.
469   *
470   * @param localIndex the index of the local variable whose value must be returned.
471   * @return the abstract type stored at the given local variable index in the output frame.
472   */
473  private int getLocal(final int localIndex) {
474    if (outputLocals == null || localIndex >= outputLocals.length) {
475      // If this local has never been assigned in this basic block, it is still equal to its value
476      // in the input frame.
477      return LOCAL_KIND | localIndex;
478    } else {
479      int abstractType = outputLocals[localIndex];
480      if (abstractType == 0) {
481        // If this local has never been assigned in this basic block, so it is still equal to its
482        // value in the input frame.
483        abstractType = outputLocals[localIndex] = LOCAL_KIND | localIndex;
484      }
485      return abstractType;
486    }
487  }
488
489  /**
490   * Replaces the abstract type stored at the given local variable index in the output frame.
491   *
492   * @param localIndex the index of the output frame local variable that must be set.
493   * @param abstractType the value that must be set.
494   */
495  private void setLocal(final int localIndex, final int abstractType) {
496    // Create and/or resize the output local variables array if necessary.
497    if (outputLocals == null) {
498      outputLocals = new int[10];
499    }
500    int outputLocalsLength = outputLocals.length;
501    if (localIndex >= outputLocalsLength) {
502      int[] newOutputLocals = new int[Math.max(localIndex + 1, 2 * outputLocalsLength)];
503      System.arraycopy(outputLocals, 0, newOutputLocals, 0, outputLocalsLength);
504      outputLocals = newOutputLocals;
505    }
506    // Set the local variable.
507    outputLocals[localIndex] = abstractType;
508  }
509
510  /**
511   * Pushes the given abstract type on the output frame stack.
512   *
513   * @param abstractType an abstract type.
514   */
515  private void push(final int abstractType) {
516    // Create and/or resize the output stack array if necessary.
517    if (outputStack == null) {
518      outputStack = new int[10];
519    }
520    int outputStackLength = outputStack.length;
521    if (outputStackTop >= outputStackLength) {
522      int[] newOutputStack = new int[Math.max(outputStackTop + 1, 2 * outputStackLength)];
523      System.arraycopy(outputStack, 0, newOutputStack, 0, outputStackLength);
524      outputStack = newOutputStack;
525    }
526    // Pushes the abstract type on the output stack.
527    outputStack[outputStackTop++] = abstractType;
528    // Updates the maximum size reached by the output stack, if needed (note that this size is
529    // relative to the input stack size, which is not known yet).
530    short outputStackSize = (short) (outputStackStart + outputStackTop);
531    if (outputStackSize > owner.outputStackMax) {
532      owner.outputStackMax = outputStackSize;
533    }
534  }
535
536  /**
537   * Pushes the abstract type corresponding to the given descriptor on the output frame stack.
538   *
539   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
540   * @param descriptor a type or method descriptor (in which case its return type is pushed).
541   */
542  private void push(final SymbolTable symbolTable, final String descriptor) {
543    int typeDescriptorOffset = descriptor.charAt(0) == '(' ? descriptor.indexOf(')') + 1 : 0;
544    int abstractType = getAbstractTypeFromDescriptor(symbolTable, descriptor, typeDescriptorOffset);
545    if (abstractType != 0) {
546      push(abstractType);
547      if (abstractType == LONG || abstractType == DOUBLE) {
548        push(TOP);
549      }
550    }
551  }
552
553  /**
554   * Pops an abstract type from the output frame stack and returns its value.
555   *
556   * @return the abstract type that has been popped from the output frame stack.
557   */
558  private int pop() {
559    if (outputStackTop > 0) {
560      return outputStack[--outputStackTop];
561    } else {
562      // If the output frame stack is empty, pop from the input stack.
563      return STACK_KIND | -(--outputStackStart);
564    }
565  }
566
567  /**
568   * Pops the given number of abstract types from the output frame stack.
569   *
570   * @param elements the number of abstract types that must be popped.
571   */
572  private void pop(final int elements) {
573    if (outputStackTop >= elements) {
574      outputStackTop -= elements;
575    } else {
576      // If the number of elements to be popped is greater than the number of elements in the output
577      // stack, clear it, and pop the remaining elements from the input stack.
578      outputStackStart -= elements - outputStackTop;
579      outputStackTop = 0;
580    }
581  }
582
583  /**
584   * Pops as many abstract types from the output frame stack as described by the given descriptor.
585   *
586   * @param descriptor a type or method descriptor (in which case its argument types are popped).
587   */
588  private void pop(final String descriptor) {
589    char firstDescriptorChar = descriptor.charAt(0);
590    if (firstDescriptorChar == '(') {
591      pop((Type.getArgumentsAndReturnSizes(descriptor) >> 2) - 1);
592    } else if (firstDescriptorChar == 'J' || firstDescriptorChar == 'D') {
593      pop(2);
594    } else {
595      pop(1);
596    }
597  }

603  /**
604   * Adds an abstract type to the list of types on which a constructor is invoked in the basic
605   * block.
606   *
607   * @param abstractType an abstract type on a which a constructor is invoked.
608   */
609  private void addInitializedType(final int abstractType) {
610    // Create and/or resize the initializations array if necessary.
611    if (initializations == null) {
612      initializations = new int[2];
613    }
614    int initializationsLength = initializations.length;
615    if (initializationCount >= initializationsLength) {
616      int[] newInitializations =
617          new int[Math.max(initializationCount + 1, 2 * initializationsLength)];
618      System.arraycopy(initializations, 0, newInitializations, 0, initializationsLength);
619      initializations = newInitializations;
620    }
621    // Store the abstract type.
622    initializations[initializationCount++] = abstractType;
623  }
624
625  /**
626   * Returns the "initialized" abstract type corresponding to the given abstract type.
627   *
628   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
629   * @param abstractType an abstract type.
630   * @return the REFERENCE_KIND abstract type corresponding to abstractType if it is
631   *     UNINITIALIZED_THIS or an UNINITIALIZED_KIND abstract type for one of the types on which a
632   *     constructor is invoked in the basic block. Otherwise returns abstractType.
633   */
634  private int getInitializedType(final SymbolTable symbolTable, final int abstractType) {
635    if (abstractType == UNINITIALIZED_THIS
636        || (abstractType & (DIM_MASK | KIND_MASK)) == UNINITIALIZED_KIND) {
637      for (int i = 0; i < initializationCount; ++i) {
638        int initializedType = initializations[i];
639        int dim = initializedType & DIM_MASK;
640        int kind = initializedType & KIND_MASK;
641        int value = initializedType & VALUE_MASK;
642        if (kind == LOCAL_KIND) {
643          initializedType = dim + inputLocals[value];
644        } else if (kind == STACK_KIND) {
645          initializedType = dim + inputStack[inputStack.length - value];
646        }
647        if (abstractType == initializedType) {
648          if (abstractType == UNINITIALIZED_THIS) {
649            return REFERENCE_KIND | symbolTable.addType(symbolTable.getClassName());
650          } else {
651            return REFERENCE_KIND
652                | symbolTable.addType(symbolTable.getType(abstractType & VALUE_MASK).value);
653          }
654        }
655      }
656    }
657    return abstractType;
658  }

660  // -----------------------------------------------------------------------------------------------
661  // Main method, to simulate the execution of each instruction on the output frame
662  // -----------------------------------------------------------------------------------------------
663
664  /**
665   * Simulates the action of the given instruction on the output stack frame.
666   *
667   * @param opcode the opcode of the instruction.
668   * @param arg the numeric operand of the instruction, if any.
669   * @param argSymbol the Symbol operand of the instruction, if any.
670   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
671   */
672  void execute(
673      final int opcode, final int arg, final Symbol argSymbol, final SymbolTable symbolTable) {
674    // Abstract types popped from the stack or read from local variables.
675    int abstractType1;
676    int abstractType2;
677    int abstractType3;
678    int abstractType4;
679    switch (opcode) {
680      case Opcodes.NOP:
681      case Opcodes.INEG:
682      case Opcodes.LNEG:
683      case Opcodes.FNEG:
684      case Opcodes.DNEG:
685      case Opcodes.I2B:
686      case Opcodes.I2C:
687      case Opcodes.I2S:
688      case Opcodes.GOTO:
689      case Opcodes.RETURN:
690        break;
691      case Opcodes.ACONST_NULL:
692        push(NULL);
693        break;
694      case Opcodes.ICONST_M1:
695      case Opcodes.ICONST_0:
696      case Opcodes.ICONST_1:
697      case Opcodes.ICONST_2:
698      case Opcodes.ICONST_3:
699      case Opcodes.ICONST_4:
700      case Opcodes.ICONST_5:
701      case Opcodes.BIPUSH:
702      case Opcodes.SIPUSH:
703      case Opcodes.ILOAD:
704        push(INTEGER);
705        break;
706      case Opcodes.LCONST_0:
707      case Opcodes.LCONST_1:
708      case Opcodes.LLOAD:
709        push(LONG);
710        push(TOP);
711        break;
712      case Opcodes.FCONST_0:
713      case Opcodes.FCONST_1:
714      case Opcodes.FCONST_2:
715      case Opcodes.FLOAD:
716        push(FLOAT);
717        break;
718      case Opcodes.DCONST_0:
719      case Opcodes.DCONST_1:
720      case Opcodes.DLOAD:
721        push(DOUBLE);
722        push(TOP);
723        break;
724      case Opcodes.LDC:
725        switch (argSymbol.tag) {
726          case Symbol.CONSTANT_INTEGER_TAG:
727            push(INTEGER);
728            break;
729          case Symbol.CONSTANT_LONG_TAG:
730            push(LONG);
731            push(TOP);
732            break;
733          case Symbol.CONSTANT_FLOAT_TAG:
734            push(FLOAT);
735            break;
736          case Symbol.CONSTANT_DOUBLE_TAG:
737            push(DOUBLE);
738            push(TOP);
739            break;
740          case Symbol.CONSTANT_CLASS_TAG:
741            push(REFERENCE_KIND | symbolTable.addType("java/lang/Class"));
742            break;
743          case Symbol.CONSTANT_STRING_TAG:
744            push(REFERENCE_KIND | symbolTable.addType("java/lang/String"));
745            break;
746          case Symbol.CONSTANT_METHOD_TYPE_TAG:
747            push(REFERENCE_KIND | symbolTable.addType("java/lang/invoke/MethodType"));
748            break;
749          case Symbol.CONSTANT_METHOD_HANDLE_TAG:
750            push(REFERENCE_KIND | symbolTable.addType("java/lang/invoke/MethodHandle"));
751            break;
752          case Symbol.CONSTANT_DYNAMIC_TAG:
753            push(symbolTable, argSymbol.value);
754            break;
755          default:
756            throw new AssertionError();
757        }
758        break;
759      case Opcodes.ALOAD:
760        push(getLocal(arg));
761        break;
762      case Opcodes.LALOAD:
763      case Opcodes.D2L:
764        pop(2);
765        push(LONG);
766        push(TOP);
767        break;
768      case Opcodes.DALOAD:
769      case Opcodes.L2D:
770        pop(2);
771        push(DOUBLE);
772        push(TOP);
773        break;
774      case Opcodes.AALOAD:
775        pop(1);
776        abstractType1 = pop();
777        push(abstractType1 == NULL ? abstractType1 : ELEMENT_OF + abstractType1);
778        break;
779      case Opcodes.ISTORE:
780      case Opcodes.FSTORE:
781      case Opcodes.ASTORE:
782        abstractType1 = pop();
783        setLocal(arg, abstractType1);
784        if (arg > 0) {
785          int previousLocalType = getLocal(arg - 1);
786          if (previousLocalType == LONG || previousLocalType == DOUBLE) {
787            setLocal(arg - 1, TOP);
788          } else if ((previousLocalType & KIND_MASK) == LOCAL_KIND
789              || (previousLocalType & KIND_MASK) == STACK_KIND) {
790            // The type of the previous local variable is not known yet, but if it later appears
791            // to be LONG or DOUBLE, we should then use TOP instead.
792            setLocal(arg - 1, previousLocalType | TOP_IF_LONG_OR_DOUBLE_FLAG);
793          }
794        }
795        break;
796      case Opcodes.LSTORE:
797      case Opcodes.DSTORE:
798        pop(1);
799        abstractType1 = pop();
800        setLocal(arg, abstractType1);
801        setLocal(arg + 1, TOP);
802        if (arg > 0) {
803          int previousLocalType = getLocal(arg - 1);
804          if (previousLocalType == LONG || previousLocalType == DOUBLE) {
805            setLocal(arg - 1, TOP);
806          } else if ((previousLocalType & KIND_MASK) == LOCAL_KIND
807              || (previousLocalType & KIND_MASK) == STACK_KIND) {
808            // The type of the previous local variable is not known yet, but if it later appears
809            // to be LONG or DOUBLE, we should then use TOP instead.
810            setLocal(arg - 1, previousLocalType | TOP_IF_LONG_OR_DOUBLE_FLAG);
811          }
812        }
813        break;
814      case Opcodes.IASTORE:
815      case Opcodes.BASTORE:
816      case Opcodes.CASTORE:
817      case Opcodes.SASTORE:
818      case Opcodes.FASTORE:
819      case Opcodes.AASTORE:
820        pop(3);
821        break;
822      case Opcodes.LASTORE:
823      case Opcodes.DASTORE:
824        pop(4);
825        break;
826      case Opcodes.POP:
827      case Opcodes.IFEQ:
828      case Opcodes.IFNE:
829      case Opcodes.IFLT:
830      case Opcodes.IFGE:
831      case Opcodes.IFGT:
832      case Opcodes.IFLE:
833      case Opcodes.IRETURN:
834      case Opcodes.FRETURN:
835      case Opcodes.ARETURN:
836      case Opcodes.TABLESWITCH:
837      case Opcodes.LOOKUPSWITCH:
838      case Opcodes.ATHROW:
839      case Opcodes.MONITORENTER:
840      case Opcodes.MONITOREXIT:
841      case Opcodes.IFNULL:
842      case Opcodes.IFNONNULL:
843        pop(1);
844        break;
845      case Opcodes.POP2:
846      case Opcodes.IF_ICMPEQ:
847      case Opcodes.IF_ICMPNE:
848      case Opcodes.IF_ICMPLT:
849      case Opcodes.IF_ICMPGE:
850      case Opcodes.IF_ICMPGT:
851      case Opcodes.IF_ICMPLE:
852      case Opcodes.IF_ACMPEQ:
853      case Opcodes.IF_ACMPNE:
854      case Opcodes.LRETURN:
855      case Opcodes.DRETURN:
856        pop(2);
857        break;
858      case Opcodes.DUP:
859        abstractType1 = pop();
860        push(abstractType1);
861        push(abstractType1);
862        break;
863      case Opcodes.DUP_X1:
864        abstractType1 = pop();
865        abstractType2 = pop();
866        push(abstractType1);
867        push(abstractType2);
868        push(abstractType1);
869        break;
870      case Opcodes.DUP_X2:
871        abstractType1 = pop();
872        abstractType2 = pop();
873        abstractType3 = pop();
874        push(abstractType1);
875        push(abstractType3);
876        push(abstractType2);
877        push(abstractType1);
878        break;
879      case Opcodes.DUP2:
880        abstractType1 = pop();
881        abstractType2 = pop();
882        push(abstractType2);
883        push(abstractType1);
884        push(abstractType2);
885        push(abstractType1);
886        break;
887      case Opcodes.DUP2_X1:
888        abstractType1 = pop();
889        abstractType2 = pop();
890        abstractType3 = pop();
891        push(abstractType2);
892        push(abstractType1);
893        push(abstractType3);
894        push(abstractType2);
895        push(abstractType1);
896        break;
897      case Opcodes.DUP2_X2:
898        abstractType1 = pop();
899        abstractType2 = pop();
900        abstractType3 = pop();
901        abstractType4 = pop();
902        push(abstractType2);
903        push(abstractType1);
904        push(abstractType4);
905        push(abstractType3);
906        push(abstractType2);
907        push(abstractType1);
908        break;
909      case Opcodes.SWAP:
910        abstractType1 = pop();
911        abstractType2 = pop();
912        push(abstractType1);
913        push(abstractType2);
914        break;
915      case Opcodes.IALOAD:
916      case Opcodes.BALOAD:
917      case Opcodes.CALOAD:
918      case Opcodes.SALOAD:
919      case Opcodes.IADD:
920      case Opcodes.ISUB:
921      case Opcodes.IMUL:
922      case Opcodes.IDIV:
923      case Opcodes.IREM:
924      case Opcodes.IAND:
925      case Opcodes.IOR:
926      case Opcodes.IXOR:
927      case Opcodes.ISHL:
928      case Opcodes.ISHR:
929      case Opcodes.IUSHR:
930      case Opcodes.L2I:
931      case Opcodes.D2I:
932      case Opcodes.FCMPL:
933      case Opcodes.FCMPG:
934        pop(2);
935        push(INTEGER);
936        break;
937      case Opcodes.LADD:
938      case Opcodes.LSUB:
939      case Opcodes.LMUL:
940      case Opcodes.LDIV:
941      case Opcodes.LREM:
942      case Opcodes.LAND:
943      case Opcodes.LOR:
944      case Opcodes.LXOR:
945        pop(4);
946        push(LONG);
947        push(TOP);
948        break;
949      case Opcodes.FALOAD:
950      case Opcodes.FADD:
951      case Opcodes.FSUB:
952      case Opcodes.FMUL:
953      case Opcodes.FDIV:
954      case Opcodes.FREM:
955      case Opcodes.L2F:
956      case Opcodes.D2F:
957        pop(2);
958        push(FLOAT);
959        break;
960      case Opcodes.DADD:
961      case Opcodes.DSUB:
962      case Opcodes.DMUL:
963      case Opcodes.DDIV:
964      case Opcodes.DREM:
965        pop(4);
966        push(DOUBLE);
967        push(TOP);
968        break;
969      case Opcodes.LSHL:
970      case Opcodes.LSHR:
971      case Opcodes.LUSHR:
972        pop(3);
973        push(LONG);
974        push(TOP);
975        break;
976      case Opcodes.IINC:
977        setLocal(arg, INTEGER);
978        break;
979      case Opcodes.I2L:
980      case Opcodes.F2L:
981        pop(1);
982        push(LONG);
983        push(TOP);
984        break;
985      case Opcodes.I2F:
986        pop(1);
987        push(FLOAT);
988        break;
989      case Opcodes.I2D:
990      case Opcodes.F2D:
991        pop(1);
992        push(DOUBLE);
993        push(TOP);
994        break;
995      case Opcodes.F2I:
996      case Opcodes.ARRAYLENGTH:
997      case Opcodes.INSTANCEOF:
998        pop(1);
999        push(INTEGER);
1000        break;
1001      case Opcodes.LCMP:
1002      case Opcodes.DCMPL:
1003      case Opcodes.DCMPG:
1004        pop(4);
1005        push(INTEGER);
1006        break;
1007      case Opcodes.JSR:
1008      case Opcodes.RET:
1009        throw new IllegalArgumentException("JSR/RET are not supported with computeFrames option");
1010      case Opcodes.GETSTATIC:
1011        push(symbolTable, argSymbol.value);
1012        break;
1013      case Opcodes.PUTSTATIC:
1014        pop(argSymbol.value);
1015        break;
1016      case Opcodes.GETFIELD:
1017        pop(1);
1018        push(symbolTable, argSymbol.value);
1019        break;
1020      case Opcodes.PUTFIELD:
1021        pop(argSymbol.value);
1022        pop();
1023        break;
1024      case Opcodes.INVOKEVIRTUAL:
1025      case Opcodes.INVOKESPECIAL:
1026      case Opcodes.INVOKESTATIC:
1027      case Opcodes.INVOKEINTERFACE:
1028        pop(argSymbol.value);
1029        if (opcode != Opcodes.INVOKESTATIC) {
1030          abstractType1 = pop();
1031          if (opcode == Opcodes.INVOKESPECIAL && argSymbol.name.charAt(0) == '<') {
1032            addInitializedType(abstractType1);
1033          }
1034        }
1035        push(symbolTable, argSymbol.value);
1036        break;
1037      case Opcodes.INVOKEDYNAMIC:
1038        pop(argSymbol.value);
1039        push(symbolTable, argSymbol.value);
1040        break;
1041      case Opcodes.NEW:
1042        push(UNINITIALIZED_KIND | symbolTable.addUninitializedType(argSymbol.value, arg));
1043        break;
1044      case Opcodes.NEWARRAY:
1045        pop();
1046        switch (arg) {
1047          case Opcodes.T_BOOLEAN:
1048            push(ARRAY_OF | BOOLEAN);
1049            break;
1050          case Opcodes.T_CHAR:
1051            push(ARRAY_OF | CHAR);
1052            break;
1053          case Opcodes.T_BYTE:
1054            push(ARRAY_OF | BYTE);
1055            break;
1056          case Opcodes.T_SHORT:
1057            push(ARRAY_OF | SHORT);
1058            break;
1059          case Opcodes.T_INT:
1060            push(ARRAY_OF | INTEGER);
1061            break;
1062          case Opcodes.T_FLOAT:
1063            push(ARRAY_OF | FLOAT);
1064            break;
1065          case Opcodes.T_DOUBLE:
1066            push(ARRAY_OF | DOUBLE);
1067            break;
1068          case Opcodes.T_LONG:
1069            push(ARRAY_OF | LONG);
1070            break;
1071          default:
1072            throw new IllegalArgumentException();
1073        }
1074        break;
1075      case Opcodes.ANEWARRAY:
1076        String arrayElementType = argSymbol.value;
1077        pop();
1078        if (arrayElementType.charAt(0) == '[') {
1079          push(symbolTable, '[' + arrayElementType);
1080        } else {
1081          push(ARRAY_OF | REFERENCE_KIND | symbolTable.addType(arrayElementType));
1082        }
1083        break;
1084      case Opcodes.CHECKCAST:
1085        String castType = argSymbol.value;
1086        pop();
1087        if (castType.charAt(0) == '[') {
1088          push(symbolTable, castType);
1089        } else {
1090          push(REFERENCE_KIND | symbolTable.addType(castType));
1091        }
1092        break;
1093      case Opcodes.MULTIANEWARRAY:
1094        pop(arg);
1095        push(symbolTable, argSymbol.value);
1096        break;
1097      default:
1098        throw new IllegalArgumentException();
1099    }
1100  }

1102  // -----------------------------------------------------------------------------------------------
1103  // Frame merging methods, used in the second step of the stack map frame computation algorithm
1104  // -----------------------------------------------------------------------------------------------
1105
1106  /**
1107   * Merges the input frame of the given {@link Frame} with the input and output frames of this
1108   * {@link Frame}. Returns <tt>true</tt> if the given frame has been changed by this operation (the
1109   * input and output frames of this {@link Frame} are never changed).
1110   *
1111   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
1112   * @param dstFrame the {@link Frame} whose input frame must be updated. This should be the frame
1113   *     of a successor, in the control flow graph, of the basic block corresponding to this frame.
1114   * @param catchTypeIndex if 'frame' corresponds to an exception handler basic block, the type
1115   *     table index of the caught exception type, otherwise 0.
1116   * @return <tt>true</tt> if the input frame of 'frame' has been changed by this operation.
1117   */
1118  final boolean merge(
1119      final SymbolTable symbolTable, final Frame dstFrame, final int catchTypeIndex) {
1120    boolean frameChanged = false;
1121
1122    // Compute the concrete types of the local variables at the end of the basic block corresponding
1123    // to this frame, by resolving its abstract output types, and merge these concrete types with
1124    // those of the local variables in the input frame of dstFrame.
1125    int nLocal = inputLocals.length;
1126    int nStack = inputStack.length;
1127    if (dstFrame.inputLocals == null) {
1128      dstFrame.inputLocals = new int[nLocal];
1129      frameChanged = true;
1130    }
1131    for (int i = 0; i < nLocal; ++i) {
1132      int concreteOutputType;
1133      if (outputLocals != null && i < outputLocals.length) {
1134        int abstractOutputType = outputLocals[i];
1135        if (abstractOutputType == 0) {
1136          // If the local variable has never been assigned in this basic block, it is equal to its
1137          // value at the beginning of the block.
1138          concreteOutputType = inputLocals[i];
1139        } else {
1140          int dim = abstractOutputType & DIM_MASK;
1141          int kind = abstractOutputType & KIND_MASK;
1142          if (kind == LOCAL_KIND) {
1143            // By definition, a LOCAL_KIND type designates the concrete type of a local variable at
1144            // the beginning of the basic block corresponding to this frame (which is known when
1145            // this method is called, but was not when the abstract type was computed).
1146            concreteOutputType = dim + inputLocals[abstractOutputType & VALUE_MASK];
1147            if ((abstractOutputType & TOP_IF_LONG_OR_DOUBLE_FLAG) != 0
1148                && (concreteOutputType == LONG || concreteOutputType == DOUBLE)) {
1149              concreteOutputType = TOP;
1150            }
1151          } else if (kind == STACK_KIND) {
1152            // By definition, a STACK_KIND type designates the concrete type of a local variable at
1153            // the beginning of the basic block corresponding to this frame (which is known when
1154            // this method is called, but was not when the abstract type was computed).
1155            concreteOutputType = dim + inputStack[nStack - (abstractOutputType & VALUE_MASK)];
1156            if ((abstractOutputType & TOP_IF_LONG_OR_DOUBLE_FLAG) != 0
1157                && (concreteOutputType == LONG || concreteOutputType == DOUBLE)) {
1158              concreteOutputType = TOP;
1159            }
1160          } else {
1161            concreteOutputType = abstractOutputType;
1162          }
1163        }
1164      } else {
1165        // If the local variable has never been assigned in this basic block, it is equal to its
1166        // value at the beginning of the block.
1167        concreteOutputType = inputLocals[i];
1168      }
1169      // concreteOutputType might be an uninitialized type from the input locals or from the input
1170      // stack. However, if a constructor has been called for this class type in the basic block,
1171      // then this type is no longer uninitialized at the end of basic block.
1172      if (initializations != null) {
1173        concreteOutputType = getInitializedType(symbolTable, concreteOutputType);
1174      }
1175      frameChanged |= merge(symbolTable, concreteOutputType, dstFrame.inputLocals, i);
1176    }
1177
1178    // If dstFrame is an exception handler block, it can be reached from any instruction of the
1179    // basic block corresponding to this frame, in particular from the first one. Therefore, the
1180    // input locals of dstFrame should be compatible (i.e. merged) with the input locals of this
1181    // frame (and the input stack of dstFrame should be compatible, i.e. merged, with a one
1182    // element stack containing the caught exception type).
1183    if (catchTypeIndex > 0) {
1184      for (int i = 0; i < nLocal; ++i) {
1185        frameChanged |= merge(symbolTable, inputLocals[i], dstFrame.inputLocals, i);
1186      }
1187      if (dstFrame.inputStack == null) {
1188        dstFrame.inputStack = new int[1];
1189        frameChanged = true;
1190      }
1191      frameChanged |= merge(symbolTable, catchTypeIndex, dstFrame.inputStack, 0);
1192      return frameChanged;
1193    }
1194
1195    // Compute the concrete types of the stack operands at the end of the basic block corresponding
1196    // to this frame, by resolving its abstract output types, and merge these concrete types with
1197    // those of the stack operands in the input frame of dstFrame.
1198    int nInputStack = inputStack.length + outputStackStart;
1199    if (dstFrame.inputStack == null) {
1200      dstFrame.inputStack = new int[nInputStack + outputStackTop];
1201      frameChanged = true;
1202    }
1203    // First, do this for the stack operands that have not been popped in the basic block
1204    // corresponding to this frame, and which are therefore equal to their value in the input
1205    // frame (except for uninitialized types, which may have been initialized).
1206    for (int i = 0; i < nInputStack; ++i) {
1207      int concreteOutputType = inputStack[i];
1208      if (initializations != null) {
1209        concreteOutputType = getInitializedType(symbolTable, concreteOutputType);
1210      }
1211      frameChanged |= merge(symbolTable, concreteOutputType, dstFrame.inputStack, i);
1212    }
1213    // Then, do this for the stack operands that have pushed in the basic block (this code is the
1214    // same as the one above for local variables).
1215    for (int i = 0; i < outputStackTop; ++i) {
1216      int concreteOutputType;
1217      int abstractOutputType = outputStack[i];
1218      int dim = abstractOutputType & DIM_MASK;
1219      int kind = abstractOutputType & KIND_MASK;
1220      if (kind == LOCAL_KIND) {
1221        concreteOutputType = dim + inputLocals[abstractOutputType & VALUE_MASK];
1222        if ((abstractOutputType & TOP_IF_LONG_OR_DOUBLE_FLAG) != 0
1223            && (concreteOutputType == LONG || concreteOutputType == DOUBLE)) {
1224          concreteOutputType = TOP;
1225        }
1226      } else if (kind == STACK_KIND) {
1227        concreteOutputType = dim + inputStack[nStack - (abstractOutputType & VALUE_MASK)];
1228        if ((abstractOutputType & TOP_IF_LONG_OR_DOUBLE_FLAG) != 0
1229            && (concreteOutputType == LONG || concreteOutputType == DOUBLE)) {
1230          concreteOutputType = TOP;
1231        }
1232      } else {
1233        concreteOutputType = abstractOutputType;
1234      }
1235      if (initializations != null) {
1236        concreteOutputType = getInitializedType(symbolTable, concreteOutputType);
1237      }
1238      frameChanged |= merge(symbolTable, concreteOutputType, dstFrame.inputStack, nInputStack + i);
1239    }
1240    return frameChanged;
1241  }

1243  /**
1244   * Merges the type at the given index in the given abstract type array with the given type.
1245   * Returns <tt>true</tt> if the type array has been modified by this operation.
1246   *
1247   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
1248   * @param sourceType the abstract type with which the abstract type array element must be merged.
1249   *     This type should be of {@link #CONSTANT_KIND}, {@link #REFERENCE_KIND} or {@link
1250   *     #UNINITIALIZED_KIND} kind, with positive or null array dimensions.
1251   * @param dstTypes an array of abstract types. These types should be of {@link #CONSTANT_KIND},
1252   *     {@link #REFERENCE_KIND} or {@link #UNINITIALIZED_KIND} kind, with positive or null array
1253   *     dimensions.
1254   * @param dstIndex the index of the type that must be merged in dstTypes.
1255   * @return <tt>true</tt> if the type array has been modified by this operation.
1256   */
1257  private static boolean merge(
1258      final SymbolTable symbolTable,
1259      final int sourceType,
1260      final int[] dstTypes,
1261      final int dstIndex) {
1262    int dstType = dstTypes[dstIndex];
1263    if (dstType == sourceType) {
1264      // If the types are equal, merge(sourceType, dstType) = dstType, so there is no change.
1265      return false;
1266    }
1267    int srcType = sourceType;
1268    if ((sourceType & ~DIM_MASK) == NULL) {
1269      if (dstType == NULL) {
1270        return false;
1271      }
1272      srcType = NULL;
1273    }
1274    if (dstType == 0) {
1275      // If dstTypes[dstIndex] has never been assigned, merge(srcType, dstType) = srcType.
1276      dstTypes[dstIndex] = srcType;
1277      return true;
1278    }
1279    int mergedType;
1280    if ((dstType & DIM_MASK) != 0 || (dstType & KIND_MASK) == REFERENCE_KIND) {
1281      // If dstType is a reference type of any array dimension.
1282      if (srcType == NULL) {
1283        // If srcType is the NULL type, merge(srcType, dstType) = dstType, so there is no change.
1284        return false;
1285      } else if ((srcType & (DIM_MASK | KIND_MASK)) == (dstType & (DIM_MASK | KIND_MASK))) {
1286        // If srcType has the same array dimension and the same kind as dstType.
1287        if ((dstType & KIND_MASK) == REFERENCE_KIND) {
1288          // If srcType and dstType are reference types with the same array dimension,
1289          // merge(srcType, dstType) = dim(srcType) | common super class of srcType and dstType.
1290          mergedType =
1291              (srcType & DIM_MASK)
1292                  | REFERENCE_KIND
1293                  | symbolTable.addMergedType(srcType & VALUE_MASK, dstType & VALUE_MASK);
1294        } else {
1295          // If srcType and dstType are array types of equal dimension but different element types,
1296          // merge(srcType, dstType) = dim(srcType) - 1 | java/lang/Object.
1297          int mergedDim = ELEMENT_OF + (srcType & DIM_MASK);
1298          mergedType = mergedDim | REFERENCE_KIND | symbolTable.addType("java/lang/Object");
1299        }
1300      } else if ((srcType & DIM_MASK) != 0 || (srcType & KIND_MASK) == REFERENCE_KIND) {
1301        // If srcType is any other reference or array type,
1302        // merge(srcType, dstType) = min(srcDdim, dstDim) | java/lang/Object
1303        // where srcDim is the array dimension of srcType, minus 1 if srcType is an array type
1304        // with a non reference element type (and similarly for dstDim).
1305        int srcDim = srcType & DIM_MASK;
1306        if (srcDim != 0 && (srcType & KIND_MASK) != REFERENCE_KIND) {
1307          srcDim = ELEMENT_OF + srcDim;
1308        }
1309        int dstDim = dstType & DIM_MASK;
1310        if (dstDim != 0 && (dstType & KIND_MASK) != REFERENCE_KIND) {
1311          dstDim = ELEMENT_OF + dstDim;
1312        }
1313        mergedType =
1314            Math.min(srcDim, dstDim) | REFERENCE_KIND | symbolTable.addType("java/lang/Object");
1315      } else {
1316        // If srcType is any other type, merge(srcType, dstType) = TOP.
1317        mergedType = TOP;
1318      }
1319    } else if (dstType == NULL) {
1320      // If dstType is the NULL type, merge(srcType, dstType) = srcType, or TOP if srcType is not a
1321      // an array type or a reference type.
1322      mergedType =
1323          (srcType & DIM_MASK) != 0 || (srcType & KIND_MASK) == REFERENCE_KIND ? srcType : TOP;
1324    } else {
1325      // If dstType is any other type, merge(srcType, dstType) = TOP whatever srcType.
1326      mergedType = TOP;
1327    }
1328    if (mergedType != dstType) {
1329      dstTypes[dstIndex] = mergedType;
1330      return true;
1331    }
1332    return false;
1333  }

1339  /**
1340   * Makes the given {@link MethodWriter} visit the input frame of this {@link Frame}. The visit is
1341   * done with the {@link MethodWriter#visitFrameStart}, {@link MethodWriter#visitAbstractType} and
1342   * {@link MethodWriter#visitFrameEnd} methods.
1343   *
1344   * @param methodWriter the {@link MethodWriter} that should visit the input frame of this {@link
1345   *     Frame}.
1346   */
1347  final void accept(final MethodWriter methodWriter) {
1348    // Compute the number of locals, ignoring TOP types that are just after a LONG or a DOUBLE, and
1349    // all trailing TOP types.
1350    int[] localTypes = inputLocals;
1351    int nLocal = 0;
1352    int nTrailingTop = 0;
1353    int i = 0;
1354    while (i < localTypes.length) {
1355      int localType = localTypes[i];
1356      i += (localType == LONG || localType == DOUBLE) ? 2 : 1;
1357      if (localType == TOP) {
1358        nTrailingTop++;
1359      } else {
1360        nLocal += nTrailingTop + 1;
1361        nTrailingTop = 0;
1362      }
1363    }
1364    // Compute the stack size, ignoring TOP types that are just after a LONG or a DOUBLE.
1365    int[] stackTypes = inputStack;
1366    int nStack = 0;
1367    i = 0;
1368    while (i < stackTypes.length) {
1369      int stackType = stackTypes[i];
1370      i += (stackType == LONG || stackType == DOUBLE) ? 2 : 1;
1371      nStack++;
1372    }
1373    // Visit the frame and its content.
1374    int frameIndex = methodWriter.visitFrameStart(owner.bytecodeOffset, nLocal, nStack);
1375    i = 0;
1376    while (nLocal-- > 0) {
1377      int localType = localTypes[i];
1378      i += (localType == LONG || localType == DOUBLE) ? 2 : 1;
1379      methodWriter.visitAbstractType(frameIndex++, localType);
1380    }
1381    i = 0;
1382    while (nStack-- > 0) {
1383      int stackType = stackTypes[i];
1384      i += (stackType == LONG || stackType == DOUBLE) ? 2 : 1;
1385      methodWriter.visitAbstractType(frameIndex++, stackType);
1386    }
1387    methodWriter.visitFrameEnd();
1388  }

1390  /**
1391   * Put the given abstract type in the given ByteVector, using the JVMS verification_type_info
1392   * format used in StackMapTable attributes.
1393   *
1394   * @param symbolTable the type table to use to lookup and store type {@link Symbol}.
1395   * @param abstractType an abstract type, restricted to {@link Frame#CONSTANT_KIND}, {@link
1396   *     Frame#REFERENCE_KIND} or {@link Frame#UNINITIALIZED_KIND} types.
1397   * @param output where the abstract type must be put.
1398   * @see <a href="https://docs.oracle.com/javase/specs/jvms/se9/html/jvms-4.html#jvms-4.7.4">JVMS
1399   *     4.7.4</a>
1400   */
1401  static void putAbstractType(
1402      final SymbolTable symbolTable, final int abstractType, final ByteVector output) {
1403    int arrayDimensions = (abstractType & Frame.DIM_MASK) >> DIM_SHIFT;
1404    if (arrayDimensions == 0) {
1405      int typeValue = abstractType & VALUE_MASK;
1406      switch (abstractType & KIND_MASK) {
1407        case CONSTANT_KIND:
1408          output.putByte(typeValue);
1409          break;
1410        case REFERENCE_KIND:
1411          output
1412              .putByte(ITEM_OBJECT)
1413              .putShort(symbolTable.addConstantClass(symbolTable.getType(typeValue).value).index);
1414          break;
1415        case UNINITIALIZED_KIND:
1416          output.putByte(ITEM_UNINITIALIZED).putShort((int) symbolTable.getType(typeValue).data);
1417          break;
1418        default:
1419          throw new AssertionError();
1420      }
1421    } else {
1422      // Case of an array type, we need to build its descriptor first.
1423      StringBuilder typeDescriptor = new StringBuilder();
1424      while (arrayDimensions-- > 0) {
1425        typeDescriptor.append('[');
1426      }
1427      if ((abstractType & KIND_MASK) == REFERENCE_KIND) {
1428        typeDescriptor
1429            .append('L')
1430            .append(symbolTable.getType(abstractType & VALUE_MASK).value)
1431            .append(';');
1432      } else {
1433        switch (abstractType & VALUE_MASK) {
1434          case Frame.ITEM_ASM_BOOLEAN:
1435            typeDescriptor.append('Z');
1436            break;
1437          case Frame.ITEM_ASM_BYTE:
1438            typeDescriptor.append('B');
1439            break;
1440          case Frame.ITEM_ASM_CHAR:
1441            typeDescriptor.append('C');
1442            break;
1443          case Frame.ITEM_ASM_SHORT:
1444            typeDescriptor.append('S');
1445            break;
1446          case Frame.ITEM_INTEGER:
1447            typeDescriptor.append('I');
1448            break;
1449          case Frame.ITEM_FLOAT:
1450            typeDescriptor.append('F');
1451            break;
1452          case Frame.ITEM_LONG:
1453            typeDescriptor.append('J');
1454            break;
1455          case Frame.ITEM_DOUBLE:
1456            typeDescriptor.append('D');
1457            break;
1458          default:
1459            throw new AssertionError();
1460        }
1461      }
1462      output
1463          .putByte(ITEM_OBJECT)
1464          .putShort(symbolTable.addConstantClass(typeDescriptor.toString()).index);
1465    }
1466  }
