030/**
031 * A visitor to visit a Java method. The methods of this class must be called in the following
032 * order: ( <tt>visitParameter</tt> )* [ <tt>visitAnnotationDefault</tt> ] (
033 * <tt>visitAnnotation</tt> | <tt>visitAnnotableParameterCount</tt> |
034 * <tt>visitParameterAnnotation</tt> <tt>visitTypeAnnotation</tt> | <tt>visitAttribute</tt> )* [
035 * <tt>visitCode</tt> ( <tt>visitFrame</tt> | <tt>visit<i>X</i>Insn</tt> | <tt>visitLabel</tt> |
036 * <tt>visitInsnAnnotation</tt> | <tt>visitTryCatchBlock</tt> | <tt>visitTryCatchAnnotation</tt> |
037 * <tt>visitLocalVariable</tt> | <tt>visitLocalVariableAnnotation</tt> | <tt>visitLineNumber</tt> )*
038 * <tt>visitMaxs</tt> ] <tt>visitEnd</tt>. In addition, the <tt>visit<i>X</i>Insn</tt> and
039 * <tt>visitLabel</tt> methods must be called in the sequential order of the bytecode instructions
040 * of the visited code, <tt>visitInsnAnnotation</tt> must be called <i>after</i> the annotated
041 * instruction, <tt>visitTryCatchBlock</tt> must be called <i>before</i> the labels passed as
042 * arguments have been visited, <tt>visitTryCatchBlockAnnotation</tt> must be called <i>after</i>
043 * the corresponding try catch block has been visited, and the <tt>visitLocalVariable</tt>,
044 * <tt>visitLocalVariableAnnotation</tt> and <tt>visitLineNumber</tt> methods must be called
045 * <i>after</i> the labels passed as arguments have been visited.
046 *
047 * @author Eric Bruneton
048 */
049public abstract class MethodVisitor {
050
051  private static final String REQUIRES_ASM5 = "This feature requires ASM5";
052
053  /**
054   * The ASM API version implemented by this visitor. The value of this field must be one of {@link
055   * Opcodes#ASM4}, {@link Opcodes#ASM5}, {@link Opcodes#ASM6} or {@link Opcodes#ASM7_EXPERIMENTAL}.
056   */
057  protected final int api;
058
059  /** The method visitor to which this visitor must delegate method calls. May be null. */
060  protected MethodVisitor mv;
061
062  /**
063   * Constructs a new {@link MethodVisitor}.
064   *
065   * @param api the ASM API version implemented by this visitor. Must be one of {@link
066   *     Opcodes#ASM4}, {@link Opcodes#ASM5}, {@link Opcodes#ASM6} or {@link
067   *     Opcodes#ASM7_EXPERIMENTAL}.
068   */
069  public MethodVisitor(final int api) {
070    this(api, null);
071  }
072
073  /**
074   * Constructs a new {@link MethodVisitor}.
075   *
076   * @param api the ASM API version implemented by this visitor. Must be one of {@link
077   *     Opcodes#ASM4}, {@link Opcodes#ASM5}, {@link Opcodes#ASM6} or {@link
078   *     Opcodes#ASM7_EXPERIMENTAL}.
079   * @param methodVisitor the method visitor to which this visitor must delegate method calls. May
080   *     be null.
081   */
082  public MethodVisitor(final int api, final MethodVisitor methodVisitor) {
083    if (api != Opcodes.ASM6
084        && api != Opcodes.ASM5
085        && api != Opcodes.ASM4
086        && api != Opcodes.ASM7_EXPERIMENTAL) {
087      throw new IllegalArgumentException();
088    }
089    this.api = api;
090    this.mv = methodVisitor;
091  }

097  /**
098   * Visits a parameter of this method.
099   *
100   * @param name parameter name or null if none is provided.
101   * @param access the parameter's access flags, only <tt>ACC_FINAL</tt>, <tt>ACC_SYNTHETIC</tt>
102   *     or/and <tt>ACC_MANDATED</tt> are allowed (see {@link Opcodes}).
103   */
104  public void visitParameter(final String name, final int access) {
105    if (api < Opcodes.ASM5) {
106      throw new UnsupportedOperationException(REQUIRES_ASM5);
107    }
108    if (mv != null) {
109      mv.visitParameter(name, access);
110    }
111  }
112
113  /**
114   * Visits the default value of this annotation interface method.
115   *
116   * @return a visitor to the visit the actual default value of this annotation interface method, or
117   *     <tt>null</tt> if this visitor is not interested in visiting this default value. The 'name'
118   *     parameters passed to the methods of this annotation visitor are ignored. Moreover, exacly
119   *     one visit method must be called on this annotation visitor, followed by visitEnd.
120   */
121  public AnnotationVisitor visitAnnotationDefault() {
122    if (mv != null) {
123      return mv.visitAnnotationDefault();
124    }
125    return null;
126  }

128  /**
129   * Visits an annotation of this method.
130   *
131   * @param descriptor the class descriptor of the annotation class.
132   * @param visible <tt>true</tt> if the annotation is visible at runtime.
133   * @return a visitor to visit the annotation values, or <tt>null</tt> if this visitor is not
134   *     interested in visiting this annotation.
135   */
136  public AnnotationVisitor visitAnnotation(final String descriptor, final boolean visible) {
137    if (mv != null) {
138      return mv.visitAnnotation(descriptor, visible);
139    }
140    return null;
141  }

143  /**
144   * Visits an annotation on a type in the method signature.
145   *
146   * @param typeRef a reference to the annotated type. The sort of this type reference must be
147   *     {@link TypeReference#METHOD_TYPE_PARAMETER}, {@link
148   *     TypeReference#METHOD_TYPE_PARAMETER_BOUND}, {@link TypeReference#METHOD_RETURN}, {@link
149   *     TypeReference#METHOD_RECEIVER}, {@link TypeReference#METHOD_FORMAL_PARAMETER} or {@link
150   *     TypeReference#THROWS}. See {@link TypeReference}.
151   * @param typePath the path to the annotated type argument, wildcard bound, array element type, or
152   *     static inner type within 'typeRef'. May be <tt>null</tt> if the annotation targets
153   *     'typeRef' as a whole.
154   * @param descriptor the class descriptor of the annotation class.
155   * @param visible <tt>true</tt> if the annotation is visible at runtime.
156   * @return a visitor to visit the annotation values, or <tt>null</tt> if this visitor is not
157   *     interested in visiting this annotation.
158   */
159  public AnnotationVisitor visitTypeAnnotation(
160      final int typeRef, final TypePath typePath, final String descriptor, final boolean visible) {
161    if (api < Opcodes.ASM5) {
162      throw new UnsupportedOperationException(REQUIRES_ASM5);
163    }
164    if (mv != null) {
165      return mv.visitTypeAnnotation(typeRef, typePath, descriptor, visible);
166    }
167    return null;
168  }

170  /**
171   * Visits the number of method parameters that can have annotations. By default (i.e. when this
172   * method is not called), all the method parameters defined by the method descriptor can have
173   * annotations.
174   *
175   * @param parameterCount the number of method parameters than can have annotations. This number
176   *     must be less or equal than the number of parameter types in the method descriptor. It can
177   *     be strictly less when a method has synthetic parameters and when these parameters are
178   *     ignored when computing parameter indices for the purpose of parameter annotations (see
179   *     https://docs.oracle.com/javase/specs/jvms/se9/html/jvms-4.html#jvms-4.7.18).
180   * @param visible <tt>true</tt> to define the number of method parameters that can have
181   *     annotations visible at runtime, <tt>false</tt> to define the number of method parameters
182   *     that can have annotations invisible at runtime.
183   */
184  public void visitAnnotableParameterCount(final int parameterCount, final boolean visible) {
185    if (mv != null) {
186      mv.visitAnnotableParameterCount(parameterCount, visible);
187    }
188  }
189
190  /**
191   * Visits an annotation of a parameter this method.
192   *
193   * @param parameter the parameter index. This index must be strictly smaller than the number of
194   *     parameters in the method descriptor, and strictly smaller than the parameter count
195   *     specified in {@link #visitAnnotableParameterCount}. Important note: <i>a parameter index i
196   *     is not required to correspond to the i'th parameter descriptor in the method
197   *     descriptor</i>, in particular in case of synthetic parameters (see
198   *     https://docs.oracle.com/javase/specs/jvms/se9/html/jvms-4.html#jvms-4.7.18).
199   * @param descriptor the class descriptor of the annotation class.
200   * @param visible <tt>true</tt> if the annotation is visible at runtime.
201   * @return a visitor to visit the annotation values, or <tt>null</tt> if this visitor is not
202   *     interested in visiting this annotation.
203   */
204  public AnnotationVisitor visitParameterAnnotation(
205      final int parameter, final String descriptor, final boolean visible) {
206    if (mv != null) {
207      return mv.visitParameterAnnotation(parameter, descriptor, visible);
208    }
209    return null;
210  }

230  /**
231   * Visits the current state of the local variables and operand stack elements. This method must(*)
232   * be called <i>just before</i> any instruction <b>i</b> that follows an unconditional branch
233   * instruction such as GOTO or THROW, that is the target of a jump instruction, or that starts an
234   * exception handler block. The visited types must describe the values of the local variables and
235   * of the operand stack elements <i>just before</i> <b>i</b> is executed.<br>
236   * <br>
237   * (*) this is mandatory only for classes whose version is greater than or equal to {@link
238   * Opcodes#V1_6}. <br>
239   * <br>
240   * The frames of a method must be given either in expanded form, or in compressed form (all frames
241   * must use the same format, i.e. you must not mix expanded and compressed frames within a single
242   * method):
243   *
244   * <ul>
245   *   <li>In expanded form, all frames must have the F_NEW type.
246   *   <li>In compressed form, frames are basically "deltas" from the state of the previous frame:
247   *       <ul>
248   *         <li>{@link Opcodes#F_SAME} representing frame with exactly the same locals as the
249   *             previous frame and with the empty stack.
250   *         <li>{@link Opcodes#F_SAME1} representing frame with exactly the same locals as the
251   *             previous frame and with single value on the stack ( <code>nStack</code> is 1 and
252   *             <code>stack[0]</code> contains value for the type of the stack item).
253   *         <li>{@link Opcodes#F_APPEND} representing frame with current locals are the same as the
254   *             locals in the previous frame, except that additional locals are defined (<code>
255   *             nLocal</code> is 1, 2 or 3 and <code>local</code> elements contains values
256   *             representing added types).
257   *         <li>{@link Opcodes#F_CHOP} representing frame with current locals are the same as the
258   *             locals in the previous frame, except that the last 1-3 locals are absent and with
259   *             the empty stack (<code>nLocals</code> is 1, 2 or 3).
260   *         <li>{@link Opcodes#F_FULL} representing complete frame data.
261   *       </ul>
262   * </ul>
263   *
264   * <br>
265   * In both cases the first frame, corresponding to the method's parameters and access flags, is
266   * implicit and must not be visited. Also, it is illegal to visit two or more frames for the same
267   * code location (i.e., at least one instruction must be visited between two calls to visitFrame).
268   *
269   * @param type the type of this stack map frame. Must be {@link Opcodes#F_NEW} for expanded
270   *     frames, or {@link Opcodes#F_FULL}, {@link Opcodes#F_APPEND}, {@link Opcodes#F_CHOP}, {@link
271   *     Opcodes#F_SAME} or {@link Opcodes#F_APPEND}, {@link Opcodes#F_SAME1} for compressed frames.
272   * @param nLocal the number of local variables in the visited frame.
273   * @param local the local variable types in this frame. This array must not be modified. Primitive
274   *     types are represented by {@link Opcodes#TOP}, {@link Opcodes#INTEGER}, {@link
275   *     Opcodes#FLOAT}, {@link Opcodes#LONG}, {@link Opcodes#DOUBLE}, {@link Opcodes#NULL} or
276   *     {@link Opcodes#UNINITIALIZED_THIS} (long and double are represented by a single element).
277   *     Reference types are represented by String objects (representing internal names), and
278   *     uninitialized types by Label objects (this label designates the NEW instruction that
279   *     created this uninitialized value).
280   * @param nStack the number of operand stack elements in the visited frame.
281   * @param stack the operand stack types in this frame. This array must not be modified. Its
282   *     content has the same format as the "local" array.
283   * @throws IllegalStateException if a frame is visited just after another one, without any
284   *     instruction between the two (unless this frame is a Opcodes#F_SAME frame, in which case it
285   *     is silently ignored).
286   */
287  public void visitFrame(
288      final int type,
289      final int nLocal,
290      final Object[] local,
291      final int nStack,
292      final Object[] stack) {
293    if (mv != null) {
294      mv.visitFrame(type, nLocal, local, nStack, stack);
295    }
296  }

497  /**
498   * Visits a LDC instruction. Note that new constant types may be added in future versions of the
499   * Java Virtual Machine. To easily detect new constant types, implementations of this method
500   * should check for unexpected constant types, like this:
501   *
502   * <pre>
503   * if (cst instanceof Integer) {
504   *     // ...
505   * } else if (cst instanceof Float) {
506   *     // ...
507   * } else if (cst instanceof Long) {
508   *     // ...
509   * } else if (cst instanceof Double) {
510   *     // ...
511   * } else if (cst instanceof String) {
512   *     // ...
513   * } else if (cst instanceof Type) {
514   *     int sort = ((Type) cst).getSort();
515   *     if (sort == Type.OBJECT) {
516   *         // ...
517   *     } else if (sort == Type.ARRAY) {
518   *         // ...
519   *     } else if (sort == Type.METHOD) {
520   *         // ...
521   *     } else {
522   *         // throw an exception
523   *     }
524   * } else if (cst instanceof Handle) {
525   *     // ...
526   * } else if (cst instanceof Condy) {
527   *     // ...
528   * } else {
529   *     // throw an exception
530   * }
531   * </pre>
532   *
533   * @param value the constant to be loaded on the stack. This parameter must be a non null {@link
534   *     Integer}, a {@link Float}, a {@link Long}, a {@link Double}, a {@link String}, a {@link
535   *     Type} of OBJECT or ARRAY sort for <tt>.class</tt> constants, for classes whose version is
536   *     49, a {@link Type} of METHOD sort for MethodType, a {@link Handle} for MethodHandle
537   *     constants, for classes whose version is 51 or a {@link ConstantDynamic} for a constant
538   *     dynamic for classes whose version is 55.
539   */
540  public void visitLdcInsn(final Object value) {
541    if (api < Opcodes.ASM5
542        && (value instanceof Handle
543            || (value instanceof Type && ((Type) value).getSort() == Type.METHOD))) {
544      throw new UnsupportedOperationException(REQUIRES_ASM5);
545    }
546    if (api != Opcodes.ASM7_EXPERIMENTAL && value instanceof ConstantDynamic) {
547      throw new UnsupportedOperationException("This feature requires ASM7");
548    }
549    if (mv != null) {
550      mv.visitLdcInsn(value);
551    }
552  }
