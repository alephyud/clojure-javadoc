011package clojure;
012
013import clojure.lang.Symbol;
014import clojure.lang.Var;
015import clojure.lang.RT;
016
017public class main{
018
019final static private Symbol CLOJURE_MAIN = Symbol.intern("clojure.main");
020final static private Var REQUIRE = RT.var("clojure.core", "require");
021final static private Var LEGACY_REPL = RT.var("clojure.main", "legacy-repl");
022final static private Var LEGACY_SCRIPT = RT.var("clojure.main", "legacy-script");
023final static private Var MAIN = RT.var("clojure.main", "main");
024
025public static void legacy_repl(String[] args) {
026    REQUIRE.invoke(CLOJURE_MAIN);
027    LEGACY_REPL.invoke(RT.seq(args));
028}
029
030public static void legacy_script(String[] args) {
031    REQUIRE.invoke(CLOJURE_MAIN);
032    LEGACY_SCRIPT.invoke(RT.seq(args));
033}
034
035public static void main(String[] args) {
036    REQUIRE.invoke(CLOJURE_MAIN);
037    MAIN.applyTo(RT.seq(args));
038}
039}
