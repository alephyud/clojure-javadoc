041static boolean doEquals(IPersistentVector v, Object obj){
042    if(obj instanceof IPersistentVector)
043        {
044        IPersistentVector ov = (IPersistentVector) obj;
045        if(ov.count() != v.count())
046            return false;
047        for(int i = 0;i< v.count();i++)
048            {
049            if(!Util.equals(v.nth(i), ov.nth(i)))
050                return false;
051            }
052        return true;
053        }
054        else if(obj instanceof List)
055        {
056                Collection ma = (Collection) obj;
057                if(ma.size() != v.count() || ma.hashCode() != v.hashCode())
058                        return false;
059                for(Iterator i1 = ((List) v).iterator(), i2 = ma.iterator();
060                    i1.hasNext();)
061                        {
062                        if(!Util.equals(i1.next(), i2.next()))
063                                return false;
064                        }
065                return true;
066                }
067        else
068        {
069                if(!(obj instanceof Sequential))
070                        return false;
071                ISeq ms = RT.seq(obj);
072                for(int i = 0; i < v.count(); i++, ms = ms.next())
073                        {
074                        if(ms == null || !Util.equals(v.nth(i), ms.first()))
075                                return false;
076                        }
077                if(ms != null)
078                        return false;
079                }
080
081        return true;
082
083}

141public int hashCode(){
142    int hash = this._hash;
143        if(hash == 0)
144                {
145                hash = 1;
146                for(int i = 0;i<count();i++)
147                        {
148                        Object obj = nth(i);
149                        hash = 31 * hash + (obj == null ? 0 : obj.hashCode());
150                        }
151                this._hash = hash;
152                }
153        return hash;
154}
155
156public int hasheq(){
157    int hash = this._hasheq;
158        if(hash == 0) {
159        int n;
160        hash = 1;
161
162        for(n=0;n<count();++n)
163            {
164            hash = 31 * hash + Util.hasheq(nth(n));
165            }
166
167        this._hasheq = hash = Murmur3.mixCollHash(hash, n);
168        }
169        return hash;
170}

431public int compareTo(Object o){
432        IPersistentVector v = (IPersistentVector) o;
433        if(count() < v.count())
434                return -1;
435        else if(count() > v.count())
436                return 1;
437        for(int i = 0; i < count(); i++)
438                {
439                int c = Util.compare(nth(i),v.nth(i));
440                if(c != 0)
441                        return c;
442                }
443        return 0;
444}

487        public Object reduce(IFn f) {
488                Object ret = v.nth(i);
489                for(int x = i + 1; x < v.count(); x++) {
490            ret = f.invoke(ret, v.nth(x));
491            if (RT.isReduced(ret)) return ((IDeref)ret).deref();
492        }
493                return ret;

496        public Object reduce(IFn f, Object start) {
497                Object ret = f.invoke(start, v.nth(i));
498                for(int x = i + 1; x < v.count(); x++) {
499            if (RT.isReduced(ret)) return ((IDeref)ret).deref();
500            ret = f.invoke(ret, v.nth(x));
501        }
502                if (RT.isReduced(ret)) return ((IDeref)ret).deref();
503                return ret;
504        }

547public static class SubVector extends APersistentVector implements IObj{
548        public final IPersistentVector v;
549        public final int start;
550        public final int end;
551        final IPersistentMap _meta;
552
553
554
555        public SubVector(IPersistentMap meta, IPersistentVector v, int start, int end){
556                this._meta = meta;
557
558                if(v instanceof APersistentVector.SubVector)
559                        {
560                        APersistentVector.SubVector sv = (APersistentVector.SubVector) v;
561                        start += sv.start;
562                        end += sv.start;
563                        v = sv.v;
564                        }
565                this.v = v;
566                this.start = start;
567                this.end = end;
568        }
569
570        public Iterator iterator(){
571                if (v instanceof APersistentVector) {
572                        return ((APersistentVector)v).rangedIterator(start,end);
573                }
574                return super.iterator();
575        }
576
577        public Object nth(int i){
578                if((start + i >= end) || (i < 0))
579                        throw new IndexOutOfBoundsException();
580                return v.nth(start + i);
581        }
582
583        public IPersistentVector assocN(int i, Object val){
584                if(start + i > end)
585                        throw new IndexOutOfBoundsException();
586                else if(start + i == end)
587                        return cons(val);
588                return new SubVector(_meta, v.assocN(start + i, val), start, end);
589        }
590
591        public int count(){
592                return end - start;
593        }
594
595        public IPersistentVector cons(Object o){
596                return new SubVector(_meta, v.assocN(end, o), start, end + 1);
597        }
598
599        public IPersistentCollection empty(){
600                return PersistentVector.EMPTY.withMeta(meta());
601        }
602
603        public IPersistentStack pop(){
604                if(end - 1 == start)
605                        {
606                        return PersistentVector.EMPTY;
607                        }
608                return new SubVector(_meta, v, start, end - 1);
609        }
610
611        public SubVector withMeta(IPersistentMap meta){
612                if(meta == _meta)
613                        return this;
614                return new SubVector(meta, v, start, end);
615        }
616
617        public IPersistentMap meta(){
618                return _meta;
619        }
620}
