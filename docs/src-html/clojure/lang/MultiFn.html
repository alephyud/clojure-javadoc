018public class MultiFn extends AFn{
019final public IFn dispatchFn;
020final public Object defaultDispatchVal;
021final public IRef hierarchy;
022final String name;
023final ReentrantReadWriteLock rw;
024volatile IPersistentMap methodTable;
025volatile IPersistentMap preferTable;
026volatile IPersistentMap methodCache;
027volatile Object cachedHierarchy;
028
029static final Var assoc = RT.var("clojure.core", "assoc");
030static final Var dissoc = RT.var("clojure.core", "dissoc");
031static final Var isa = RT.var("clojure.core", "isa?");
032static final Var parents = RT.var("clojure.core", "parents");
033
034public MultiFn(String name, IFn dispatchFn, Object defaultDispatchVal, IRef hierarchy) {
035        this.rw = new ReentrantReadWriteLock();
036        this.name = name;
037        this.dispatchFn = dispatchFn;
038        this.defaultDispatchVal = defaultDispatchVal;
039        this.methodTable = PersistentHashMap.EMPTY;
040        this.methodCache = getMethodTable();
041        this.preferTable = PersistentHashMap.EMPTY;
042    this.hierarchy = hierarchy;
043        cachedHierarchy = null;
044}
045
046public MultiFn reset(){
047        rw.writeLock().lock();
048        try{
049                methodTable = methodCache = preferTable = PersistentHashMap.EMPTY;
050                cachedHierarchy = null;
051                return this;
052        }
053        finally {
054                rw.writeLock().unlock();
055        }
056}
057
058public MultiFn addMethod(Object dispatchVal, IFn method) {
059        rw.writeLock().lock();
060        try{
061                methodTable = getMethodTable().assoc(dispatchVal, method);
062                resetCache();
063                return this;
064        }
065        finally {
066                rw.writeLock().unlock();
067        }
068}
069
070public MultiFn removeMethod(Object dispatchVal) {
071        rw.writeLock().lock();
072        try
073                {
074                methodTable = getMethodTable().without(dispatchVal);
075                resetCache();
076                return this;
077                }
078        finally
079                {
080                rw.writeLock().unlock();
081                }
082}
083
084public MultiFn preferMethod(Object dispatchValX, Object dispatchValY) {
085        rw.writeLock().lock();
086        try
087                {
088                if(prefers(dispatchValY, dispatchValX))
089                        throw new IllegalStateException(
090                                        String.format("Preference conflict in multimethod '%s': %s is already preferred to %s",
091                                                      name, dispatchValY, dispatchValX));
092                preferTable = getPreferTable().assoc(dispatchValX, RT.conj((IPersistentCollection) RT.get(getPreferTable(),
093                                                                                                     dispatchValX,
094                                                                                                     PersistentHashSet.EMPTY),
095                                                                      dispatchValY));
096                resetCache();
097                return this;
098                }
099        finally
100                {
101                rw.writeLock().unlock();
102                }
103}
104
105private boolean prefers(Object x, Object y) {
106        IPersistentSet xprefs = (IPersistentSet) getPreferTable().valAt(x);
107        if(xprefs != null && xprefs.contains(y))
108                return true;
109        for(ISeq ps = RT.seq(parents.invoke(y)); ps != null; ps = ps.next())
110                {
111                if(prefers(x, ps.first()))
112                        return true;
113                }
114        for(ISeq ps = RT.seq(parents.invoke(x)); ps != null; ps = ps.next())
115                {
116                if(prefers(ps.first(), y))
117                        return true;
118                }
119        return false;
120}
121
122private boolean isA(Object x, Object y) {
123    return RT.booleanCast(isa.invoke(hierarchy.deref(), x, y));
124}
125
126private boolean dominates(Object x, Object y) {
127        return prefers(x, y) || isA(x, y);
128}
129
130private IPersistentMap resetCache() {
131        rw.writeLock().lock();
132        try
133                {
134                methodCache = getMethodTable();
135                cachedHierarchy = hierarchy.deref();
136                return methodCache;
137                }
138        finally
139                {
140                rw.writeLock().unlock();
141                }
142}
143
144 public IFn getMethod(Object dispatchVal) {
145        if(cachedHierarchy != hierarchy.deref())
146                resetCache();
147        IFn targetFn = (IFn) methodCache.valAt(dispatchVal);
148        if(targetFn != null)
149                return targetFn;
150        return findAndCacheBestMethod(dispatchVal);
151}
152
153private IFn getFn(Object dispatchVal) {
154        IFn targetFn = getMethod(dispatchVal);
155        if(targetFn == null)
156                throw new IllegalArgumentException(String.format("No method in multimethod '%s' for dispatch value: %s",
157                                                                 name, dispatchVal));
158        return targetFn;
159}
160
161private IFn findAndCacheBestMethod(Object dispatchVal) {
162        rw.readLock().lock();
163        Object bestValue;
164        IPersistentMap mt = methodTable;
165        IPersistentMap pt = preferTable;
166        Object ch = cachedHierarchy;
167        try
168                {
169                Map.Entry bestEntry = null;
170                for(Object o : getMethodTable())
171                        {
172                        Map.Entry e = (Map.Entry) o;
173                        if(isA(dispatchVal, e.getKey()))
174                                {
175                                if(bestEntry == null || dominates(e.getKey(), bestEntry.getKey()))
176                                        bestEntry = e;
177                                if(!dominates(bestEntry.getKey(), e.getKey()))
178                                        throw new IllegalArgumentException(
179                                                        String.format(
180                                                                        "Multiple methods in multimethod '%s' match dispatch value: %s -> %s and %s, and neither is preferred",
181                                                                        name, dispatchVal, e.getKey(), bestEntry.getKey()));
182                                }
183                        }
184                if(bestEntry == null)
185                        {
186                        bestValue = methodTable.valAt(defaultDispatchVal);
187                        if(bestValue == null)
188                                return null;
189                        }
190                else
191                        bestValue = bestEntry.getValue();
192                }
193        finally
194                {
195                rw.readLock().unlock();
196                }
197
198
199        //ensure basis has stayed stable throughout, else redo
200        rw.writeLock().lock();
201        try
202                {
203                if( mt == methodTable &&
204                    pt == preferTable &&
205                    ch == cachedHierarchy &&
206                        cachedHierarchy == hierarchy.deref())
207                        {
208                        //place in cache
209                        methodCache = methodCache.assoc(dispatchVal, bestValue);
210                        return (IFn) bestValue;
211                        }
212                else
213                        {
214                        resetCache();
215                        return findAndCacheBestMethod(dispatchVal);
216                        }
217                }
218        finally
219                {
220                rw.writeLock().unlock();
221                }
222}
