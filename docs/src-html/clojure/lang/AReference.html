013package clojure.lang;
014
015public class AReference implements IReference {
016    private IPersistentMap _meta;
017
018    public AReference() {
019        this(null);
020    }
021
022    public AReference(IPersistentMap meta) {
023        _meta = meta;
024    }
025
026    synchronized public IPersistentMap meta() {
027        return _meta;
028    }
029
030    synchronized public IPersistentMap alterMeta(IFn alter, ISeq args)  {
031        _meta = (IPersistentMap) alter.applyTo(new Cons(_meta, args));
032        return _meta;
033    }
034
035    synchronized public IPersistentMap resetMeta(IPersistentMap m) {
036        _meta = m;
037        return m;
038    }
039
040}
