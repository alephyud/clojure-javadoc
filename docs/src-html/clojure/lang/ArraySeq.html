017public class ArraySeq extends ASeq implements IndexedSeq, IReduce{
018public final Object[] array;
019final int i;
020//ISeq _rest;
021
022static public ArraySeq create(){
023        return null;
024}
025
026static public ArraySeq create(Object... array){
027        if(array == null || array.length == 0)
028                return null;
029        return new ArraySeq(array, 0);
030}
031
032static ISeq createFromObject(Object array){
033        if(array == null || Array.getLength(array) == 0)
034                return null;
035        Class aclass = array.getClass();
036        if(aclass == int[].class)
037                return new ArraySeq_int(null, (int[]) array, 0);
038        if(aclass == float[].class)
039                return new ArraySeq_float(null, (float[]) array, 0);
040        if(aclass == double[].class)
041                return new ArraySeq_double(null, (double[]) array, 0);
042        if(aclass == long[].class)
043                return new ArraySeq_long(null, (long[]) array, 0);
044        if(aclass == byte[].class)
045                return new ArraySeq_byte(null, (byte[]) array, 0);
046        if(aclass == char[].class)
047                return new ArraySeq_char(null, (char[]) array, 0);
048        if(aclass == short[].class)
049            return new ArraySeq_short(null, (short[]) array, 0);
050        if(aclass == boolean[].class)
051                return new ArraySeq_boolean(null, (boolean[]) array, 0);
052        return new ArraySeq(array, 0);
053}
054
055ArraySeq(Object array, int i){
056        this.i = i;
057        this.array = (Object[]) array;
058//    this._rest = this;
059}
060
061ArraySeq(IPersistentMap meta, Object array, int i){
062        super(meta);
063        this.i = i;
064        this.array = (Object[]) array;
065}
066
067public Object first(){
068        if(array != null)
069                return array[i];
070        return null;
071}
072
073public ISeq next(){
074        if(array != null && i + 1 < array.length)
075                return new ArraySeq(array, i + 1);
076        return null;
077}
078
079public int count(){
080        if(array != null)
081                return array.length - i;
082        return 0;
083}
084
085public int index(){
086        return i;
087}
088
089public ArraySeq withMeta(IPersistentMap meta){
090        if(meta() == meta)
091                return this;
092        return new ArraySeq(meta, array, i);
093}
094
095public Object reduce(IFn f) {
096        if(array != null) {
097                Object ret = array[i];
098                for(int x = i + 1; x < array.length; x++)
099                        {
100                        ret = f.invoke(ret, array[x]);
101                        if(RT.isReduced(ret))
102                                return ((IDeref)ret).deref();
103                        }
104                return ret;
105        }
106        return null;
107}
108
109public Object reduce(IFn f, Object start) {
110        if(array != null) {
111                Object ret = f.invoke(start, array[i]);
112                for(int x = i + 1; x < array.length; x++)
113                        {
114                        if(RT.isReduced(ret))
115                                return ((IDeref)ret).deref();
116                        ret = f.invoke(ret, array[x]);
117                        }
118                if(RT.isReduced(ret))
119                        return ((IDeref)ret).deref();
120                return ret;
121        }
122        return null;
123}
124
125public int indexOf(Object o) {
126        if(array != null)
127                for (int j = i; j < array.length; j++)
128                        if (Util.equals(o, array[j])) return j - i;
129        return -1;
130}
131
132public int lastIndexOf(Object o) {
133        if (array != null) {
134                if (o == null) {
135                        for (int j = array.length - 1 ; j >= i; j--)
136                                if (array[j] == null) return j - i;
137                } else {
138                        for (int j = array.length - 1 ; j >= i; j--)
139                                if (o.equals(array[j])) return j - i;
140                }
141        }
142        return -1;
143}
