013package clojure.lang;
014
015import java.util.Iterator;
016
017public final class RecordIterator implements Iterator {
018
019    int i = 0;
020    final int basecnt;
021    final ILookup rec;
022    final IPersistentVector basefields;
023    final Iterator extmap;
024
025    public RecordIterator (ILookup rec, IPersistentVector basefields, Iterator extmap) {
026        this.rec = rec;
027        this.basefields = basefields;
028        this.basecnt = basefields.count();
029        this.extmap = extmap;
030    }
031
032    public boolean hasNext() {
033        if (i < basecnt) {
034            return true;
035        } else {
036            return extmap.hasNext();
037        }
038    }
039
040    public Object next() {
041        if (i < basecnt) {
042            Object k = basefields.nth(i);
043            i++;
044            return MapEntry.create(k, rec.valAt(k));
045        } else  {
046            return extmap.next();
047        }
048    }
049
050    public void remove() {
051        throw new UnsupportedOperationException();
052    }
053}
