206static public double quotient(double n, double d){
207        if(d == 0)
208                throw new ArithmeticException("Divide by zero");
209
210        double q = n / d;
211        if(q <= Long.MAX_VALUE && q >= Long.MIN_VALUE)
212                {
213                return (double)(long) q;
214                }
215        else
216                { //bigint quotient
217                return new BigDecimal(q).toBigInteger().doubleValue();
218                }

222        if(d == 0)
223                throw new ArithmeticException("Divide by zero");
224
225        double q = n / d;
226        if(q <= Long.MAX_VALUE && q >= Long.MIN_VALUE)
227                {
228                return (n - ((long) q) * d);
229                }
230        else
231                { //bigint quotient
232                Number bq = new BigDecimal(q).toBigInteger();
233                return (n - bq.doubleValue() * d);
234                }

295@WarnBoxedMath(false)
296static BigDecimal toBigDecimal(Object x){
297        if(x instanceof BigDecimal)
298                return (BigDecimal) x;
299        else if(x instanceof BigInt)
300                {
301                BigInt bi = (BigInt) x;
302                if(bi.bipart == null)
303                        return BigDecimal.valueOf(bi.lpart);
304                else
305                        return new BigDecimal(bi.bipart);
306                }
307        else if(x instanceof BigInteger)
308                return new BigDecimal((BigInteger) x);
309        else if(x instanceof Double)
310                return new BigDecimal(((Number) x).doubleValue());
311        else if(x instanceof Float)
312                return new BigDecimal(((Number) x).doubleValue());
313        else if(x instanceof Ratio)
314                {
315                Ratio r = (Ratio)x;
316                return (BigDecimal)divide(new BigDecimal(r.numerator), r.denominator);
317                }
318        else
319                return BigDecimal.valueOf(((Number) x).longValue());
320}

322@WarnBoxedMath(false)
323static public Ratio toRatio(Object x){
324        if(x instanceof Ratio)
325                return (Ratio) x;
326        else if(x instanceof BigDecimal)
327                {
328                BigDecimal bx = (BigDecimal) x;
329                BigInteger bv = bx.unscaledValue();
330                int scale = bx.scale();
331                if(scale < 0)
332                        return new Ratio(bv.multiply(BigInteger.TEN.pow(-scale)), BigInteger.ONE);
333                else
334                        return new Ratio(bv, BigInteger.TEN.pow(scale));
335                }
336        return new Ratio(toBigInteger(x), BigInteger.ONE);
337}

340static public Number rationalize(Number x){
341        if(x instanceof Float || x instanceof Double)
342                return rationalize(BigDecimal.valueOf(x.doubleValue()));
343        else if(x instanceof BigDecimal)
344                {
345                BigDecimal bx = (BigDecimal) x;
346                BigInteger bv = bx.unscaledValue();
347                int scale = bx.scale();
348                if(scale < 0)
349                        return BigInt.fromBigInteger(bv.multiply(BigInteger.TEN.pow(-scale)));
350                else
351                        return divide(bv, BigInteger.TEN.pow(scale));
352                }
353        return x;
354}

412static public Number divide(BigInteger n, BigInteger d){
413        if(d.equals(BigInteger.ZERO))
414                throw new ArithmeticException("Divide by zero");
415        BigInteger gcd = n.gcd(d);
416        if(gcd.equals(BigInteger.ZERO))
417                return BigInt.ZERO;
418        n = n.divide(gcd);
419        d = d.divide(gcd);
420        if(d.equals(BigInteger.ONE))
421                return BigInt.fromBigInteger(n);
422        else if(d.equals(BigInteger.ONE.negate()))
423                return BigInt.fromBigInteger(n.negate());
424        return new Ratio((d.signum() < 0 ? n.negate() : n),
425                         (d.signum() < 0 ? d.negate() : d));
426}

1123static int hasheqFrom(Number x, Class xc){
1124        if(xc == Integer.class
1125                        || xc == Short.class
1126                        || xc == Byte.class
1127                        || (xc == BigInteger.class && lte(x, Long.MAX_VALUE) && gte(x,Long.MIN_VALUE)))
1128        {
1129                long lpart = x.longValue();
1130                return Murmur3.hashLong(lpart);
1131                //return (int) (lpart ^ (lpart >>> 32));
1132        }
1133        if(xc == BigDecimal.class)
1134        {
1135                // stripTrailingZeros() to make all numerically equal
1136                // BigDecimal values come out the same before calling
1137                // hashCode.  Special check for 0 because
1138                // stripTrailingZeros() does not do anything to values
1139                // equal to 0 with different scales.
1140                if (isZero(x))
1141                        return BigDecimal.ZERO.hashCode();
1142                else
1143                {
1144                        BigDecimal tmp = ((BigDecimal) x).stripTrailingZeros();
1145                        return tmp.hashCode();
1146                }
1147        }
1148        if(xc == Float.class && x.equals(-0.0f))
1149        {
1150                return 0;  // match 0.0f
1151        }
1152        return x.hashCode();
