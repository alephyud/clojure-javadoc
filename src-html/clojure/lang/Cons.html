013package clojure.lang;
014
015import java.io.Serializable;
016
017final public class Cons extends ASeq implements Serializable {
018
019private final Object _first;
020private final ISeq _more;
021
022public Cons(Object first, ISeq _more){
023        this._first = first;
024        this._more = _more;
025}
026
027
028public Cons(IPersistentMap meta, Object _first, ISeq _more){
029        super(meta);
030        this._first = _first;
031        this._more = _more;
032}
033
034public Object first(){
035        return _first;
036}
037
038public ISeq next(){
039        return more().seq();
040}
041
042public ISeq more(){
043        if(_more == null)
044                return PersistentList.EMPTY;
045        return _more;
046}
047
048public int count(){
049        return 1 + RT.count(_more);
050}
051
052public Cons withMeta(IPersistentMap meta){
053        if(meta() == meta)
054                return this;
055        return new Cons(meta, _first, _more);
056}
057}
