185    boolean eofIsError = true;
186    Object eofValue = null;
187    if(opts != null && opts instanceof IPersistentMap)
188    {
189        Object eof = ((IPersistentMap)opts).valAt(OPT_EOF, EOFTHROW);

192            eofValue = eof;
193        }
194    }
195    return read(r,eofIsError,eofValue,false,opts);

233static private Object read(PushbackReader r, boolean eofIsError, Object eofValue, Character returnOn,
234                           Object returnOnValue, boolean isRecursive, Object opts, Object pendingForms,
235                           Resolver resolver)
236{
237    if(RT.READEVAL.deref() == UNKNOWN)
238        throw Util.runtimeException("Reading disallowed - *read-eval* bound to :unknown");
239
240    opts = installPlatformFeature(opts);
241
242        try
243                {
244                for(; ;)
245                        {
246
247                        if(pendingForms instanceof List && !((List)pendingForms).isEmpty())
248                                return ((List)pendingForms).remove(0);
249
250                        int ch = read1(r);
251
252                        while(isWhitespace(ch))
253                                ch = read1(r);
254
255                        if(ch == -1)
256                                {
257                                if(eofIsError)
258                                        throw Util.runtimeException("EOF while reading");
259                                return eofValue;
260                                }
261
262                        if(returnOn != null && (returnOn.charValue() == ch)) {
263                                return returnOnValue;
264                        }
265
266                        if(Character.isDigit(ch))
267                                {
268                                Object n = readNumber(r, (char) ch);
269                                return n;
270                                }
271
272                        IFn macroFn = getMacro(ch);
273                        if(macroFn != null)
274                                {
275                                Object ret = macroFn.invoke(r, (char) ch, opts, pendingForms);
276                                //no op macros return the reader
277                                if(ret == r)
278                                        continue;
279                                return ret;
280                                }
281
282                        if(ch == '+' || ch == '-')
283                                {
284                                int ch2 = read1(r);
285                                if(Character.isDigit(ch2))
286                                        {
287                                        unread(r, ch2);
288                                        Object n = readNumber(r, (char) ch);
289                                        return n;
290                                        }
291                                unread(r, ch2);
292                                }
293
294                        String token = readToken(r, (char) ch);
295                        return interpretToken(token, resolver);
296                        }
297                }
298        catch(Exception e)
299                {
300                if(isRecursive || !(r instanceof LineNumberingPushbackReader))
301                        throw Util.sneakyThrow(e);
302                LineNumberingPushbackReader rdr = (LineNumberingPushbackReader) r;
303                //throw Util.runtimeException(String.format("ReaderError:(%d,1) %s", rdr.getLineNumber(), e.getMessage()), e);
304                throw new ReaderException(rdr.getLineNumber(), rdr.getColumnNumber(), e);
305                }
306}

308static private String readToken(PushbackReader r, char initch) {
309        StringBuilder sb = new StringBuilder();
310        sb.append(initch);
311
312        for(; ;)
313                {
314                int ch = read1(r);
315                if(ch == -1 || isWhitespace(ch) || isTerminatingMacro(ch))
316                        {
317                        unread(r, ch);
318                        return sb.toString();
319                        }
320                sb.append((char) ch);
321                }
322}

324static private Object readNumber(PushbackReader r, char initch) {
325        StringBuilder sb = new StringBuilder();
326        sb.append(initch);
327
328        for(; ;)
329                {
330                int ch = read1(r);
331                if(ch == -1 || isWhitespace(ch) || isMacro(ch))
332                        {
333                        unread(r, ch);
334                        break;
335                        }
336                sb.append((char) ch);
337                }
338
339        String s = sb.toString();
340        Object n = matchNumber(s);
341        if(n == null)
342                throw new NumberFormatException("Invalid number: " + s);
343        return n;
344}

346static private int readUnicodeChar(String token, int offset, int length, int base) {
347        if(token.length() != offset + length)
348                throw new IllegalArgumentException("Invalid unicode character: \\" + token);
349        int uc = 0;
350        for(int i = offset; i < offset + length; ++i)
351                {
352                int d = Character.digit(token.charAt(i), base);
353                if(d == -1)
354                        throw new IllegalArgumentException("Invalid digit: " + token.charAt(i));
355                uc = uc * base + d;
356                }
357        return (char) uc;
358}

360static private int readUnicodeChar(PushbackReader r, int initch, int base, int length, boolean exact) {
361        int uc = Character.digit(initch, base);
362        if(uc == -1)
363                throw new IllegalArgumentException("Invalid digit: " + (char) initch);
364        int i = 1;
365        for(; i < length; ++i)
366                {
367                int ch = read1(r);
368                if(ch == -1 || isWhitespace(ch) || isMacro(ch))
369                        {
370                        unread(r, ch);
371                        break;
372                        }
373                int d = Character.digit(ch, base);
374                if(d == -1)
375                        throw new IllegalArgumentException("Invalid digit: " + (char) ch);
376                uc = uc * base + d;
377                }
378        if(i != length && exact)
379                throw new IllegalArgumentException("Invalid character length: " + i + ", should be: " + length);
380        return uc;
381}

383static private Object interpretToken(String s, Resolver resolver) {
384        if(s.equals("nil"))
385                {
386                return null;
387                }
388        else if(s.equals("true"))
389                {
390                return RT.T;
391                }
392        else if(s.equals("false"))
393                {
394                return RT.F;
395                }
396        Object ret = null;
397
398        ret = matchSymbol(s, resolver);
399        if(ret != null)
400                return ret;
401
402        throw Util.runtimeException("Invalid token: " + s);
403}

406private static Object matchSymbol(String s, Resolver resolver){
407        Matcher m = symbolPat.matcher(s);
408        if(m.matches())
409                {
410                int gc = m.groupCount();
411                String ns = m.group(1);
412                String name = m.group(2);
413                if(ns != null && ns.endsWith(":/")
414                   || name.endsWith(":")
415                   || s.indexOf("::", 1) != -1)
416                        return null;
417                if(s.startsWith("::"))
418                        {
419                        Symbol ks = Symbol.intern(s.substring(2));
420            if(resolver != null)
421                {
422                Symbol nsym;
423                if(ks.ns != null)
424                    nsym = resolver.resolveAlias(Symbol.intern(ks.ns));
425                else
426                    nsym = resolver.currentNS();
427                //auto-resolving keyword
428                if(nsym != null)
429                    return Keyword.intern(nsym.name, ks.name);
430                else
431                    return null;
432                }
433            else
434                {
435                Namespace kns;
436                if(ks.ns != null)
437                    kns = Compiler.currentNS().lookupAlias(Symbol.intern(ks.ns));
438                else
439                    kns = Compiler.currentNS();
440                //auto-resolving keyword
441                if(kns != null)
442                    return Keyword.intern(kns.name.name, ks.name);
443                else
444                    return null;
445                }
446            }
447                boolean isKeyword = s.charAt(0) == ':';
448                Symbol sym = Symbol.intern(s.substring(isKeyword ? 1 : 0));
449                if(isKeyword)
450                        return Keyword.intern(sym);
451                return sym;
452                }
453        return null;
454}

457private static Object matchNumber(String s){
458        Matcher m = intPat.matcher(s);
459        if(m.matches())
460                {
461                if(m.group(2) != null)
462                        {
463                        if(m.group(8) != null)
464                                return BigInt.ZERO;
465                        return Numbers.num(0);
466                        }
467                boolean negate = (m.group(1).equals("-"));
468                String n;
469                int radix = 10;
470                if((n = m.group(3)) != null)
471                        radix = 10;
472                else if((n = m.group(4)) != null)
473                        radix = 16;
474                else if((n = m.group(5)) != null)
475                        radix = 8;
476                else if((n = m.group(7)) != null)
477                        radix = Integer.parseInt(m.group(6));
478                if(n == null)
479                        return null;
480                BigInteger bn = new BigInteger(n, radix);
481                if(negate)
482                        bn = bn.negate();
483                if(m.group(8) != null)
484                        return BigInt.fromBigInteger(bn);
485                return bn.bitLength() < 64 ?
486                       Numbers.num(bn.longValue())
487                                           : BigInt.fromBigInteger(bn);
488                }
489        m = floatPat.matcher(s);
490        if(m.matches())
491                {
492                if(m.group(4) != null)
493                        return new BigDecimal(m.group(1));
494                return Double.parseDouble(s);
495                }
496        m = ratioPat.matcher(s);
497        if(m.matches())
498                {
499                String numerator = m.group(1);
500                if (numerator.startsWith("+")) numerator = numerator.substring(1);
501
502                return Numbers.divide(Numbers.reduceBigInt(BigInt.fromBigInteger(new BigInteger(numerator))),
503                                      Numbers.reduceBigInt(BigInt.fromBigInteger(new BigInteger(m.group(2)))));
504                }
505        return null;
506}

545public static class StringReader extends AFn{
546        public Object invoke(Object reader, Object doublequote, Object opts, Object pendingForms) {
547                StringBuilder sb = new StringBuilder();
548                Reader r = (Reader) reader;
549
550                for(int ch = read1(r); ch != '"'; ch = read1(r))
551                        {
552                        if(ch == -1)
553                                throw Util.runtimeException("EOF while reading string");
554                        if(ch == '\\')  //escape
555                                {
556                                ch = read1(r);
557                                if(ch == -1)
558                                        throw Util.runtimeException("EOF while reading string");
559                                switch(ch)
560                                        {
561                                        case 't':
562                                                ch = '\t';
563                                                break;
564                                        case 'r':
565                                                ch = '\r';
566                                                break;
567                                        case 'n':
568                                                ch = '\n';
569                                                break;
570                                        case '\\':
571                                                break;
572                                        case '"':
573                                                break;
574                                        case 'b':
575                                                ch = '\b';
576                                                break;
577                                        case 'f':
578                                                ch = '\f';
579                                                break;
580                                        case 'u':
581                                        {
582                                        ch = read1(r);
583                                        if (Character.digit(ch, 16) == -1)
584                                                throw Util.runtimeException("Invalid unicode escape: \\u" + (char) ch);
585                                        ch = readUnicodeChar((PushbackReader) r, ch, 16, 4, true);
586                                        break;
587                                        }
588                                        default:
589                                        {
590                                        if(Character.isDigit(ch))
591                                                {
592                                                ch = readUnicodeChar((PushbackReader) r, ch, 8, 3, false);
593                                                if(ch > 0377)
594                                                        throw Util.runtimeException("Octal escape sequence must be in range [0, 377].");
595                                                }
596                                        else
597                                                throw Util.runtimeException("Unsupported escape character: \\" + (char) ch);
598                                        }
599                                        }
600                                }
601                        sb.append((char) ch);
602                        }
603                return sb.toString();
604        }

628// :a.b{:c 1} => {:a.b/c 1}
629// ::{:c 1}   => {:a.b/c 1}  (where *ns* = a.b)
630// ::a{:c 1}  => {:a.b/c 1}  (where a is aliased to a.b)
631public static class NamespaceMapReader extends AFn{
632        public Object invoke(Object reader, Object colon, Object opts, Object pendingForms) {
633                PushbackReader r = (PushbackReader) reader;
634
635                boolean auto = false;
636                int autoChar = read1(r);
637                if(autoChar == ':')
638                        auto = true;
639                else
640                    unread(r, autoChar);
641
642                Object sym = null;
643                int nextChar = read1(r);
644                if(isWhitespace(nextChar)) {  // the #:: { } case or an error
645                        if(auto) {
646                                while (isWhitespace(nextChar))
647                                        nextChar = read1(r);
648                                if(nextChar != '{') {
649                                        unread(r, nextChar);
650                                        throw Util.runtimeException("Namespaced map must specify a namespace");
651                                }
652                        } else {
653                                unread(r, nextChar);
654                                throw Util.runtimeException("Namespaced map must specify a namespace");
655                        }
656                } else if(nextChar != '{') {  // #:foo { } or #::foo { }
657                        unread(r, nextChar);
658                        sym = read(r, true, null, false, opts, pendingForms);
659                        nextChar = read1(r);
660                        while(isWhitespace(nextChar))
661                                nextChar = read1(r);
662                }
663                if(nextChar != '{')
664                        throw Util.runtimeException("Namespaced map must specify a map");
665
666                // Resolve autoresolved ns
667                String ns;
668                if (auto) {
669            Resolver resolver = (Resolver) RT.READER_RESOLVER.deref();
670                        if (sym == null) {
671                if(resolver != null)
672                    ns = resolver.currentNS().name;
673                else

691                        }
692                } else if (!(sym instanceof Symbol) || ((Symbol)sym).getNamespace() != null) {

694                } else {
695                        ns = ((Symbol)sym).getName();
696                }

699                List kvs = readDelimitedList('}', r, true, opts, ensurePending(pendingForms));
700                if((kvs.size() & 1) == 1)
701                        throw Util.runtimeException("Namespaced map literal must contain an even number of forms");
702
703                // Construct output map
704                Object[] a = new Object[kvs.size()];
705                Iterator iter = kvs.iterator();
706                for(int i = 0; iter.hasNext(); i += 2) {
707                        Object key = iter.next();
708                        Object val = iter.next();
709
710                        if(key instanceof Keyword) {
711                                Keyword kw = (Keyword) key;
712                                if (kw.getNamespace() == null) {
713                                        key = Keyword.intern(ns, kw.getName());
714                                } else if (kw.getNamespace().equals("_")) {
715                                        key = Keyword.intern(null, kw.getName());
716                                }
717                        } else if(key instanceof Symbol) {
718                                Symbol s = (Symbol) key;
719                                if (s.getNamespace() == null) {
720                                        key = Symbol.intern(ns, s.getName());
721                                } else if (s.getNamespace().equals("_")) {
722                                        key = Symbol.intern(null, s.getName());
723                                }
724                        }
725                        a[i] = key;
726                        a[i+1] = val;
727                }
728                return RT.map(a);

1015                                {

1025                                }
1026                        else if(sym.ns == null && sym.name.startsWith("."))
1027                                {
1028                                // Simply quote method names.
1029                                }
1030            else if(resolver != null)
1031                {
1032                Symbol nsym = null;
1033                if(sym.ns != null){
1034                    Symbol alias = Symbol.intern(null, sym.ns);
1035                    nsym = resolver.resolveClass(alias);

1068                                }
1069                        ret = RT.list(Compiler.QUOTE, sym);
1070                        }
1071                else if(isUnquote(form))
1072                        return RT.second(form);
1073                else if(isUnquoteSplicing(form))
1074                        throw new IllegalStateException("splice not in list");

1104                        || form instanceof Number
1105                        || form instanceof Character

1500public static class ConditionalReader extends AFn {
1501
1502        final static private Object READ_STARTED = new Object();
1503        final static public Keyword DEFAULT_FEATURE = Keyword.intern(null, "default");
1504        final static public IPersistentSet RESERVED_FEATURES =
1505                RT.set(Keyword.intern(null, "else"), Keyword.intern(null, "none"));
1506
1507        public static boolean hasFeature(Object feature, Object opts) {
1508                if (! (feature instanceof Keyword))
1509                        throw Util.runtimeException("Feature should be a keyword: " + feature);
1510
1511                if(DEFAULT_FEATURE.equals(feature))
1512                        return true;
1513
1514                IPersistentSet custom = (IPersistentSet) ((IPersistentMap)opts).valAt(OPT_FEATURES);
1515                return custom != null && custom.contains(feature);
1516        }
1517
1518        public static Object readCondDelimited(PushbackReader r, boolean splicing, Object opts, Object pendingForms) {
1519                Object result = READ_STARTED;
1520                Object form; // The most recently ready form
1521                boolean toplevel = (pendingForms == null);
1522                pendingForms = ensurePending(pendingForms);
1523
1524                final int firstline =
1525                                (r instanceof LineNumberingPushbackReader) ?
1526                                                ((LineNumberingPushbackReader) r).getLineNumber() : -1;
1527
1528                for(; ;) {
1529                        if(result == READ_STARTED) {
1530                                // Read the next feature
1531                                form = read(r, false, READ_EOF, ')', READ_FINISHED, true, opts, pendingForms, null);
1532
1533                                if (form == READ_EOF) {
1534                                        if (firstline < 0)
1535                                                throw Util.runtimeException("EOF while reading");
1536                                        else
1537                                                throw Util.runtimeException("EOF while reading, starting at line " + firstline);
1538                                } else if (form == READ_FINISHED) {
1539                                        break; // read-cond form is done
1540                                }
1541
1542                                if(RESERVED_FEATURES.contains(form))
1543                                        throw Util.runtimeException("Feature name " + form + " is reserved.");
1544
1545                                if (hasFeature(form, opts)) {
1546
1547                                        //Read the form corresponding to the feature, and assign it to result if everything is kosher
1548
1549                                        form = read(r, false, READ_EOF, ')', READ_FINISHED, true, opts, pendingForms, (Resolver) RT.READER_RESOLVER.deref());
1550
1551                                        if (form == READ_EOF) {
1552                                                if (firstline < 0)
1553                                                        throw Util.runtimeException("EOF while reading");
1554                                                else
1555                                                        throw Util.runtimeException("EOF while reading, starting at line " + firstline);
1556                                        } else if (form == READ_FINISHED) {
1557                                                if (firstline < 0)
1558                                                        throw Util.runtimeException("read-cond requires an even number of forms.");
1559                                                else
1560                                                        throw Util.runtimeException("read-cond starting on line " + firstline + " requires an even number of forms");
1561                                        } else {
1562                                                result = form;
1563                                        }
1564                                }
1565                        }
1566
1567                        // When we already have a result, or when the feature didn't match, discard the next form in the reader
1568                        try {
1569                                Var.pushThreadBindings(RT.map(RT.SUPPRESS_READ, RT.T));
1570                                form = read(r, false, READ_EOF, ')', READ_FINISHED, true, opts, pendingForms, (Resolver) RT.READER_RESOLVER.deref());
1571
1572                                if (form == READ_EOF) {
1573                                        if (firstline < 0)
1574                                                throw Util.runtimeException("EOF while reading");
1575                                        else
1576                                                throw Util.runtimeException("EOF while reading, starting at line " + firstline);
1577                                } else if (form == READ_FINISHED) {
1578                                        break;
1579                                }
1580                        }
1581                        finally {
1582                                Var.popThreadBindings();
1583                        }
1584
1585                }
1586
1587                if (result == READ_STARTED)  // no features matched
1588            return r;
1589
1590                if (splicing) {
1591                        if(! (result instanceof List))
1592                                throw Util.runtimeException("Spliced form list in read-cond-splicing must implement java.util.List");
1593
1594                        if(toplevel)
1595                                throw Util.runtimeException("Reader conditional splicing not allowed at the top level.");
1596
1597                        ((List)pendingForms).addAll(0, (List)result);
1598
1599                        return r;
1600                } else {
1601                        return result;
1602                }
1603        };

1605    private static void checkConditionalAllowed(Object opts) {
1606        IPersistentMap mopts = (IPersistentMap)opts;
1607        if(! (opts != null && (COND_ALLOW.equals(mopts.valAt(OPT_READ_COND)) ||
1608                               COND_PRESERVE.equals(mopts.valAt(OPT_READ_COND)))))
1609            throw Util.runtimeException("Conditional read not allowed");
1610    }

1612        public Object invoke(Object reader, Object mode, Object opts, Object pendingForms) {
1613                checkConditionalAllowed(opts);
1614
1615                PushbackReader r = (PushbackReader) reader;
1616                int ch = read1(r);
1617                if (ch == -1)
1618                        throw Util.runtimeException("EOF while reading character");
1619
1620                boolean splicing = false;
1621
1622                if (ch == '@') {
1623                        splicing = true;
1624                        ch = read1(r);
1625                }
1626
1627                while(isWhitespace(ch))
1628                        ch = read1(r);
1629
1630                if (ch == -1)
1631                        throw Util.runtimeException("EOF while reading character");
1632
1633                if(ch != '(')
1634                        throw Util.runtimeException("read-cond body must be a list");
1635
1636                try {
1637                        Var.pushThreadBindings(RT.map(READ_COND_ENV, RT.T));
1638
1639                        if (isPreserveReadCond(opts)) {
1640                                IFn listReader = getMacro(ch); // should always be a list
1641                                Object form = listReader.invoke(r, ch, opts, ensurePending(pendingForms));
1642

1644                        } else {
1645                                return readCondDelimited(r, splicing, opts, pendingForms);
1646                        }
1647                } finally {

1649                }
1650        }
