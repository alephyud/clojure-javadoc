013package clojure.lang;
014
015import org.xml.sax.Attributes;
016import org.xml.sax.ContentHandler;
017import org.xml.sax.Locator;
018import org.xml.sax.SAXException;
019import org.xml.sax.helpers.DefaultHandler;
020
021public class XMLHandler extends DefaultHandler{
022ContentHandler h;
023
024
025public XMLHandler(ContentHandler h){
026        this.h = h;
027}
028
029public void setDocumentLocator(Locator locator){
030        h.setDocumentLocator(locator);
031}
032
033public void startDocument() throws SAXException{
034        h.startDocument();
035}
036
037public void endDocument() throws SAXException{
038        h.endDocument();
039}
040
041public void startPrefixMapping(String prefix, String uri) throws SAXException{
042        h.startPrefixMapping(prefix, uri);
043}
044
045public void endPrefixMapping(String prefix) throws SAXException{
046        h.endPrefixMapping(prefix);
047}
048
049public void startElement(String uri, String localName, String qName, Attributes atts) throws SAXException{
050        h.startElement(uri, localName, qName, atts);
051}
052
053public void endElement(String uri, String localName, String qName) throws SAXException{
054        h.endElement(uri, localName, qName);
055}
056
057public void characters(char ch[], int start, int length) throws SAXException{
058        h.characters(ch, start, length);
059}
060
061public void ignorableWhitespace(char ch[], int start, int length) throws SAXException{
062        h.ignorableWhitespace(ch, start, length);
063}
064
065public void processingInstruction(String target, String data) throws SAXException{
066        h.processingInstruction(target, data);
067}
068
069public void skippedEntity(String name) throws SAXException{
070        h.skippedEntity(name);
071}
