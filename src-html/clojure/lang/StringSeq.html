013package clojure.lang;
014
015public class StringSeq extends ASeq implements IndexedSeq{
016public final CharSequence s;
017public final int i;
018
019static public StringSeq create(CharSequence s){
020        if(s.length() == 0)
021                return null;
022        return new StringSeq(null, s, 0);
023}
024
025StringSeq(IPersistentMap meta, CharSequence s, int i){
026        super(meta);
027        this.s = s;
028        this.i = i;
029}
030
031public Obj withMeta(IPersistentMap meta){
032        if(meta == meta())
033                return this;
034        return new StringSeq(meta, s, i);
035}
036
037public Object first(){
038        return Character.valueOf(s.charAt(i));
039}
040
041public ISeq next(){
042        if(i + 1 < s.length())
043                return new StringSeq(_meta, s, i + 1);
044        return null;
045}
046
047public int index(){
048        return i;
049}
050
051public int count(){
052        return s.length() - i;
053}
054}
