256public static List<String> processCommandLine(String[] args){
257        List<String> arglist = Arrays.asList(args);
258        int split = arglist.indexOf("--");
259        if(split >= 0) {
260                CMD_LINE_ARGS.bindRoot(RT.seq(arglist.subList(split + 1, args.length)));
261                return arglist.subList(0, split);
262        }
263        return arglist;
264}

373public static void loadResourceScript(Class c, String name, boolean failIfNotFound) throws IOException{
374        int slash = name.lastIndexOf('/');
375        String file = slash >= 0 ? name.substring(slash + 1) : name;
376        InputStream ins = resourceAsStream(baseLoader(), name);
377        if(ins != null) {
378                try {
379                        Compiler.load(new InputStreamReader(ins, UTF8), name, file);
380                }
381                finally {
382                        ins.close();
383                }
384        }
385        else if(failIfNotFound) {
386                throw new FileNotFoundException("Could not locate Clojure resource on classpath: " + name);
387        }
388}

394static public long lastModified(URL url, String libfile) throws IOException{
395        URLConnection connection = url.openConnection();
396        try {
397                if (url.getProtocol().equals("jar"))
398                        return ((JarURLConnection) connection).getJarFile().getEntry(libfile).getTime();
399                else
400                        return connection.getLastModified();
401        }
402        finally {
403                InputStream ins = connection.getInputStream();
404                if (ins != null)
405                        ins.close();
406        }

429static public void load(String scriptbase, boolean failIfNotFound) throws IOException, ClassNotFoundException{
430        String classfile = scriptbase + LOADER_SUFFIX + ".class";
431        String cljfile = scriptbase + ".clj";
432        String cljcfile = scriptbase + ".cljc";
433        String scriptfile = cljfile;
434        URL classURL = getResource(baseLoader(),classfile);
435        URL cljURL = getResource(baseLoader(), scriptfile);
436        if(cljURL == null) {
437                scriptfile = cljcfile;
438                cljURL = getResource(baseLoader(), scriptfile);
439        }
440        boolean loaded = false;
441
442        if((classURL != null &&
443            (cljURL == null
444             || lastModified(classURL, classfile) > lastModified(cljURL, scriptfile)))
445           || classURL == null) {
446                try {
447                        Var.pushThreadBindings(
448                                        RT.mapUniqueKeys(CURRENT_NS, CURRENT_NS.deref(),
449                                               WARN_ON_REFLECTION, WARN_ON_REFLECTION.deref()
450                                                        ,RT.UNCHECKED_MATH, RT.UNCHECKED_MATH.deref()));
451                        loaded = (loadClassForName(scriptbase.replace('/', '.') + LOADER_SUFFIX) != null);
452                }
453                finally {
454                        Var.popThreadBindings();
455                }
456        }
457        if(!loaded && cljURL != null) {
458                if(booleanCast(Compiler.COMPILE_FILES.deref()))
459                        compile(scriptfile);
460                else
461                        loadResourceScript(RT.class, scriptfile);
462        }
463        else if(!loaded && failIfNotFound)
464                throw new FileNotFoundException(String.format("Could not locate %s, %s or %s on classpath.%s", classfile, cljfile, cljcfile,
465                        scriptbase.contains("_") ? " Please check that namespaces with dashes use underscores in the Clojure file name." : ""));
466}

468static void doInit() throws ClassNotFoundException, IOException{
469        load("clojure/core");
470
471        Var.pushThreadBindings(
472                        RT.mapUniqueKeys(CURRENT_NS, CURRENT_NS.deref(),
473                               WARN_ON_REFLECTION, WARN_ON_REFLECTION.deref()
474                                        ,RT.UNCHECKED_MATH, RT.UNCHECKED_MATH.deref()));
475        try {
476                Symbol USER = Symbol.intern("user");
477                Symbol CLOJURE = Symbol.intern("clojure.core");
478
479                Var in_ns = var("clojure.core", "in-ns");
480                Var refer = var("clojure.core", "refer");
481                in_ns.invoke(USER);
482                refer.invoke(CLOJURE);
483                maybeLoadResourceScript("user.clj");
484
485                // start socket servers
486                Var require = var("clojure.core", "require");
487                Symbol SERVER = Symbol.intern("clojure.core.server");
488                require.invoke(SERVER);
489                Var start_servers = var("clojure.core.server", "start-servers");
490                start_servers.invoke(System.getProperties());
491        }
492        finally {
493                Var.popThreadBindings();
494        }
495}

509private static final int CHUNK_SIZE = 32;
510public static ISeq chunkIteratorSeq(final Iterator iter){
511    if(iter.hasNext()) {
512        return new LazySeq(new AFn() {
513            public Object invoke() {
514                Object[] arr = new Object[CHUNK_SIZE];
515                int n = 0;
516                while(iter.hasNext() && n < CHUNK_SIZE)
517                    arr[n++] = iter.next();
518                return new ChunkedCons(new ArrayChunk(arr, 0, n), chunkIteratorSeq(iter));
519            }
520        });
521    }
522    return null;
523}

525static public ISeq seq(Object coll){
526        if(coll instanceof ASeq)
527                return (ASeq) coll;
528        else if(coll instanceof LazySeq)
529                return ((LazySeq) coll).seq();
530        else
531                return seqFrom(coll);
532}

534// N.B. canSeq must be kept in sync with this!
535static ISeq seqFrom(Object coll){
536        if(coll instanceof Seqable)
537                return ((Seqable) coll).seq();
538        else if(coll == null)
539                return null;
540        else if(coll instanceof Iterable)
541                return chunkIteratorSeq(((Iterable) coll).iterator());
542        else if(coll.getClass().isArray())
543                return ArraySeq.createFromObject(coll);
544        else if(coll instanceof CharSequence)
545                return StringSeq.create((CharSequence) coll);
546        else if(coll instanceof Map)
547                return seq(((Map) coll).entrySet());
548        else {
549                Class c = coll.getClass();
550                Class sc = c.getSuperclass();
551                throw new IllegalArgumentException("Don't know how to create ISeq from: " + c.getName());
552        }
553}

656        else if(o instanceof Collection)

662        else if(o.getClass().isArray())
663                return Array.getLength(o);

665        throw new UnsupportedOperationException("count not supported on this type: " + o.getClass().getSimpleName());

894static Object nthFrom(Object coll, int n){
895        if(coll == null)
896                return null;
897        else if(coll instanceof CharSequence)
898                return Character.valueOf(((CharSequence) coll).charAt(n));
899        else if(coll.getClass().isArray())
900                return Reflector.prepRet(coll.getClass().getComponentType(),Array.get(coll, n));
901        else if(coll instanceof RandomAccess)
902                return ((List) coll).get(n);
903        else if(coll instanceof Matcher)
904                return ((Matcher) coll).group(n);
905
906        else if(coll instanceof Map.Entry) {
907                Map.Entry e = (Map.Entry) coll;
908                if(n == 0)
909                        return e.getKey();
910                else if(n == 1)
911                        return e.getValue();
912                throw new IndexOutOfBoundsException();
913        }
914
915        else if(coll instanceof Sequential) {
916                ISeq seq = RT.seq(coll);
917                coll = null;
918                for(int i = 0; i <= n && seq != null; ++i, seq = seq.next()) {
919                        if(i == n)
920                                return seq.first();
921                }
922                throw new IndexOutOfBoundsException();
923        }
924        else
925                throw new UnsupportedOperationException(
926                                "nth not supported on this type: " + coll.getClass().getSimpleName());
927}

976                coll = null;

2095static public ISeq doFormat(Writer w, String s, ISeq args) throws IOException{
2096        for(int i = 0; i < s.length();) {
2097                char c = s.charAt(i++);
2098                switch(Character.toLowerCase(c)) {
2099                        case '~':

2101                                switch(Character.toLowerCase(d)) {
2102                                        case '%':
2103                                                w.write('\n');
2104                                                break;
2105                                        case 't':
2106                                                w.write('\t');
2107                                                break;
2108                                        case 'a':
2109                                                if(args == null)
2110                                                        throw new IllegalArgumentException("Missing argument");
2111                                                RT.formatAesthetic(w, RT.first(args));

2113                                                break;
2114                                        case 's':

2116                                                        throw new IllegalArgumentException("Missing argument");
2117                                                RT.formatStandard(w, RT.first(args));

2119                                                break;
2120                                        case '{':
2121                                                int j = s.indexOf("~}", i);    //note - does not nest
2122                                                if(j == -1)
2123                                                        throw new IllegalArgumentException("Missing ~}");
2124                                                String subs = s.substring(i, j);
2125                                                for(ISeq sargs = RT.seq(RT.first(args)); sargs != null;)
2126                                                        sargs = doFormat(w, subs, sargs);
2127                                                args = RT.next(args);
2128                                                i = j + 2; //skip ~}
2129                                                break;
2130                                        case '^':
2131                                                if(args == null)
2132                                                        return null;
2133                                                break;
2134                                        case '~':
2135                                                w.write('~');

2138                                                throw new IllegalArgumentException("Unsupported ~ directive: " + d);
2139                                }
2140                                break;
2141                        default:
