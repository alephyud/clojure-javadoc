025public static Object invokeInstanceMethod(Object target, String methodName, Object[] args) {
026        Class c = target.getClass();
027        List methods = getMethods(c, args.length, methodName, false);
028        return invokeMatchingMethod(methodName, methods, target, args);
029}

031private static Throwable getCauseOrElse(Exception e) {
032        if (e.getCause() != null)
033                return e.getCause();
034        return e;
035}

037private static RuntimeException throwCauseOrElseException(Exception e) {
038        if (e.getCause() != null)
039                throw Util.sneakyThrow(e.getCause());
040        throw Util.sneakyThrow(e);

043private static String noMethodReport(String methodName, Object target, Object[] args){
044         return "No matching method " + methodName + " found taking " + args.length + " args"
045                        + (target==null?"":" for " + target.getClass());
046}
047static Object invokeMatchingMethod(String methodName, List methods, Object target, Object[] args)
048                {
049        Method m = null;
050        Object[] boxedArgs = null;
051        if(methods.isEmpty())
052                {
053                throw new IllegalArgumentException(noMethodReport(methodName,target,args));
054                }
055        else if(methods.size() == 1)
056                {
057                m = (Method) methods.get(0);
058                boxedArgs = boxArgs(m.getParameterTypes(), args);
059                }
060        else //overloaded w/same arity
061                {
062                Method foundm = null;
063                for(Iterator i = methods.iterator(); i.hasNext();)
064                        {
065                        m = (Method) i.next();
066
067                        Class[] params = m.getParameterTypes();
068                        if(isCongruent(params, args))
069                                {
070                                if(foundm == null || Compiler.subsumes(params, foundm.getParameterTypes()))
071                                        {
072                                        foundm = m;
073                                        boxedArgs = boxArgs(params, args);
074                                        }
075                                }
076                        }
077                m = foundm;
078                }
079        if(m == null)
080                throw new IllegalArgumentException(noMethodReport(methodName,target,args));
081
082        if(!Modifier.isPublic(m.getDeclaringClass().getModifiers()))
083                {
084                //public method of non-public class, try to find it in hierarchy
085                Method oldm = m;
086                m = getAsMethodOfPublicBase(target.getClass(), m);
087                if(m == null)
088                        throw new IllegalArgumentException("Can't call public method of non-public class: " +
089                                                            oldm.toString());
090                }
091        try
092                {
093                return prepRet(m.getReturnType(), m.invoke(target, boxedArgs));
094                }
095        catch(Exception e)
096                {
097                throw Util.sneakyThrow(getCauseOrElse(e));
098                }
099
100}

102public static Method getAsMethodOfPublicBase(Class c, Method m){
103        for(Class iface : c.getInterfaces())
104                {
105                for(Method im : iface.getMethods())
106                        {
107                        if(isMatch(im, m))
108                                {
109                                return im;
110                                }
111                        }
112                }
113        Class sc = c.getSuperclass();
114        if(sc == null)
115                return null;
116        for(Method scm : sc.getMethods())
117                {
118                if(isMatch(scm, m))
119                        {
120                        return scm;
121                        }
122                }
123        return getAsMethodOfPublicBase(sc, m);
124}

126public static boolean isMatch(Method lhs, Method rhs) {
127        if(!lhs.getName().equals(rhs.getName())
128                        || !Modifier.isPublic(lhs.getDeclaringClass().getModifiers()))
129                {
130                return false;
131                }
132
133                Class[] types1 = lhs.getParameterTypes();
134                Class[] types2 = rhs.getParameterTypes();
135                if(types1.length != types2.length)
136                        return false;
137
138                boolean match = true;
139                for (int i=0; i<types1.length; ++i)
140                        {
141                        if(!types1[i].isAssignableFrom(types2[i]))
142                                {
143                                match = false;
144                                break;
145                                }
146                        }
147                return match;
148}

150public static Object invokeConstructor(Class c, Object[] args) {
151        try
152                {
153                Constructor[] allctors = c.getConstructors();
154                ArrayList ctors = new ArrayList();
155                for(int i = 0; i < allctors.length; i++)
156                        {
157                        Constructor ctor = allctors[i];
158                        if(ctor.getParameterTypes().length == args.length)
159                                ctors.add(ctor);
160                        }
161                if(ctors.isEmpty())
162                        {
163                        throw new IllegalArgumentException("No matching ctor found"
164                                + " for " + c);
165                        }
166                else if(ctors.size() == 1)
167                        {
168                        Constructor ctor = (Constructor) ctors.get(0);
169                        return ctor.newInstance(boxArgs(ctor.getParameterTypes(), args));
170                        }
171                else //overloaded w/same arity
172                        {
173                        for(Iterator iterator = ctors.iterator(); iterator.hasNext();)
174                                {
175                                Constructor ctor = (Constructor) iterator.next();
176                                Class[] params = ctor.getParameterTypes();
177                                if(isCongruent(params, args))
178                                        {
179                                        Object[] boxedArgs = boxArgs(params, args);
180                                        return ctor.newInstance(boxedArgs);
181                                        }
182                                }
183                        throw new IllegalArgumentException("No matching ctor found"
184                                + " for " + c);
185                        }
186                }
187        catch(Exception e)
188                {
189                throw Util.sneakyThrow(getCauseOrElse(e));
190                }
191}

203public static Object invokeStaticMethod(Class c, String methodName, Object[] args) {
204        if(methodName.equals("new"))
205                return invokeConstructor(c, args);
206        List methods = getMethods(c, args.length, methodName, true);
207        return invokeMatchingMethod(methodName, methods, null, args);
208}

215public static Object getStaticField(Class c, String fieldName) {
216//      if(fieldName.equals("class"))
217//              return c;
218        Field f = getField(c, fieldName, true);
219        if(f != null)
220                {
221                try
222                        {
223                        return prepRet(f.getType(), f.get(null));
224                        }
225                catch(IllegalAccessException e)
226                        {
227                        throw Util.sneakyThrow(e);
228                        }
229                }
230        throw new IllegalArgumentException("No matching field found: " + fieldName
231                + " for " + c);
232}

246                        }
247                catch(IllegalAccessException e)
248                        {
249                        throw Util.sneakyThrow(e);
250                        }
251                return val;

300public static Object invokeNoArgInstanceMember(Object target, String name, boolean requireField) {
301        Class c = target.getClass();
302
303        if(requireField) {
304                Field f = getField(c, name, false);
305                if(f != null)
306                        return getInstanceField(target, name);
307                else
308                        throw new IllegalArgumentException("No matching field found: " + name
309                                        + " for " + target.getClass());
310        } else {
311                List meths = getMethods(c, 0, name, false);
312                if(meths.size() > 0)
313                        return invokeMatchingMethod(name, meths, target, RT.EMPTY_ARRAY);
314                else
315                        return getInstanceField(target, name);
316        }
317}

361static public Field getField(Class c, String name, boolean getStatics){
362        Field[] allfields = c.getFields();
363        for(int i = 0; i < allfields.length; i++)
364                {
365                if(name.equals(allfields[i].getName())
366                   && Modifier.isStatic(allfields[i].getModifiers()) == getStatics)
367                        return allfields[i];
368                }
369        return null;
370}

372static public List getMethods(Class c, int arity, String name, boolean getStatics){
373        Method[] allmethods = c.getMethods();
374        ArrayList methods = new ArrayList();
375        ArrayList bridgeMethods = new ArrayList();
376        for(int i = 0; i < allmethods.length; i++)
377                {
378                Method method = allmethods[i];
379                if(name.equals(method.getName())
380                   && Modifier.isStatic(method.getModifiers()) == getStatics
381                   && method.getParameterTypes().length == arity)
382                        {
383                        try
384                                {
385                                if(method.isBridge()
386                                   && c.getMethod(method.getName(), method.getParameterTypes())
387                                                .equals(method))
388                                        bridgeMethods.add(method);
389                                else
390                                        methods.add(method);
391                                }
392                        catch(NoSuchMethodException e)
393                                {
394                                }
395                        }
396//                         && (!method.isBridge()
397//                             || (c == StringBuilder.class &&
398//                                c.getMethod(method.getName(), method.getParameterTypes())
399//                                      .equals(method))))
400//                              {
401//                              methods.add(allmethods[i]);
402//                              }
403                }
404
405        if(methods.isEmpty())
406                methods.addAll(bridgeMethods);
407        
408        if(!getStatics && c.isInterface())
409                {
410                allmethods = Object.class.getMethods();
411                for(int i = 0; i < allmethods.length; i++)
412                        {
413                        if(name.equals(allmethods[i].getName())
414                           && Modifier.isStatic(allmethods[i].getModifiers()) == getStatics
415                           && allmethods[i].getParameterTypes().length == arity)
416                                {
417                                methods.add(allmethods[i]);
418                                }
419                        }
420                }
421        return methods;
422}

425static Object boxArg(Class paramType, Object arg){
426        if(!paramType.isPrimitive())
427                return paramType.cast(arg);
428        else if(paramType == boolean.class)
429                return Boolean.class.cast(arg);
430        else if(paramType == char.class)
431                return Character.class.cast(arg);
432        else if(arg instanceof Number)
433                {
434                Number n = (Number) arg;
435                if(paramType == int.class)
436                        return n.intValue();
437                else if(paramType == float.class)
438                        return n.floatValue();
439                else if(paramType == double.class)
440                        return n.doubleValue();
441                else if(paramType == long.class)
442                        return n.longValue();
443                else if(paramType == short.class)
444                        return n.shortValue();
445                else if(paramType == byte.class)
446                        return n.byteValue();
447                }
448        throw new IllegalArgumentException("Unexpected param type, expected: " + paramType +
449                                           ", given: " + arg.getClass().getName());
450}
451
452static Object[] boxArgs(Class[] params, Object[] args){
453        if(params.length == 0)
454                return null;
455        Object[] ret = new Object[params.length];
456        for(int i = 0; i < params.length; i++)
457                {
458                Object arg = args[i];
459                Class paramType = params[i];
460                ret[i] = boxArg(paramType, arg);
461                }
462        return ret;
463}

465static public boolean paramArgTypeMatch(Class paramType, Class argType){
466        if(argType == null)
467                return !paramType.isPrimitive();
468        if(paramType == argType || paramType.isAssignableFrom(argType))
469                return true;
470        if(paramType == int.class)
471                return argType == Integer.class
472                       || argType == long.class
473                                || argType == Long.class
474                                || argType == short.class
475                                || argType == byte.class;// || argType == FixNum.class;
476        else if(paramType == float.class)
477                return argType == Float.class
478                                || argType == double.class;
479        else if(paramType == double.class)
480                return argType == Double.class
481                                || argType == float.class;// || argType == DoubleNum.class;
482        else if(paramType == long.class)
483                return argType == Long.class
484                                || argType == int.class
485                                || argType == short.class
486                                || argType == byte.class;// || argType == BigNum.class;
487        else if(paramType == char.class)
488                return argType == Character.class;
489        else if(paramType == short.class)
490                return argType == Short.class;
491        else if(paramType == byte.class)
492                return argType == Byte.class;
493        else if(paramType == boolean.class)
494                return argType == Boolean.class;
495        return false;
496}

498static boolean isCongruent(Class[] params, Object[] args){
499        boolean ret = false;
500        if(args == null)
501                return params.length == 0;
502        if(params.length == args.length)
503                {
504                ret = true;
505                for(int i = 0; ret && i < params.length; i++)
506                        {
507                        Object arg = args[i];
508                        Class argType = (arg == null) ? null : arg.getClass();
509                        Class paramType = params[i];
510                        ret = paramArgTypeMatch(paramType, argType);
511                        }
512                }
513        return ret;
514}

516public static Object prepRet(Class c, Object x){
517        if (!(c.isPrimitive() || c == Boolean.class))
518                return x;
519        if(x instanceof Boolean)
520                return ((Boolean) x)?Boolean.TRUE:Boolean.FALSE;
521//      else if(x instanceof Integer)
522//              {
523//              return ((Integer)x).longValue();
524//              }
525//      else if(x instanceof Float)
526//                      return Double.valueOf(((Float) x).doubleValue());
527        return x;
528}
