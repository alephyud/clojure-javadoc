011package clojure.lang;
012
013/**
014 * Exception that carries data (a map) as additional payload. Clojure programs that need
015 * richer semantics for exceptions should use this in lieu of defining project-specific
016 * exception classes.
017 */
018public class ExceptionInfo extends RuntimeException implements IExceptionInfo {
019    public final IPersistentMap data;
020
021    public ExceptionInfo(String s, IPersistentMap data) {
022        this(s, data, null);
023    }
024
025    public ExceptionInfo(String s, IPersistentMap data, Throwable throwable) {
026        // null cause is equivalent to not passing a cause
027        super(s, throwable);
028        if (data != null) {
029            this.data = data;
030        }  else {
031            throw new IllegalArgumentException("Additional data must be non-nil.");
032        }
033    }
034
035    public IPersistentMap getData() {
036        return data;
037    }
038
039    public String toString() {
040        return "clojure.lang.ExceptionInfo: " + getMessage() + " " + data.toString();
041    }
042}
