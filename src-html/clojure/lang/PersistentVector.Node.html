070static public PersistentVector create(ISeq items){
071    Object[] arr = new Object[32];
072    int i = 0;
073    for(;items != null && i < 32; items = items.next())
074        arr[i++] = items.first();
075
076    if(items != null) {  // >32, construct with array directly
077        PersistentVector start = new PersistentVector(32, 5, EMPTY_NODE, arr);
078        TransientVector ret = start.asTransient();
079        for (; items != null; items = items.next())
080            ret = ret.conj(items.first());
081        return ret.persistent();
082    } else if(i == 32) {   // exactly 32, skip copy
083        return new PersistentVector(32, 5, EMPTY_NODE, arr);
084    } else {  // <32, copy to minimum array and construct
085        Object[] arr2 = new Object[i];
086        System.arraycopy(arr, 0, arr2, 0, i);
087        return new PersistentVector(i, 5, EMPTY_NODE, arr2);
088    }
089}

091static public PersistentVector create(List list){
092    int size = list.size();
093    if (size <= 32)
094        return new PersistentVector(size, 5, PersistentVector.EMPTY_NODE, list.toArray());
095
096    TransientVector ret = EMPTY.asTransient();
097    for(int i=0; i<size; i++)
098        ret = ret.conj(list.get(i));
099    return ret.persistent();
100}

102static public PersistentVector create(Iterable items){
103    // optimize common case
104    if(items instanceof ArrayList)
105        return create((ArrayList)items);
106
107    Iterator iter = items.iterator();
108    TransientVector ret = EMPTY.asTransient();
109    while(iter.hasNext())
110        ret = ret.conj(iter.next());
111    return ret.persistent();
112}

142final int tailoff(){
143        if(cnt < 32)
144                return 0;
145        return ((cnt - 1) >>> 5) << 5;

148public Object[] arrayFor(int i){
149        if(i >= 0 && i < cnt)
150                {
151                if(i >= tailoff())
152                        return tail;
153                Node node = root;
154                for(int level = shift; level > 0; level -= 5)
155                        node = (Node) node.array[(i >>> level) & 0x01f];
156                return node.array;
157                }
158        throw new IndexOutOfBoundsException();
159}

172public PersistentVector assocN(int i, Object val){
173        if(i >= 0 && i < cnt)
174                {
175                if(i >= tailoff())
176                        {
177                        Object[] newTail = new Object[tail.length];
178                        System.arraycopy(tail, 0, newTail, 0, tail.length);
179                        newTail[i & 0x01f] = val;
180
181                        return new PersistentVector(meta(), cnt, shift, root, newTail);
182                        }
183
184                return new PersistentVector(meta(), cnt, shift, doAssoc(shift, root, i, val), tail);
185                }
186        if(i == cnt)
187                return cons(val);
188        throw new IndexOutOfBoundsException();
189}

191private static Node doAssoc(int level, Node node, int i, Object val){
192        Node ret = new Node(node.edit,node.array.clone());
193        if(level == 0)
194                {
195                ret.array[i & 0x01f] = val;
196                }
197        else
198                {
199                int subidx = (i >>> level) & 0x01f;
200                ret.array[subidx] = doAssoc(level - 5, (Node) node.array[subidx], i, val);
201                }
202        return ret;
203}

220public PersistentVector cons(Object val){
221        //room in tail?
222//      if(tail.length < 32)
223        if(cnt - tailoff() < 32)
224                {
225                Object[] newTail = new Object[tail.length + 1];
226                System.arraycopy(tail, 0, newTail, 0, tail.length);
227                newTail[tail.length] = val;
228                return new PersistentVector(meta(), cnt + 1, shift, root, newTail);
229                }
230        //full tail, push into tree
231        Node newroot;
232        Node tailnode = new Node(root.edit,tail);
233        int newshift = shift;
234        //overflow root?
235        if((cnt >>> 5) > (1 << shift))
236                {
237                newroot = new Node(root.edit);
238                newroot.array[0] = root;
239                newroot.array[1] = newPath(root.edit,shift, tailnode);
240                newshift += 5;
241                }
242        else
243                newroot = pushTail(shift, root, tailnode);
244        return new PersistentVector(meta(), cnt + 1, newshift, newroot, new Object[]{val});
245}

247private Node pushTail(int level, Node parent, Node tailnode){
248        //if parent is leaf, insert node,
249        // else does it map to an existing child? -> nodeToInsert = pushNode one more level
250        // else alloc new path
251        //return  nodeToInsert placed in copy of parent
252        int subidx = ((cnt - 1) >>> level) & 0x01f;
253        Node ret = new Node(parent.edit, parent.array.clone());
254        Node nodeToInsert;
255        if(level == 5)
256                {
257                nodeToInsert = tailnode;
258                }
259        else
260                {
261                Node child = (Node) parent.array[subidx];
262                nodeToInsert = (child != null)?
263                                pushTail(level-5,child, tailnode)
264                                :newPath(root.edit,level-5, tailnode);
265                }
266        ret.array[subidx] = nodeToInsert;
267        return ret;
268}

270private static Node newPath(AtomicReference<Thread> edit,int level, Node node){
271        if(level == 0)
272                return node;
273        Node ret = new Node(edit);
274        ret.array[0] = newPath(edit, level - 5, node);
275        return ret;
276}

288@Override
289Iterator rangedIterator(final int start, final int end){
290        return new Iterator(){
291                int i = start;
292                int base = i - (i%32);
293                Object[] array = (start < count())?arrayFor(i):null;
294
295                public boolean hasNext(){
296                        return i < end;
297                        }
298
299                public Object next(){
300                        if(i < end) {
301                                if(i-base == 32){
302                                        array = arrayFor(i);
303                                        base += 32;
304                                }
305                                return array[i++ & 0x01f];
306                        } else {
307                                throw new NoSuchElementException();
308                        }
309                }
310
311                public void remove(){
312                        throw new UnsupportedOperationException();
313                }
314        };
315}

319public Object reduce(IFn f){
320    Object init;
321    if (cnt > 0)
322        init = arrayFor(0)[0];
323    else
324        return f.invoke();
325    int step = 0;
326    for(int i=0;i<cnt;i+=step){
327        Object[] array = arrayFor(i);
328        for(int j = (i==0)?1:0;j<array.length;++j){
329            init = f.invoke(init,array[j]);
330            if(RT.isReduced(init))
331                    return ((IDeref)init).deref();
332            }
333        step = array.length;
334    }
335    return init;
336}

341        Object[] array = arrayFor(i);
342        for(int j =0;j<array.length;++j){
343            init = f.invoke(init,array[j]);

352public Object kvreduce(IFn f, Object init){
353    int step = 0;
354    for(int i=0;i<cnt;i+=step){
355        Object[] array = arrayFor(i);
356        for(int j =0;j<array.length;++j){
357            init = f.invoke(init,j+i,array[j]);
358            if(RT.isReduced(init))
359                    return ((IDeref)init).deref();
360            }
361        step = array.length;
362    }
363    return init;
364}

468public PersistentVector pop(){
469        if(cnt == 0)
470                throw new IllegalStateException("Can't pop empty vector");
471        if(cnt == 1)
472                return EMPTY.withMeta(meta());
473        //if(tail.length > 1)
474        if(cnt-tailoff() > 1)
475                {
476                Object[] newTail = new Object[tail.length - 1];
477                System.arraycopy(tail, 0, newTail, 0, newTail.length);
478                return new PersistentVector(meta(), cnt - 1, shift, root, newTail);
479                }
480        Object[] newtail = arrayFor(cnt - 2);
481
482        Node newroot = popTail(shift, root);
483        int newshift = shift;
484        if(newroot == null)
485                {
486                newroot = EMPTY_NODE;
487                }
488        if(shift > 5 && newroot.array[1] == null)
489                {
490                newroot = (Node) newroot.array[0];
491                newshift -= 5;
492                }
493        return new PersistentVector(meta(), cnt - 1, newshift, newroot, newtail);
494}

496private Node popTail(int level, Node node){
497        int subidx = ((cnt-2) >>> level) & 0x01f;
498        if(level > 5)
499                {
500                Node newchild = popTail(level - 5, (Node) node.array[subidx]);
501                if(newchild == null && subidx == 0)
502                        return null;
503                else
504                        {
505                        Node ret = new Node(root.edit, node.array.clone());
506                        ret.array[subidx] = newchild;
507                        return ret;
508                        }
509                }
510        else if(subidx == 0)
511                return null;
512        else
513                {
514                Node ret = new Node(root.edit, node.array.clone());
515                ret.array[subidx] = null;
516                return ret;
517                }
518}

579        public TransientVector conj(Object val){
580                ensureEditable();
581                int i = cnt;
582                //room in tail?
583                if(i - tailoff() < 32)
584                        {
585                        tail[i & 0x01f] = val;
586                        ++cnt;
587                        return this;
588                        }
589                //full tail, push into tree
590                Node newroot;
591                Node tailnode = new Node(root.edit, tail);
592                tail = new Object[32];
593                tail[0] = val;
594                int newshift = shift;
595                //overflow root?
596                if((cnt >>> 5) > (1 << shift))
597                        {
598                        newroot = new Node(root.edit);
599                        newroot.array[0] = root;
600                        newroot.array[1] = newPath(root.edit,shift, tailnode);
601                        newshift += 5;
602                        }
603                else
604                        newroot = pushTail(shift, root, tailnode);
605                root = newroot;
606                shift = newshift;
607                ++cnt;
608                return this;
609        }

757        public TransientVector pop(){
758                ensureEditable();
759                if(cnt == 0)
760                        throw new IllegalStateException("Can't pop empty vector");
761                if(cnt == 1)
762                        {
763                        cnt = 0;
764                        return this;
765                        }
766                int i = cnt - 1;
767                //pop in tail?
768                if((i & 0x01f) > 0)
769                        {
770                        --cnt;
771                        return this;
772                        }

783                        {
784                        newroot = ensureEditable((Node) newroot.array[0]);
785                        newshift -= 5;
786                        }
787                root = newroot;
788                shift = newshift;
789                --cnt;
790                tail = newtail;
791                return this;
792        }

794        private Node popTail(int level, Node node){
795                node = ensureEditable(node);
796                int subidx = ((cnt - 2) >>> level) & 0x01f;
797                if(level > 5)
798                        {
799                        Node newchild = popTail(level - 5, (Node) node.array[subidx]);
800                        if(newchild == null && subidx == 0)
801                                return null;
802                        else
803                                {
804                                Node ret = node;
805                                ret.array[subidx] = newchild;
806                                return ret;
807                                }
808                        }
809                else if(subidx == 0)
810                        return null;
811                else
812                        {
813                        Node ret = node;
814                        ret.array[subidx] = null;
815                        return ret;
816                        }
817        }
